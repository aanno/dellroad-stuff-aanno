008    package org.dellroad.stuff.spring;
009    
010    import java.lang.annotation.Documented;
011    import java.lang.annotation.ElementType;
012    import java.lang.annotation.Inherited;
013    import java.lang.annotation.Retention;
014    import java.lang.annotation.RetentionPolicy;
015    import java.lang.annotation.Target;
016    
017    import org.springframework.transaction.annotation.Propagation;
018    
019    /**
020     * Indicates how an {@link javax.persistence.EntityManager} should be opened for the duration of the annotated method.
021     * This works like Spring's {@link org.springframework.transaction.annotation.Transactional @Transactional} annotation
022     * except it applies to the thread-local {@link javax.persistence.EntityManager}.
023     *
024     * <p>
025     * This annotation is handy in setups where the container does not provide an {@link javax.persistence.EntityManager}.
026     *
027     * <p>
028     * To activate this annotation, an {@link EntityManagerAvailableAspect} bean must exist in the application context
029     * and <code>@AspectJ</code> support must be enabled. For example:
030     * <blockquote><pre>
031     *  &lt;!-- enable AOP support --&gt;
032     *  &lt;aop:aspectj-autoproxy/&gt;
033     *
034     *  &lt;!-- activate @EntityManagerAvailable annotations --&gt;
035     *  &lt;bean class="org.dellroad.stuff.spring.EntityManagerAvailableAspect"
036     *    entityManagerFactory-ref="myEntityManagerFactory"/&gt;
037     * </pre></blockquote>
038     * </p>
039     *
040     * @see EntityManagerAvailableAspect
041     */
042    @Target({ ElementType.METHOD })
043    @Retention(RetentionPolicy.RUNTIME)
044    @Inherited
045    @Documented
046    public @interface EntityManagerAvailable {
047    
048        /**
049         * Propagation specification. This determines when to create a new {@link javax.persistence.EntityManager}
050         * and how to handle the situation when one already exists (or not).
051         */
052        Propagation value() default Propagation.REQUIRED;
053    }
