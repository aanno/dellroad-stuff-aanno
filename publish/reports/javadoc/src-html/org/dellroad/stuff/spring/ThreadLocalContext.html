008    package org.dellroad.stuff.spring;
009    
010    import org.springframework.beans.factory.DisposableBean;
011    import org.springframework.context.ApplicationContext;
012    import org.springframework.context.ApplicationContextAware;
013    import org.springframework.context.ConfigurableApplicationContext;
014    
015    /**
016     * Associates a {@link ConfigurableApplicationContext} with each thread. Child threads inherit their parents' value.
017     *
018     * <p>
019     * Typically used as a singleton to determine the application context for autowiring
020     * {@link ThreadConfigurable @ThreadConfigurable} and {@link ThreadTransactional @ThreadTransactional} beans.
021     * </p>
022     *
023     * <p>
024     * If an instance of this class is included in an application context, then that context will become the default value
025     * (by virtue of this class being {@link ApplicationContextAware}). That is, that application context will be returned by
026     * {@link #get ThreadLocalContext.get()} until another application context is explicitly set.
027     * Doing this for the singleton instance makes the including application context the default for all
028     * {@link ThreadConfigurable @ThreadConfigurable}-annotated beans.
029     * </p>
030     *
031     * <p>
032     * For any instance, attempting to configure the default application context more than once (prior to
033     * {@linkplain #destroy bean disposal}) will generate an exception.
034     * </p>
035     *
036     * @see ThreadConfigurable
037     * @see ThreadTransactional
038     */
039    public class ThreadLocalContext extends InheritableThreadLocal<ConfigurableApplicationContext>
040      implements ApplicationContextAware, DisposableBean {
041    
042        private static ThreadLocalContext instance = new ThreadLocalContext();
043    
044        private ConfigurableApplicationContext defaultContext;
045    
046        /**
047         * Get the singleton instance.
048         */
049        public static ThreadLocalContext getInstance() {
050            return ThreadLocalContext.instance;
051        }
052    
053        /**
054         * Change the singleton {@link ThreadLocalContext} instance.
055         *
056         * <p>
057         * This method is normally not needed. However, by changing the behavior of the singleton instance,
058         * any arbitrary criteria can be used to determine which application context is used to configure
059         * {@link ThreadConfigurable @ThreadConfigurable}-annotated beans.
060         * </p>
061         */
062        public static void setInstance(ThreadLocalContext alternate) {
063            ThreadLocalContext.instance = alternate;
064        }
065    
066        /**
067         * Configure the default application context. Automatically invoked when this instance
068         * is included in a application context.
069         *
070         * @param context default application context, or null to have no default
071         * @throws IllegalStateException if the default application context is already configured
072         * @throws IllegalArgumentException if {@code context} is not a {@link ConfigurableApplicationContext}
073         */
074        @Override
075        public void setApplicationContext(ApplicationContext context) {
076            if (this.defaultContext != null) {
077                throw new IllegalStateException("cannot configure default context " + context
078                  + ": another default context is already configured: " + this.defaultContext);
079            }
080            if (!(context instanceof ConfigurableApplicationContext)) {
081                throw new IllegalArgumentException("cannot configure default context " + context
082                  + ": it must be an instance of " + ConfigurableApplicationContext.class.getName());
083            }
084            this.defaultContext = (ConfigurableApplicationContext)context;
085        }
086    
087        /**
088         * Removes the default application context. Automatically invoked when an instance of this
089         * class is included in a application context that is shut down.
090         */
091        @Override
092        public void destroy() {
093            this.defaultContext = null;
094        }
095    
096        /**
097         * Returns the default application context configured by {@link #setApplicationContext setApplicationContext()}, if any.
098         */
099        @Override
100        protected ConfigurableApplicationContext initialValue() {
101            return this.defaultContext;
102        }
103    }
