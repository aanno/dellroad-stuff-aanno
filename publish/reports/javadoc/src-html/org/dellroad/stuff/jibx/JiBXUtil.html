008    package org.dellroad.stuff.jibx;
009    
010    import java.io.BufferedWriter;
011    import java.io.IOException;
012    import java.io.InputStream;
013    import java.io.OutputStream;
014    import java.io.OutputStreamWriter;
015    import java.io.StringWriter;
016    import java.io.Writer;
017    import java.net.URL;
018    
019    import javax.xml.transform.stream.StreamResult;
020    import javax.xml.transform.stream.StreamSource;
021    
022    import org.jibx.runtime.JiBXException;
023    import org.springframework.oxm.jibx.JibxMarshaller;
024    
025    /**
026     * Some simplified API methods for JiBX XML encoding/decoding.
027     */
028    public final class JiBXUtil {
029    
030        private JiBXUtil() {
031        }
032    
033        /**
034         * Read in an object encoded as XML.
035         *
036         * <p>
037         * The {@code input} is not closed by this method.
038         *
039         * @param input source for the XML document
040         * @throws JiBXException if there is a JiBX parse error
041         * @throws IOException if an error occurs reading from {@code input}
042         */
043        @SuppressWarnings("unchecked")
044        public static <T> T readObject(Class<T> clazz, InputStream input) throws JiBXException, IOException {
045    
046            // Set up JiBX
047            JibxMarshaller unmarshaller = new JibxMarshaller();
048            unmarshaller.setTargetClass(clazz);
049            unmarshaller.afterPropertiesSet();
050    
051            // Read and parse XML
052            return (T)unmarshaller.unmarshal(new StreamSource(input));
053        }
054    
055        /**
056         * Read in an object encoded as XML from an {@link URL}.
057         *
058         * @param url source for the XML document
059         * @throws JiBXException if there is a JiBX parse error
060         * @throws IOException if an error occurs reading the referenced document
061         */
062        public static <T> T readObject(Class<T> clazz, URL url) throws JiBXException, IOException {
063            InputStream in = url.openStream();
064            try {
065                return JiBXUtil.readObject(clazz, in);
066            } finally {
067                try {
068                    in.close();
069                } catch (IOException e) {
070                    // ignore
071                }
072            }
073        }
074    
075        /**
076         * Write out the given instance encoded as a UTF-8 encoded XML document.
077         *
078         * @param output output destination; will not be closed by this method
079         * @throws JiBXException if there is a JiBX encoding error
080         * @throws IOException if an error occurs writing to {@code output}
081         */
082        public static <T> void writeObject(T obj, OutputStream output) throws JiBXException, IOException {
083            JiBXUtil.writeObject(obj, new OutputStreamWriter(output, "UTF-8"));
084        }
085    
086        /**
087         * Write out the given instance encoded as an XML document with "UTF-8" as the declared encoding.
088         *
089         * @param output output destination; will be closed by this method if successful
090         */
091        public static <T> void writeObject(T obj, Writer output) throws JiBXException, IOException {
092    
093            // Set up JiBX
094            JibxMarshaller marshaller = new JibxMarshaller();
095            marshaller.setTargetClass(obj.getClass());
096            marshaller.setEncoding("UTF-8");
097            marshaller.setIndent(4);
098            marshaller.afterPropertiesSet();
099    
100            // Output XML
101            BufferedWriter bufferedWriter = new BufferedWriter(output);
102            marshaller.marshal(obj, new StreamResult(bufferedWriter));
103        }
104    
105        /**
106         * Encode the given instance as an XML document and return it as a {@link String}.
107         *
108         * @throws JiBXException if there is a JiBX encoding error
109         */
110        public static <T> String toString(T obj) throws IOException, JiBXException {
111            final StringWriter w = new StringWriter();
112            JiBXUtil.writeObject(obj, w);
113            return w.toString();
114        }
115    }
