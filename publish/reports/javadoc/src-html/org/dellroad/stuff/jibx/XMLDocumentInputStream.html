008    package org.dellroad.stuff.jibx;
009    
010    import java.io.BufferedInputStream;
011    import java.io.IOException;
012    import java.io.InputStream;
013    
014    import org.dellroad.stuff.io.InputStreamReader;
015    import org.jibx.runtime.JiBXException;
016    
017    /**
018     * {@link InputStream} over which XML documents are passed. This class is a companion to {@link XMLDocumentOutputStream}.
019     *
020     * <p>
021     * XML documents are converted into Java objects via {@link JiBXUtil#readObject(Class, InputStream) JiBXUtil.readObject()}.
022     * </p>
023     *
024     * @param <T> XML document type
025     * @see XMLDocumentOutputStream
026     */
027    public class XMLDocumentInputStream<T> {
028    
029        private final Class<T> type;
030        private final InputStreamReader input;
031    
032        /**
033         * Constructor.
034         *
035         * @param type Java type for XML documents
036         * @param input data source
037         */
038        public XMLDocumentInputStream(Class<T> type, InputStream input) {
039            if (type == null)
040                throw new IllegalArgumentException("null type");
041            if (input == null)
042                throw new IllegalArgumentException("null input");
043            this.type = type;
044            this.input = new InputStreamReader(new BufferedInputStream(input));
045        }
046    
047        /**
048         * Read the next XML document, parsed and objectified.
049         *
050         * @return decoded object or {@code null} on EOF
051         */
052        public T read() throws IOException, JiBXException {
053            InputStream xml = this.input.read();
054            if (xml == null)
055                return null;
056            try {
057                return JiBXUtil.readObject(this.type, xml);
058            } finally {
059                try {
060                    xml.close();
061                } catch (IOException e) {
062                    // ignore
063                }
064            }
065        }
066    
067        public void close() throws IOException {
068            this.input.close();
069        }
070    }
