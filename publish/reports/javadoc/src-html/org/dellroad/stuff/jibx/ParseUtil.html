111        /**
112         * Derialize a millisecond-granularity time interval, e.g., "30s", "1d12h", "0.250s", etc.
113         *
114         * @see #serializeTimeInterval
115         */
116        public static long deserializeTimeInterval(String string) throws JiBXParseException {
117    
118            // Apply regular expression
119            Matcher m = Pattern.compile(TIME_INTERVAL_PATTERN).matcher(string);
120            if (string.length() == 0 || !m.matches())
121                throw new JiBXParseException("invalid time interval", string);
122    
123            // Parse regex groups
124            Object[][] groups = new Object[][] {
125                { 2, 24 * 60 * 60 * 1000 },
126                { 4,      60 * 60 * 1000 },
127                { 6,           60 * 1000 },
128                { 8,                1000 },
129            };
130            long value = 0;
131            for (int i = 0; i < groups.length; i++) {
132                String group = m.group((Integer)groups[i][0]);
133                if (group == null || group.length() == 0)
134                    continue;
135                double num;
136                try {
137                    num = Double.parseDouble(group);
138                } catch (NumberFormatException e) {
139                    throw new JiBXParseException("invalid time interval", string);
140                }
141                value += Math.round(num * (Integer)groups[i][1]);
142                if (value < 0)
143                    throw new JiBXParseException("time interval is too large", string);
144            }
145    
146            // Done
147            return value;
148        }

150        /**
151         * Serialize a millisecond-granularity time interval, e.g., "30s", "1d12h", "0.250s", etc.
152         *
153         * @see #deserializeTimeInterval
154         */
155        public static String serializeTimeInterval(long value) {
156            if (value < 0)
157                throw new IllegalArgumentException("negative value");
158            StringBuilder b = new StringBuilder(32);
159            long days = value / (24 * 60 * 60 * 1000);
160            value = value % (24 * 60 * 60 * 1000);
161            if (days > 0)
162                b.append(days).append('d');
163            long hours = value / (60 * 60 * 1000);
164            value = value % (60 * 60 * 1000);
165            if (hours > 0)
166                b.append(hours).append('h');
167            long minutes = value / (60 * 1000);
168            value = value % (60 * 1000);
169            if (minutes > 0)
170                b.append(minutes).append('m');
171            if (value != 0 || b.length() == 0)
172                b.append(String.format("%.3fs", (double)value / 1000.0));
173            return b.toString();
174        }

191        /**
192         * Deserialize a byte array using MAC address notation (colon-separated). Each byte must have two digits.
193         *
194         * @see #serializeByteArrayWithColons
195         */
196        public static byte[] deserializeByteArrayWithColons(String string) throws JiBXParseException {
197            if (string.length() == 0)
198                return new byte[0];
199            if (string.length() % 3 != 2)
200                throw new JiBXParseException("invalid byte array", string);
201            char[] nocolons = new char[((string.length() + 1) / 3) * 2];
202            int j = 0;
203            for (int i = 0; i < string.length(); i++) {
204                if (i % 3 == 2) {
205                    if (string.charAt(i) != ':')
206                        throw new JiBXParseException("invalid byte array", string);
207                    continue;
208                }
209                nocolons[j++] = string.charAt(i);
210            }
211            try {
212                return ByteArrayEncoder.decode(new String(nocolons));
213            } catch (IllegalArgumentException e) {
214                throw new JiBXParseException("invalid byte array", string, e);
215            }
216        }
217    
218        /**
219         * Serialize a byte array using MAC address notation (colon-separated). Each byte will have two digits.
220         *
221         * @see #deserializeByteArrayWithColons
222         */
223        public static String serializeByteArrayWithColons(byte[] array) {
224            char[] colons = new char[array.length * 3 - 1];
225            String nocolons = ByteArrayEncoder.encode(array);
226            int j = 0;
227            for (int i = 0; i < nocolons.length(); i += 2) {
228                colons[j++] = nocolons.charAt(i);
229                colons[j++] = nocolons.charAt(i + 1);
230                if (j < colons.length)
231                    colons[j++] = ':';
232            }
233            return new String(colons);
234        }

375        /**
376         * JiBX {@link String} deserializer that normalizes a string as is required by the {@code xsd:token} XSD type.
377         * This removes leading and trailing whitespace, and collapses all interior whitespace
378         * down to a single space character.
379         *
380         * @throws NullPointerException if {@code string} is null
381         */
382        public static String normalize(String string) {
383            return string.trim().replaceAll("\\s+", " ");
384        }

386        /**
387         * JiBX {@link String} deserializer support method that verifies that the input string matches the
388         * given regular expression. This method can be invoked by custom deserializers that supply the
389         * regular expression to it.
390         *
391         * @throws NullPointerException if {@code string} of {@code regex} is null
392         * @throws JiBXParseException   if {@code string} does not match {@code regex}
393         * @throws java.util.regex.PatternSyntaxException
394         *                              if {@code regex} is not a valid regular expression
395         */
396        public static String deserializeMatching(String regex, String string) throws JiBXParseException {
397            if (!string.matches(regex))
398                throw new JiBXParseException("input does not match pattern \"" + regex + "\"", string);
399            return string;
400        }

402        /**
403         * Boolean parser that allows "yes" and "no" as well as the usual "true", "false", "0", "1".
404         *
405         * @throws JiBXParseException if the value is not recognizable as a boolean
406         */
407        public static boolean deserializeBoolean(String string) throws JiBXParseException {
408            for (String s : BOOLEAN_TRUES) {
409                if (string.equalsIgnoreCase(s))
410                    return true;
411            }
412            for (String s : BOOLEAN_FALSES) {
413                if (string.equalsIgnoreCase(s))
414                    return false;
415            }
416            throw new JiBXParseException("invalid Boolean value", string);
417        }

419        /**
420         * Deserialize an array of integers separated by commas and/or whitespace.
421         *
422         * @see #serializeIntArray
423         */
424        public static int[] deserializeIntArray(String string) throws JiBXParseException {
425            String[] ints = string.trim().split("(\\s*,\\s*|\\s+)");
426            if (ints.length == 0 || ints[0].length() == 0)
427                return new int[0];
428            int[] array = new int[ints.length];
429            try {
430                for (int i = 0; i < array.length; i++)
431                    array[i] = Integer.parseInt(ints[i]);
432            } catch (NumberFormatException e) {
433                throw new JiBXParseException("invalid integer list", string, e);
434            }
435            return array;
436        }

438        /**
439         * Serialize an array of integers. Example: "1, 2, 3".
440         *
441         * @see #deserializeIntArray
442         */
443        public static String serializeIntArray(int[] array) throws JiBXParseException {
444            StringBuilder buf = new StringBuilder(array.length * 2);
445            for (int i = 0; i < array.length; i++) {
446                if (i > 0)
447                    buf.append(", ");
448                buf.append(array[i]);
449            }
450            return buf.toString();
451        }
