008    package org.dellroad.stuff.jibx;
009    
010    import java.util.Iterator;
011    import java.util.Map;
012    
013    /**
014     * Utility class for modeling {@link Map} properties in JiBX.
015     *
016     * <p>
017     * For example, suppose you have a class {@code Company} with a property named {@code directory} that has
018     * type {@code Map<String, Person>}.  Then you would define these methods in {@code Company.java}:
019     *
020     * <blockquote><code>
021     * // Getter and setter
022     * public Map<String, Person> getDirectory() {
023     *     return this.directory;
024     * }
025     * public void setDirectory(Map<String, Person> directory) {
026     *     this.directory = directory;
027     * }
028     *
029     * // JiBX "add-method"
030     * public void addDirectoryEntry(MapEntry<String, Person> entry) {
031     *     MapEntry.add(this.directory, entry);
032     * }
033     *
034     * // JiBX "iter-method"
035     * public Iterator<MapEntry<String, Person> iterateDirectory() {
036     *     return MapEntry.iterate(this.directory);
037     * }
038     * </code></blockquote>
039     *
040     * <p>
041     * Then in your JiBX binding definition, you would do something like this:
042     *
043     * <blockquote><code>
044     * &lt;binding package="com.example"&gt;
045     *
046     *     &lt;!-- Include XML mapping definition for a Person object --&gt;
047     *     &lt;include path="person.xml"/&gt;
048     *
049     *     &lt;!-- Define XML mapping for one entry in the directory map --&gt;
050     *     &lt;mapping abstract="true" type-name="directory_entry" class="org.dellroad.stuff.jibx.MapEntry"&gt;
051     *         &lt;value name="name" field="key" type="java.lang.String" style="attribute"/&gt;
052     *         &lt;structure name="person" map-as="person"/&gt;
053     *     &lt;/mapping&gt;
054     *
055     *     &lt;!-- Define XML mapping for a Company object --&gt;
056     *     &lt;mapping abstract="true" type-name="company" class="com.example.Company"&gt;
057     *         &lt;collection name="directory" item-type="org.dellroad.stuff.jibx.MapEntry"
058     *           add-method="addDirectoryEntry" iter-method="iterateDirectory"&gt;
059     *             &lt;structure name="entry" map-as="directory_entry"/&gt;
060     *         &lt;/collection&gt;
061     *         &lt;!-- other properties... --&gt;
062     *     &lt;/mapping&gt;
063     * &lt;/binding&gt;
064     * </code></blockquote>
065     *
066     * The resulting XML would look like this:
067     * <blockquote><code>
068     * &lt;company&gt;
069     *     &lt;directory&gt;
070     *         &lt;entry name="George Washington"&gt;
071     *             &lt;person&gt;
072     *                  &lt;!-- properties of George Washington... --&gt;
073     *             &lt;/person&gt;
074     *         &lt;/entry&gt;
075     *         &lt;entry name="Besty Ross"&gt;
076     *             &lt;person&gt;
077     *                  &lt;!-- properties of Besty Ross... --&gt;
078     *             &lt;/person&gt;
079     *         &lt;/entry&gt;
080     *     &lt;/directory&gt;
081     *     &lt;!-- other properties... --&gt;
082     * &lt;/company&gt;
083     */
084    public class MapEntry<K, V> {
085    
086        private K key;
087        private V value;
088    
089        /**
090         * Default constructor. Initializes both key and value to {@code null}.
091         */
092        public MapEntry() {
093        }
094    
095        /**
096         * Primary constructor.
097         *
098         * @param key map entry key
099         * @param value map entry value
100         */
101        public MapEntry(K key, V value) {
102            this.key = key;
103            this.value = value;
104        }
105    
106        /**
107         * Get this map entry's key.
108         */
109        public K getKey() {
110            return this.key;
111        }
112        public void setKey(K key) {
113            this.key = key;
114        }
115    
116        /**
117         * Get this map entry's value.
118         */
119        public V getValue() {
120            return this.value;
121        }
122        public void setValue(V value) {
123            this.value = value;
124        }
125    
126        /**
127         * Helper method intended to be used by a custom JiBX "iter-method".
128         * This method returns an iterator that iterates over all entries in the given map.
129         *
130         * @param map map to iterate
131         * @return map entry iterator
132         */
133        public static <K, V> Iterator<MapEntry<K, V>> iterate(Map<K, V> map) {
134            final Iterator<Map.Entry<K, V>> entryIterator = map.entrySet().iterator();
135            return new Iterator<MapEntry<K, V>>() {
136    
137                @Override
138                public boolean hasNext() {
139                    return entryIterator.hasNext();
140                }
141    
142                @Override
143                public MapEntry<K, V> next() {
144                    Map.Entry<K, V> entry = entryIterator.next();
145                    return new MapEntry<K, V>(entry.getKey(), entry.getValue());
146                }
147    
148                @Override
149                public void remove() {
150                    entryIterator.remove();
151                }
152            };
153        }
154    
155        /**
156         * Helper method intended to be used by a custom JiBX "add-method".
157         * The new entry is added to the map, replacing any existing entry with the same key.
158         *
159         * @param map map to which to add an new entry
160         * @param entry new entry to add
161         */
162        public static <K, V> void add(Map<K, V> map, MapEntry<? extends K, ? extends V> entry) {
163            map.put(entry.getKey(), entry.getValue());
164        }
165    }
