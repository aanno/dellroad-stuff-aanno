034    /**
035     * Manages an associated Spring {@link WebApplicationContext} with each {@link VaadinServiceSession} (aka, "Vaadin application").
036     * Typically created implicitly via {@link SpringVaadinServlet}.
037     *
038     * <h3>Overview</h3>
039     *
040     * <p>
041     * In Vaadin 7, the {@link com.vaadin.server.VaadinServiceSession} object holds the state associated with each client browser
042     * connection to a Vaadin servlet. For consistency with older versions of Vaadin, we'll call this a "Vaadin application" instance.
043     * This class gives each Vaadin such "Vaadin application" instance its own Spring application context, and all such
044     * application contexts share the same parent context, which is the one associated with the overal servlet web context
045     * (i.e., the one created by Spring's {@link org.springframework.web.context.ContextLoaderListener ContextLoaderListener}).
046     * This setup is analogous to how Spring's {@link org.springframework.web.servlet.DispatcherServlet DispatcherServlet}
047     * creates per-servlet application contexts that are children of the overall servlet web context.
048     * </p>
049     *
050     * <p>
051     * This class is implemented as a {@link SessionInitListener} and {@link SessionDestroyListener} on the servlet's
052     * {@link com.vaadin.server.VaadinService} object. In turn, the Spring context is created when a new Vaadin application
053     * instance is initialized, and destroyed when it is closed. To use this class, use the {@link SpringVaadinServlet}
054     * in place of the usual {@link com.vaadin.server.VaadinServlet} in {@code web.xml}.
055     * </p>
056     *
057     * <h3>Accessing the Spring Context</h3>
058     *
059     * <p>
060     * The {@link ConfigurableWebApplicationContext} itself is {@linkplain VaadinServiceSession#setAttribute stored as an attribute}
061     * of the {@link VaadinServiceSession} under the key {@link #APPLICATION_CONTEXT_ATTRIBUTE_KEY}. The
062     * {@link #getApplicationContext() getApplicationContext()} method provides convenient access from anywhere.
063     * </p>
064     *
065     * <h3>Exposing the Vaadin Application</h3>
066     *
067     * <p>
068     * The {@link VaadinServiceSession} instance representing the "Vaadin application" can be exposed in the associated Spring
069     * application context and therefore made available for autowiring, etc. Simply add a bean definition that invokes
070     * {@link VaadinUtil#getCurrentSession}:
071     * <blockquote><pre>
072     *  &lt;bean id="vaadinServiceSession" class="org.dellroad.stuff.vaadin7.VaadinUtil" factory-method="getCurrentSession"/&gt;
073     * </pre></blockquote>
074     * This bean can then be autowired into application-specific "backend" beans, allowing them to use e.g.
075     * {@link VaadinUtil#invoke VaadinUtil.invoke()}, which performs the locking necessary to avoid race conditions.
076     * But see also {@link VaadinApplication} for a convenience class that makes this process a little cleaner.
077     *
078     * <h3><code>@VaadinConfigurable</code> Beans</h3>
079     *
080     * <p>
081     * It is also possible to configure beans outside of this application context using AOP, so that any invocation of
082     * {@code new FooBar()}, where the class {@code FooBar} is marked {@link VaadinConfigurable @VaadinConfigurable},
083     * will automagically cause the new {@code FooBar} object to be configured by the application context associated with
084     * the currently running Vaadin application. In effect, this does for Vaadin application beans what Spring's
085     * {@link org.springframework.beans.factory.annotation.Configurable @Configurable} does for regular beans.
086     * </p>
087     *
088     * <p>
089     * Note however that Spring {@linkplain org.springframework.beans.factory.DisposableBean#destroy destroy methods}
090     * will not be invoked on application close for these beans, since their lifecycle is controlled outside of the
091     * Spring application context (this is also the case with
092     * {@link org.springframework.beans.factory.annotation.Configurable @Configurable} beans). Instead, these beans
093     * can themselves register as a {@link SessionDestroyListener} for shutdown notification; see
094     * {@link VaadinUtil#addSessionDestroyListener VaadinUtil.addSessionDestroyListener()} for a memory-leak free
095     * method for doing this.
096     * </p>
097     *
098     * @see VaadinConfigurable
099     * @see SpringVaadinServlet
100     * @see VaadinApplication
101     * @see com.vaadin.server.VaadinService
102     * @see com.vaadin.server.VaadinServiceSession
103     */
104    @SuppressWarnings("serial")
105    public class SpringServiceSession implements SessionInitListener, SessionDestroyListener {
106    
107        /**
108         * The {@link VaadinServiceSession} attribute key under which the Spring {@link ConfigurableWebApplicationContext}
109         * is stored.
110         */
111        public static final String APPLICATION_CONTEXT_ATTRIBUTE_KEY = "springServiceSessionApplicationContext";
112    
113        private static final AtomicLong UNIQUE_INDEX = new AtomicLong();
114    
115        private final Logger log = LoggerFactory.getLogger(this.getClass());
116        private final String applicationName;
117        private final String configLocation;
118    
119        /**
120         * Constructor.
121         *
122         * @param applicationName Vaadin application name
123         * @param configLocation XML application context config file location(s), or null for the default
124         *  {@code /WEB-INF/ServletName.xml}, where {@code ServletName} is the value of {@code applicationName}
125         * @throws IllegalArgumentException if {@code applicationName} is null
126         */
127        public SpringServiceSession(String applicationName, String configLocation) {
128            if (applicationName == null)
129                throw new IllegalArgumentException("null applicationName");
130            this.applicationName = applicationName;
131            this.configLocation = configLocation;
132        }
133    
134        /**
135         * Get the name of this Vaadin application.
136         */
137        public String getApplicationName() {
138            return this.applicationName;
139        }
140    
141        /**
142         * Get the Spring application context associated with the given {@link VaadinServiceSession}.
143         *
144         * @return Spring application context, or null if none is found
145         * @throws IllegalArgumentException if {@code session} is null
146         */
147        public static ConfigurableWebApplicationContext getApplicationContext(VaadinServiceSession session) {
148            if (session == null)
149                throw new IllegalArgumentException("null session");
150            return (ConfigurableWebApplicationContext)session.getAttribute(APPLICATION_CONTEXT_ATTRIBUTE_KEY);
151        }
152    
153        /**
154         * Get the Spring application context associated with the current thread's {@link VaadinServiceSession}.
155         *
156         * @return Spring application context, never null
157         * @throws IllegalStateException if there is no {@link VaadinServiceSession} associated with the current thread
158         * @throws IllegalStateException if there is no Spring application context associated with the {@link VaadinServiceSession}
159         */
160        public static ConfigurableWebApplicationContext getApplicationContext() {
161    
162            // Get current session
163            VaadinServiceSession session = VaadinUtil.getCurrentSession();
164    
165            // Get the associated application context
166            ConfigurableWebApplicationContext context = SpringServiceSession.getApplicationContext(session);
167            if (context == null) {
168                throw new IllegalStateException("there is no Spring application context associated with the current"
169                  + " VaadinServiceSession; are you using SpringVaadinServlet instead of VaadinServlet?");
170            }
171    
172            // Done
173            return context;
174        }
175    
176    // SessionInitListener
177    
178        @Override
179        public void sessionInit(final SessionInitEvent event) {
180            VaadinUtil.invoke(event.getSession(), new Runnable() {
181                @Override
182                public void run() {
183                    SpringServiceSession.this.loadContext(event.getSession(), event.getRequest());
184                }
185            });
186        }
187    
188    // Context loading
189    
190        /**
191         * Load the Spring application context.
192         *
193         * <p>
194         * This method expects that {@code session} is already {@linkplain VaadinServiceSession#getLock locked}.
195         *
196         * @param session the Vaadin application session
197         * @param request the triggering request
198         */
199        protected void loadContext(VaadinServiceSession session, VaadinRequest request) {
200    
201            // Sanity check
202            if (session == null)
203                throw new IllegalStateException("null session");
204            if (request == null)
205                throw new IllegalStateException("null request");
206            if (this.getApplicationContext(session) != null)
207                throw new IllegalStateException("context already loaded");
208    
209            // Logging
210            this.log.info("creating new application context for Vaadin application [" + this.getApplicationName()
211              + "] in session " + session);
212    
213            // Find the servlet context parent application context
214            ServletContext servletContext = null;
215            WebApplicationContext parent = null;
216            if (request instanceof VaadinServletRequest) {
217    
218                // Find servlet context
219                HttpServletRequest httpRequest = ((VaadinServletRequest)request).getHttpServletRequest();
220                try {
221                    // getServletContext() is a servlet AIP 3.0 method, so don't freak out if it's not there
222                    servletContext = (ServletContext)HttpServletRequest.class.getMethod("getServletContext").invoke(httpRequest);
223                } catch (Exception e) {
224                    servletContext = ContextLoader.getCurrentWebApplicationContext().getServletContext();
225                }
226    
227                // Find associated application context; it will become our parent context
228                parent = WebApplicationContextUtils.getWebApplicationContext(servletContext);
229            } else if (request instanceof VaadinPortletRequest) {
230                this.log.warn("portlets are not supported yet");
231            }
232    
233            // Create and configure a new application context for this Application instance
234            XmlWebApplicationContext context = new XmlWebApplicationContext();
235            context.setId(ConfigurableWebApplicationContext.APPLICATION_CONTEXT_ID_PREFIX
236              + (servletContext != null ? servletContext.getContextPath() + "/" : "")
237              + this.getApplicationName() + "-" + UNIQUE_INDEX.incrementAndGet());
238            if (parent != null)
239                context.setParent(parent);
240            if (servletContext != null)
241                context.setServletContext(servletContext);
242            //context.setServletConfig(??);
243            context.setNamespace(this.getApplicationName());
244    
245            // Set explicit config location(s), if any
246            if (this.configLocation != null)
247                context.setConfigLocation(this.configLocation);
248    
249            // Associate context with the current session
250            boolean success = false;
251            session.setAttribute(APPLICATION_CONTEXT_ATTRIBUTE_KEY, context);
252    
253            // Refresh context
254            try {
255                context.refresh();
256                success = true;
257            } finally {
258                if (!success)
259                    session.setAttribute(APPLICATION_CONTEXT_ATTRIBUTE_KEY, null);
260            }
261        }
262    
263    // SessionDestroyListener
264    
265        @Override
266        public void sessionDestroy(SessionDestroyEvent event) {
267            final VaadinServiceSession session = event.getSession();
268            final ConfigurableWebApplicationContext context = this.getApplicationContext(session);
269            if (context == null) {
270                this.log.error(this.getClass().getSimpleName() + ".sessionDestroy() invoked but no application context found");
271                return;
272            }
273            session.setAttribute(APPLICATION_CONTEXT_ATTRIBUTE_KEY, null);
274            VaadinUtil.invoke(session, new Runnable() {
275                @Override
276                public void run() {
277                    SpringServiceSession.this.log.info("closing Vaadin application ["
278                      + SpringServiceSession.this.getApplicationName() + "] application context: " + context);
279                    context.close();
280                }
281            });
282        }
283    
284    // Serialization
285    
286        private void readObject(ObjectInputStream input) throws IOException, ClassNotFoundException {
287            input.defaultReadObject();
288            final VaadinServiceSession session = VaadinUtil.getCurrentSession();
289            final VaadinRequest request = VaadinUtil.getCurrentRequest();
290            VaadinUtil.invoke(session, new Runnable() {
291                @Override
292                public void run() {
293                    SpringServiceSession.this.loadContext(session, request);
294                }
295            });
296        }
