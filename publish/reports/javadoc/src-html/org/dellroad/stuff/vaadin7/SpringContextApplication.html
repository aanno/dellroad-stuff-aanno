008    package org.dellroad.stuff.vaadin7;
009    
010    import org.springframework.web.context.ConfigurableWebApplicationContext;
011    
012    /**
013     * Vaadin 7 version of {@link org.dellroad.stuff.vaadin.SpringContextApplication}.
014     *
015     * @see org.dellroad.stuff.vaadin.SpringContextApplication
016     */
017    @SuppressWarnings("serial")
018    public class SpringContextApplication extends org.dellroad.stuff.vaadin.SpringContextApplication {
019    
020        /**
021         * Initialize the application. In Vaadin 7 overriding this method is optional.
022         *
023         * <p>
024         * The implementation in {@link ContextApplication} does nothing.
025         */
026        @Override
027        protected void initSpringApplication(ConfigurableWebApplicationContext context) {
028        }
029    
030    // Error handling
031    
032        @Override
033        public void showError(String title, String description) {
034            ContextApplication.showError(this.getRoots(), this.getNotificationDelay(), title, description);
035        }
036    }
