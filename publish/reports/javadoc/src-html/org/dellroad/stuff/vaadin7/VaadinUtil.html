008    package org.dellroad.stuff.vaadin7;
009    
010    import com.vaadin.server.SessionDestroyEvent;
011    import com.vaadin.server.SessionDestroyListener;
012    import com.vaadin.server.VaadinRequest;
013    import com.vaadin.server.VaadinService;
014    import com.vaadin.server.VaadinServiceSession;
015    
016    /**
017     * Miscellaneous utility methods.
018     */
019    public final class VaadinUtil {
020    
021        private VaadinUtil() {
022        }
023    
024        /**
025         * Get the {@link VaadinServiceSession} associated with the current thread.
026         * This is just a wrapper around {@link VaadinServiceSession#getCurrent} that throws an exception instead
027         * of returning null when there is no session associated with the current thread.
028         *
029         * @return current {@link VaadinServiceSession}, never null
030         *
031         * @throws IllegalStateException if there is no {@link VaadinServiceSession} associated with the current thread
032         */
033        public static VaadinServiceSession getCurrentSession() {
034            VaadinServiceSession session = VaadinServiceSession.getCurrent();
035            if (session == null) {
036                throw new IllegalStateException("there is no VaadinServiceSession associated with the current thread;"
037                  + " are we executing within a Vaadin HTTP request or VaadinUtil.invoke()?");
038            }
039            return session;
040        }
041    
042        /**
043         * Get the {@link VaadinRequest} associated with the current thread.
044         * This is just a wrapper around {@link VaadinService#getCurrentRequest} that throws an exception instead
045         * of returning null when there is no request associated with the current thread.
046         *
047         * @return current {@link VaadinRequest}, never null
048         *
049         * @throws IllegalStateException if there is no {@link VaadinRequest} associated with the current thread
050         */
051        public static VaadinRequest getCurrentRequest() {
052            VaadinRequest request = VaadinService.getCurrentRequest();
053            if (request == null) {
054                throw new IllegalStateException("there is no VaadinRequest associated with the current thread;"
055                  + " are we executing within a Vaadin HTTP request?");
056            }
057            return request;
058        }
059    
060        /**
061         * Peform some action while holding the given {@link VaadinServiceSession}'s lock.
062         *
063         * <p>
064         * All back-end threads that interact with Vaadin components must use this method (or equivalent)
065         * to avoid race conditions. Since session locks are re-entrant, it will not cause problems if this
066         * method is used by a "front-end" (i.e., Vaadin HTTP request) thread.
067         * </p>
068         *
069         * <p>
070         * Note: when executing within a Vaadin HTTP request, the current thread's {@link VaadinServiceSession}
071         * is available via {@link VaadinServiceSession#getCurrent}.
072         * </p>
073         *
074         * @throws IllegalArgumentException if either parameter is null
075         * @see VaadinApplication#invoke
076         */
077        public static void invoke(VaadinServiceSession session, Runnable action) {
078            if (session == null)
079                throw new IllegalArgumentException("null session");
080            if (action == null)
081                throw new IllegalArgumentException("null action");
082            session.getLock().lock();
083            VaadinServiceSession previousSession = VaadinServiceSession.getCurrent();
084            VaadinServiceSession.setCurrent(session);
085            try {
086                action.run();
087            } finally {
088                session.getLock().unlock();
089                VaadinServiceSession.setCurrent(previousSession);
090            }
091        }
092    
093        /**
094         * Register for a notification when the {@link VaadinServiceSession} is closed, without creating a memory leak.
095         * This method is intended to be used by listeners that are themselves part of a Vaadin application.
096         *
097         * <p>
098         * Explanation: the {@link VaadinServiceSession} class does not provide a listener API directly; instead, you must
099         * use the {@link com.vaadin.server.VaadinService} class. However, registering as a listener on the
100         * {@link com.vaadin.server.VaadinService} when you are part of a Vaadin application sets you up for a memory leak
101         * if you forget to unregister yourself when the notification arrives, because the {@link com.vaadin.server.VaadinService}
102         * lifetime is longer than the {@link VaadinServiceSession} lifetime. This method handles that de-registration for
103         * you automatically.
104         *
105         * @throws IllegalArgumentException if either parameter is null
106         * @see VaadinApplication#addSessionDestroyListener
107         */
108        public static void addSessionDestroyListener(VaadinServiceSession session, SessionDestroyListener listener) {
109            session.getService().addSessionDestroyListener(new LeakAvoidingDestroyListener(session, listener));
110        }
111    
112        /**
113         * Remove a listener added via {@link #addSessionDestroyListener addSessionDestroyListener()}.
114         *
115         * @throws IllegalArgumentException if either parameter is null
116         * @see VaadinApplication#removeSessionDestroyListener
117         */
118        public static void removeSessionDestroyListener(VaadinServiceSession session, SessionDestroyListener listener) {
119            session.getService().removeSessionDestroyListener(new LeakAvoidingDestroyListener(session, listener));
120        }
121    
122    // LeakAvoidingDestroyListener
123    
124        private static class LeakAvoidingDestroyListener implements SessionDestroyListener {
125    
126            private final VaadinServiceSession session;
127            private final SessionDestroyListener listener;
128    
129            public LeakAvoidingDestroyListener(VaadinServiceSession session, SessionDestroyListener listener) {
130                if (session == null)
131                    throw new IllegalArgumentException("null session");
132                if (listener == null)
133                    throw new IllegalArgumentException("null listener");
134                this.session = session;
135                this.listener = listener;
136            }
137    
138            @Override
139            public void sessionDestroy(SessionDestroyEvent event) {
140                final VaadinServiceSession closedSession = event.getSession();
141                if (closedSession == this.session) {
142                    this.session.getService().removeSessionDestroyListener(this);       // remove myself as listener to avoid mem leak
143                    this.listener.sessionDestroy(event);
144                }
145            }
146    
147            @Override
148            public boolean equals(Object obj) {
149                if (obj == null || obj.getClass() != this.getClass())
150                    return false;
151                LeakAvoidingDestroyListener that = (LeakAvoidingDestroyListener)obj;
152                return this.session == that.session && this.listener.equals(that.listener);
153            }
154    
155            @Override
156            public int hashCode() {
157                return this.session.hashCode() ^ this.listener.hashCode();
158            }
159        }
160    }
