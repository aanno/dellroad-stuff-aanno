008    package org.dellroad.stuff.vaadin7;
009    
010    import com.vaadin.Application;
011    import com.vaadin.ui.Notification;
012    import com.vaadin.ui.Root;
013    
014    import java.util.Collection;
015    
016    /**
017     * Vaadin 7 version of {@link org.dellroad.stuff.vaadin.ContextApplication}.
018     *
019     * @see org.dellroad.stuff.vaadin.ContextApplication
020     */
021    @SuppressWarnings("serial")
022    public class ContextApplication extends org.dellroad.stuff.vaadin.ContextApplication {
023    
024        /**
025         * Initialize the application. In Vaadin 7 overriding this method is optional.
026         *
027         * <p>
028         * The implementation in {@link ContextApplication} does nothing.
029         */
030        @Override
031        protected void initApplication() {
032        }
033    
034    // Error handling
035    
036        @Override
037        public void showError(String title, String description) {
038            ContextApplication.showError(this.getRoots(), this.getNotificationDelay(), title, description);
039        }
040    
041        static void showError(Collection<Root> roots, int notificationDelay, String title, String description) {
042    
043            // Get window
044            Root root = Root.getCurrent();
045            if (root == null) {
046                if (roots.isEmpty())
047                    return;
048                root = roots.iterator().next();
049            }
050    
051            // Show error
052            Notification notification = new Notification(title, description, Notification.TYPE_ERROR_MESSAGE);
053            notification.setStyleName("warning");
054            notification.setDelayMsec(notificationDelay);
055            notification.show(root.getPage());
056        }
057    
058        @Override
059        public void invoke(Runnable action) {
060            final Application previous = Application.getCurrent();
061            if (previous != null && previous != this)
062                throw new IllegalStateException("there is already a current application for this thread (according to Vaadin)");
063            Application.setCurrent(this);
064            try {
065                super.invoke(action);
066            } finally {
067                Application.setCurrent(previous);
068            }
069        }
070    }
