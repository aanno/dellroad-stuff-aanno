008    package org.dellroad.stuff.vaadin7;
009    
010    import com.vaadin.ui.Notification;
011    import com.vaadin.ui.Root;
012    
013    import java.util.Collection;
014    
015    /**
016     * Vaadin 7 version of {@link org.dellroad.stuff.vaadin.ContextApplication}.
017     *
018     * @see org.dellroad.stuff.vaadin.ContextApplication
019     */
020    @SuppressWarnings("serial")
021    public class ContextApplication extends org.dellroad.stuff.vaadin.ContextApplication {
022    
023        /**
024         * Initialize the application. In Vaadin 7 overriding this method is optional.
025         *
026         * <p>
027         * The implementation in {@link ContextApplication} does nothing.
028         */
029        @Override
030        protected void initApplication() {
031        }
032    
033    // Error handling
034    
035        @Override
036        public void showError(String title, String description) {
037            ContextApplication.showError(this.getRoots(), this.getNotificationDelay(), title, description);
038        }
039    
040        static void showError(Collection<Root> roots, int notificationDelay, String title, String description) {
041    
042            // Get window
043            Root root = Root.getCurrentRoot();
044            if (root == null) {
045                if (roots.isEmpty())
046                    return;
047                root = roots.iterator().next();
048            }
049    
050            // Show error
051            Notification notification = new Notification(title, description, Notification.TYPE_ERROR_MESSAGE);
052            notification.setStyleName("warning");
053            notification.setDelayMsec(notificationDelay);
054            root.showNotification(notification);
055        }
056    }
