008    package org.dellroad.stuff.vaadin7;
009    
010    import com.vaadin.server.SessionDestroyEvent;
011    import com.vaadin.server.SessionDestroyListener;
012    import com.vaadin.server.VaadinServiceSession;
013    
014    /**
015     * Support superclass customized for use by listeners that are part of a Vaadin application when listening
016     * to non-Vaadin ("external") event sources.
017     *
018     * <p>
019     * Listeners that are part of a Vaadin application should use this superclass if they are going to be registered
020     * with non-Vaadin event sources, and use the methods {@link #register()} and {@link #unregister()} to control listener
021     * registration. Subclasses implement {@link #register(Object) register(S)} and {@link #unregister(Object) register(S)}
022     * to perform the actual listener registration/unregister operations.
023     *
024     * <p>
025     * Use of this class will ensure two things:
026     * <ul>
027     *  <li>Events can be delivered {@linkplain VaadinUtil#invoke in the proper Vaadin application context}; and</li>
028     *  <li>The listener is automatically unregistered from the external event source when the Vaadin application is closed,
029     *      avoiding a memory leak</li>
030     * </ul>
031     * </p>
032     *
033     * <p>
034     * <b>Important:</b> subclass listener methods must use {@link #handleEvent handleEvent()} when handling events.
035     * This will ensure proper locking to avoid race conditions.
036     * </p>
037     *
038     * <p>
039     * Note: when listening to event sources that are scoped to specific Vaadin application instances and already originate events
040     * within the proper Vaadin application context (i.e., event sources that are not external to the Vaadin application),
041     * then the use of this superclass is not necessary (however, it also doesn't hurt to use it anyway).
042     * </p>
043     *
044     * @param <S> The type of the event source
045     * @see VaadinUtil#invoke
046     * @see VaadinApplicationScope
047     * @see VaadinApplicationListener
048     * @see SpringServiceSession
049     */
050    public abstract class VaadinExternalListener<S> {
051    
052        private final S eventSource;
053        private final VaadinServiceSession session;
054        private final CloseListener closeListener = new CloseListener();
055    
056        /**
057         * Convenience constructor. Equivalent to:
058         * <blockquote>
059         *  {@link #VaadinExternalListener(Object, VaadinServiceSession)
060         *      VaadinExternalListener(eventSource, VaadinUtil.getCurrentSession())}
061         * </blockquote>
062         *
063         * @param eventSource the event source on which this listener will register
064         * @throws IllegalArgumentException if {@code eventSource} is null
065         * @throws IllegalStateException if there is no {@link VaadinServiceSession} associated with the current thread
066         */
067        protected VaadinExternalListener(S eventSource) {
068            this(eventSource, VaadinUtil.getCurrentSession());
069        }
070    
071        /**
072         * Convenience constructor. Equivalent to:
073         * <blockquote>
074         *  {@link #VaadinExternalListener(Object, VaadinServiceSession) VaadinExternalListener(eventSource, application.getSession())}
075         * </blockquote>
076         *
077         * @param eventSource the event source on which this listener will register
078         * @param application the {@link VaadinApplication} that this listener is part of
079         * @throws IllegalArgumentException if {@code eventSource} is null
080         * @throws NullPointerException if {@code application} is null
081         */
082        protected VaadinExternalListener(S eventSource, VaadinApplication application) {
083            this(eventSource, application.getSession());
084        }
085    
086        /**
087         * Primary constructor.
088         *
089         * @param eventSource the event source on which this listener will register when {@link #register} is invoked
090         * @param session the associated Vaadin application's session
091         * @throws IllegalArgumentException if either parameter is null
092         */
093        protected VaadinExternalListener(S eventSource, VaadinServiceSession session) {
094            if (eventSource == null)
095                throw new IllegalArgumentException("null eventSource");
096            if (session == null)
097                throw new IllegalArgumentException("null session");
098            this.eventSource = eventSource;
099            this.session = session;
100        }
101    
102        /**
103         * Register as a listener on configured event source.
104         *
105         * <p>
106         * This also registers a {@link SessionDestroyListener} on the {@linkplain #getSession configured Vaadin application},
107         * so that when the application closes we can unregister this instance from the event source to avoid a memory leak.
108         */
109        public void register() {
110            VaadinUtil.addSessionDestroyListener(this.session, this.closeListener);
111            this.register(this.eventSource);
112        }
113    
114        /**
115         * Un-register as a listener on configured event source.
116         *
117         * <p>
118         * This also unregisters the {@link SessionDestroyListener} registered by {@link #register}.
119         */
120        public void unregister() {
121            VaadinUtil.removeSessionDestroyListener(this.session, this.closeListener);
122            this.unregister(this.eventSource);
123        }
124    
125        /**
126         * Get the {@link VaadinServiceSession} (aka Vaadin application) with which this instance is associated.
127         */
128        public final VaadinServiceSession getSession() {
129            return this.session;
130        }
131    
132        /**
133         * Get the event source with which this instance is (or was) registered as a listener.
134         */
135        public final S getEventSource() {
136            return this.eventSource;
137        }
138    
139        /**
140         * Execute the given listener action using the {@link VaadinServiceSession} with which this instance is associated.
141         *
142         * <p>
143         * Subclass listener methods should handle events using this method to ensure proper locking to avoid race conditions.
144         *
145         * @param action action to perform
146         * @see VaadinUtil#invoke
147         */
148        protected void handleEvent(Runnable action) {
149            VaadinUtil.invoke(this.getSession(), action);
150        }
151    
152        /**
153         * Register as a listener on the given event source.
154         *
155         * <p>
156         * Subclass must implement this to perform the actual listener registration.
157         *
158         * @param eventSource event source, never null
159         */
160        protected abstract void register(S eventSource);
161    
162        /**
163         * Register as a listener from the given event source.
164         *
165         * <p>
166         * Subclass must implement this to perform the actual listener registration.
167         *
168         * @param eventSource event source, never null
169         */
170        protected abstract void unregister(S eventSource);
171    
172    // Application close listener
173    
174        private final class CloseListener implements SessionDestroyListener {
175    
176            @Override
177            public void sessionDestroy(SessionDestroyEvent event) {
178                VaadinExternalListener.this.unregister();
179            }
180        }
181    }
