008    package org.dellroad.stuff.vaadin7;
009    
010    import com.vaadin.server.DeploymentConfiguration;
011    import com.vaadin.server.ServiceException;
012    import com.vaadin.server.VaadinRequest;
013    import com.vaadin.server.VaadinServiceSession;
014    import com.vaadin.server.VaadinServlet;
015    import com.vaadin.server.VaadinServletService;
016    
017    import java.util.Properties;
018    
019    import javax.servlet.ServletConfig;
020    import javax.servlet.ServletException;
021    
022    /**
023     * A {@link VaadinServlet} that associates and manages a Spring
024     * {@link org.springframework.web.context.ConfigurableWebApplicationContext} with each
025     * {@link com.vaadin.server.VaadinServiceSession} (aka, "Vaadin application" in the old terminology).
026     *
027     * <p>
028     * The {@code vaadinContextConfigLocation} servlet parameter may be used to specify the Spring XML config
029     * file location(s). For example:
030     *
031     * <blockquote><pre>
032     * &lt;servlet&gt;
033     *     &lt;servlet-name&gt;My Vaadin App&lt;/servlet-name&gt;
034     *     &lt;servlet-class&gt;org.dellroad.stuff.vaadin7.SpringVaadinServlet&lt;/servlet-class&gt;
035     *     &lt;init-param&gt;
036     *         &lt;param-name&gt;UI&lt;/param-name&gt;
037     *         &lt;param-value&gt;com.example.MyApplicationUI&lt;/param-value&gt;
038     *     &lt;/init-param&gt;
039     *     &lt;init-param&gt;
040     *         &lt;param-name&gt;configLocation&lt;/param-name&gt;
041     *         &lt;param-value&gt;classpath:com/example/MyApplicationContext.xml&lt;/param-value&gt;
042     *     &lt;/init-param&gt;
043     * &lt;/servlet&gt;
044     * </pre></blockquote>
045     * </p>
046     *
047     * <p>
048     * The only real function of this servlet is to create and register a {@link SpringServiceSession} as a listener on the
049     * {@link com.vaadin.server.VaadinService} associated with this servlet. The {@link SpringServiceSession} in turn detects
050     * the creation and destruction of Vaadin application instances (represented by {@link com.vaadin.server.VaadinServiceSession}
051     * instances) and does the work of managing the associated Spring application contexts.
052     * </p>
053     *
054     * <p>
055     * Use of this servlet in place of the standard Vaadin servlet is required for the {@link VaadinConfigurable @VaadinConfigurable}
056     * annotation to work.
057     * </p>
058     *
059     * <p>
060     * Supported URL parameters:
061     * <div style="margin-left: 20px;">
062     * <table border="1" cellpadding="3" cellspacing="0">
063     * <tr bgcolor="#ccffcc">
064     *  <th align="left">Parameter Name</th>
065     *  <th align="left">Required?</th>
066     *  <th align="left">Description</th>
067     * </tr>
068     * <tr>
069     * <td>{@code applicationName}</td>
070     * <td align="center">No</td>
071     * <td>
072     *  Vaadin application name. Used for logging purposes and as the name of the XML application context file
073     *  when {@code configLocation} is not specified. If this parameter is not specified, the
074     *  name of the servlet is used.
075     * </td>
076     * </tr>
077     * <tr>
078     * <td>{@code configLocation}</td>
079     * <td align="center">No</td>
080     * <td>
081     *  Location of Spring application context XML file(s). Multiple locations are separated by whitespace.
082     *  If omitted, {@code /WEB-INF/ServletName.xml} is used, where {@code ServletName} is the name of the Vaadin
083     *  application (see {@code applicationName}).
084     * </td>
085     * </tr>
086     * <tr>
087     * <td>{@code listenerClass}</td>
088     * <td align="center">No</td>
089     * <td>
090     *  Specify the name of a custom class extending {@link SpringServiceSession} and having the same constructor arguments.
091     *  If omitted, {@link SpringServiceSession} is used.
092     * </td>
093     * </tr>
094     * </table>
095     * </div>
096     * </p>
097     *
098     * @see SpringServiceSession
099     * @see VaadinConfigurable
100     * @see VaadinApplication
101     */
102    @SuppressWarnings("serial")
103    public class SpringVaadinServlet extends VaadinServlet {
104    
105        /**
106         * Servlet initialization parameter (<code>{@value #CONFIG_LOCATION_PARAMETER}</code>) used to specify
107         * the location(s) of the Spring application context XML file(s). Multiple XML files may be separated by whitespace.
108         * This parameter is optional.
109         */
110        public static final String CONFIG_LOCATION_PARAMETER = "configLocation";
111    
112        /**
113         * Servlet initialization parameter (<code>{@value #LISTENER_CLASS_PARAMETER}</code>) used to specify
114         * the name of an custom subclass of {@link SpringServiceSession}.
115         * This parameter is optional.
116         */
117        public static final String LISTENER_CLASS_PARAMETER = "listenerClass";
118    
119        /**
120         * Servlet initialization parameter (<code>{@value #APPLICATION_NAME_PARAMETER}</code>) used to specify
121         * the name the application.
122         * This parameter is optional.
123         */
124        public static final String APPLICATION_NAME_PARAMETER = "applicationName";
125    
126        private String servletName;
127    
128        @Override
129        public void init(ServletConfig config) throws ServletException {
130            this.servletName = config.getServletName();
131            if (this.servletName == null)
132                throw new IllegalArgumentException("null servlet name");
133            super.init(config);
134        }
135    
136        @Override
137        protected void servletInitialized() throws ServletException {
138    
139            // Sanity check
140            if (this.servletName == null)
141                throw new IllegalArgumentException("servlet not initialized");
142    
143            // Defer to superclass
144            super.servletInitialized();
145    
146            // Get params
147            final VaadinServletService servletService = this.getService();
148            final Properties params = servletService.getDeploymentConfiguration().getInitParameters();
149            final String contextLocation = params.getProperty(CONFIG_LOCATION_PARAMETER);
150            final String listenerClassName = params.getProperty(LISTENER_CLASS_PARAMETER);
151            String applicationName = params.getProperty(APPLICATION_NAME_PARAMETER);
152            if (applicationName == null)
153                applicationName = this.servletName;
154    
155            // Detect listener class to use
156            Class<? extends SpringServiceSession> listenerClass = SpringServiceSession.class;
157            if (listenerClassName != null) {
158                try {
159                    listenerClass = Class.forName(listenerClassName, false, Thread.currentThread().getContextClassLoader())
160                      .asSubclass(SpringServiceSession.class);
161                } catch (RuntimeException e) {
162                    throw e;
163                } catch (Exception e) {
164                    throw new ServletException("error finding class " + listenerClassName, e);
165                }
166            }
167    
168            // Create session listener
169            SpringServiceSession sessionListener;
170            try {
171                sessionListener = listenerClass.getConstructor(String.class, String.class)
172                  .newInstance(applicationName, contextLocation);
173            } catch (RuntimeException e) {
174                throw e;
175            } catch (Exception e) {
176                throw new ServletException("error instantiating " + listenerClass, e);
177            }
178    
179            // Register session listener
180            servletService.addSessionInitListener(sessionListener);
181            servletService.addSessionDestroyListener(sessionListener);
182        }
183    
184        /**
185         * Workaround for <a href="http://dev.vaadin.com/ticket/9946">Vaadin Bug #9946</a>.
186         */
187        @Override
188        protected VaadinServletService createServletService(DeploymentConfiguration deploymentConfiguration) {
189            return new VaadinServletService(this, deploymentConfiguration) {
190                @Override
191                protected VaadinServiceSession createVaadinSession(VaadinRequest request) throws ServiceException {
192                    VaadinServiceSession session = super.createVaadinSession(request);
193                    VaadinServiceSession.setCurrent(session);
194                    return session;
195                }
196            };
197        }
198    }
