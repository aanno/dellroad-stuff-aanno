022    /**
023     * Support superclass for read-only {@link Container} implementations where each {@link Item} in the container
024     * is backed by a Java object, and the Java objects are generated via {@linkplain #query a query} that returns
025     * a {@link QueryList} containing (some portion of) the backing objects. The container's item ID's are simply
026     * the indexes of the corresponding objects in this list.
027     *
028     * <p>
029     * This class will invoke {@link #query} as needed to (re)generate the query list. The query list is then cached.
030     * However, if any invocation of {@link QueryList#get} throws an {@link InvalidQueryListException}, then the cached
031     * list is discarded and {@link #query} is invoked again to regenerate it. In this way, the {@link QueryList} is
032     * allowed to decide on demand when it is invalid or incapable of providing a specific list member. For example,
033     * when using JPA, a list may be considered invalid if the current EntityManager session has changed.
034     * </p>
035     *
036     * <p>
037     * Note that the {@link QueryList} being invalid is an orthogonal concept from the contents of the list having changed.
038     * Invalid means "this list can no longer be used" while changed means "this list contains out-of-date information".
039     * Normally, the latter implies the former (but not vice-versa). The list becoming invalid does not in itself not cause
040     * any notifications to be sent, so no new query will be performed until e.g. the user interface explicitly requests
041     * more information.
042     * </p>
043     *
044     * <p>
045     * Therefore, if the list content changes, first invoke {@link #invalidate} to discard the cached {@link QueryList},
046     * and then {@link #fireItemSetChange} to notify listeners; for convenience, {@link #reload} will perform these two
047     * steps for you.
048     * </p>
049     *
050     * <p>
051     * The subclass may forcibly invalidate the current {@link QueryList} via {@link #invalidate}; this merely discards it
052     * and will force a new invocation of {@link #query} on the next container access. In many situations, however,
053     * the use of {@link #invalidate} is never required.
054     * </p>
055     *
056     * <p>
057     * For scalability reasons the {@link QueryList} may actually only contain a portion of the list, throwing
058     * {@link InvalidQueryListException}s when other list members are accessed.  An "index hint" parameter provided to
059     * {@link #query} indicates which member of the container is of current interest. The returned {@link QueryList}
060     * is required to provide exception-free access only to the indicated member, so in the extreme case only a single
061     * list member could be kept. In practice, normally range of members near to the index hint would be kept;
062     * see for example {@link WindowQueryList}.
063     * </p>
064     *
065     * @param <T> the type of the Java objects that back each {@link Item} in the container
066     * @see QueryList
067     * @see SimpleQueryList
068     * @see WindowQueryList
069     */
070    @SuppressWarnings("serial")
071    public abstract class AbstractQueryContainer<T> extends AbstractContainer implements Container.Ordered, Container.Indexed,
072      Container.PropertySetChangeNotifier, Container.ItemSetChangeNotifier {
073    
074        protected QueryList<? extends T> queryList;
075    
076        private final HashMap<String, PropertyDef<?>> propertyMap = new HashMap<String, PropertyDef<?>>();
077        private PropertyExtractor<? super T> propertyExtractor;

140        }
141    
142        /**
143         * Change the configured properties of this container.
144         *
145         * @param propertyDefs container property definitions; null is treated like the empty set
146         * @throws IllegalArgumentException if {@code propertyDefs} contains a property with a duplicate name
147         */
148        public void setProperties(Collection<? extends PropertyDef<?>> propertyDefs) {
149            if (propertyDefs == null)
150                propertyDefs = Collections.<PropertyDef<?>>emptySet();
151            this.propertyMap.clear();
152            for (PropertyDef<?> propertyDef : propertyDefs) {
153                if (this.propertyMap.put(propertyDef.getName(), propertyDef) != null)
154                    throw new IllegalArgumentException("duplicate property name `" + propertyDef.getName() + "'");
155            }
156            this.fireContainerPropertySetChange();
157        }
158    
159        /**
160         * Reload this container.
161         *
162         * <p>
163         * This discards the current cached {@link QueryList} (if any) and fires an item set change event.
164         */
165        public void reload() {
166            this.invalidate();
167            this.fireItemSetChange();
168        }

170    // Subclass hooks and methods
171    
172        /**
173         * Perform a query to (re)generate the list of Java objects backing this container.
174         *
175         * <p>
176         * The particular position in the list we are interested in is given as a hint by the {@code index} parameter.
177         * That is, an invocation of <code>{@link QueryList#get}(hint)</code> is likely immediately after this method
178         * returns and if so it must complete without throwing an exception.
179         * </p>
180         *
181         * <p>
182         * The {@code hint} can be used to implement a highly scalable query list containing external objects
183         * (such as from a database) where only a small "window" of objects is actually kept in memory at any one time.
184         * Of course, implementations are also free to ignore {@code hint}. However, the returned {@link QueryList}
185         * must at least tolerate one invocation of <code>{@link QueryList#get}(hint)</code> without throwing an exception,
186         * assuming that {@code hint} does not exceed the size of the list.
187         * </p>
188         *
189         * @param hint index of the list element we are interested in
190         * @return list of Java objects backing this container
191         */
192        protected abstract QueryList<? extends T> query(long hint);
193    
194        /**
195         * Discard the current cached {@link QueryList}, if any.
196         */
197        protected void invalidate() {
198            this.queryList = null;
199        }
200    
201        /**
202         * Get the Java backing object at the given index in the list.
203         *
204         * @return backing object, or null if {@code index} is out of range
205         */
206        protected T getJavaObject(int index) {
207            if (index < 0 || index >= this.ensureList(index).size())
208                return null;
209            try {
210                return this.queryList.get(index);
211            } catch (InvalidQueryListException e) {
212                this.invalidate();
213                try {
214                    return this.ensureList(index).get(index);
215                } catch (InvalidQueryListException e2) {
216                    throw new RuntimeException("query() was given a hint of " + index + " but QueryList.get(" + index + ") failed", e2);
217                }
218            }
219        }
220    
221        /**
222         * Ensure we have a cached query list.
223         *
224         * @param hint index of the list element we are interested in, passed to {@link #query} if no query list is cached
225         * @return cached query list, never null
226         */
227        protected QueryList<? extends T> ensureList(int hint) {
228            if (this.queryList == null)
229                this.queryList = this.query(hint);
230            return this.queryList;
231        }

235        @Override
236        public SimpleItem<T> getItem(Object itemId) {
237            if (!(itemId instanceof Integer))
238                return null;
239            int index = ((Integer)itemId).intValue();
240            T obj = this.getJavaObject(index);
241            if (obj == null)
242                return null;
243            return new SimpleItem<T>(obj, this.propertyMap, this.propertyExtractor);
244        }

352        @Override
353        public List<Integer> getItemIds(int startIndex, int numberOfItems) {
354            if (numberOfItems < 0)
355                throw new IllegalArgumentException("numberOfItems < 0");
356            long size = this.ensureList(startIndex).size();
357            if (startIndex < 0 || startIndex > size)
358                throw new IndexOutOfBoundsException("startIndex out of range");
359            if (startIndex + numberOfItems > size)
360                numberOfItems = (int)(size - startIndex);
361            return new IntList(startIndex, numberOfItems);
362        }

364        @Override
365        public int indexOfId(Object itemId) {
366            if (!(itemId instanceof Integer))
367                return -1;
368            int index = ((Integer)itemId).intValue();
369            if (index < 0 || index >= this.ensureList(index).size())
370                return -1;
371            return index;
372        }

489    // IntList
490    
491        private static class IntList extends AbstractList<Integer> {
492    
493            private final int min;
494            private final int size;
495    
496            public IntList(int size) {
497                this(0, size);
498            }
499    
500            public IntList(int min, int size) {
501                if (size < 0)
502                    throw new IllegalArgumentException("size < 0");
503                this.min = min;
504                this.size = size;
505            }
506    
507            @Override
508            public int size() {
509                return this.size;
510            }
511    
512            @Override
513            public Integer get(int index) {
514                if (index < 0 || index >= this.size)
515                    throw new IndexOutOfBoundsException();
516                return this.min + index;
517            }
518        }
