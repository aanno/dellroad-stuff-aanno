008    package org.dellroad.stuff.vaadin7;
009    
010    /**
011     * Cacheable list of Java objects backing an {@link AbstractQueryContainer}.
012     * Instances contain a list of Java objects (or some portion thereof), and may become invalid over time.
013     *
014     * @see AbstractQueryContainer
015     */
016    public interface QueryList<T> {
017    
018        /**
019         * Get the total size of this list.
020         *
021         * <p>
022         * For any given {@link QueryList} instance, this method is expected to return a the same value if invoked multiple times.
023         * Therefore, callers may safely choose to invoke it only once on a given instance and cache the result.
024         * </p>
025         */
026        long size();
027    
028        /**
029         * Get an item in the list, or throw an exception if this instance is no longer valid
030         * or cannot provide the item.
031         *
032         * @param index index of the item (zero-based)
033         * @throws IndexOutOfBoundsException if {@code index} is less than zero or greater than or equal to {@link #size}
034         * @throws InvalidQueryListException if this list has become invalid or cannot provide the item
035         */
036        T get(long index) throws InvalidQueryListException;
037    }
