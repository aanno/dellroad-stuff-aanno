008    package org.dellroad.stuff.vaadin7;
009    
010    import java.lang.annotation.Documented;
011    import java.lang.annotation.ElementType;
012    import java.lang.annotation.Inherited;
013    import java.lang.annotation.Retention;
014    import java.lang.annotation.RetentionPolicy;
015    import java.lang.annotation.Target;
016    
017    import org.dellroad.stuff.java.ErrorAction;
018    import org.springframework.beans.factory.annotation.Autowire;
019    
020    /**
021     * Indicates that the class is a candidate for configuration using the {@code VaadinConfigurableAspect} aspect.
022     *
023     * <p>
024     * Analogous to Spring's {@link org.springframework.beans.factory.annotation.Configurable @Configurable} annotation,
025     * but causes beans to be autowired into the Spring application context associated with the current
026     * {@link com.vaadin.server.VaadinSession} (aka "Vaadin application") by {@link SpringVaadinSession} instead of
027     * the Spring application context associated with the servlet context.
028     * </p>
029     *
030     * <p>
031     * For an extra safety check, consider setting {@link #ifSessionNotLocked} where appropriate.
032     * </p>
033     *
034     * <p>
035     * For the this annotation to function properly, {@link VaadinConfigurable @VaadinConfigurable} classes must be woven
036     * (either at build time or runtime) using the
037     * <a href="http://www.eclipse.org/aspectj/doc/released/faq.php#compiler">AspectJ compiler</a> with the
038     * {@code VaadinConfigurableAspect} aspect (included in the <code>dellroad-stuff</code> JAR file), and the
039     * {@link SpringVaadinServlet} must be used.
040     * </p>
041     *
042     * @see org.dellroad.stuff.vaadin
043     * @see SpringVaadinServlet
044     * @see SpringVaadinSession
045     * @see <a href="https://github.com/archiecobbs/dellroad-stuff-vaadin-spring-demo3/tree/vaadin7">Example Code on GitHub</a>
046     */
047    @Retention(RetentionPolicy.RUNTIME)
048    @Target(ElementType.TYPE)
049    @Documented
050    @Inherited
051    public @interface VaadinConfigurable {
052    
053        /**
054         * Configuration bean definition template name, if any.
055         */
056        String value() default "";
057    
058        /**
059         * Whether and how to automatically autowire dependencies.
060         */
061        Autowire autowire() default Autowire.NO;
062    
063        /**
064         * Whether to enable dependency checking.
065         */
066        boolean dependencyCheck() default false;
067    
068        /**
069         * Whether to inject dependencies prior to constructor execution.
070         */
071        boolean preConstruction() default false;
072    
073        /**
074         * What to do when we discover that the {@link com.vaadin.server.VaadinSession} is not
075         * {@linkplain com.vaadin.server.VaadinSession#getLockInstance locked} when the annotated bean is constructed.
076         * For beans that are (or will interact with) Vaadin widgets, containers, etc., this typically
077         * indicates a programming error. In such cases, this property configures what to do, if anything.
078         *
079         * @see com.vaadin.server.VaadinSession#getLockInstance
080         */
081        ErrorAction ifSessionNotLocked() default ErrorAction.IGNORE;
082    }
