008    package org.dellroad.stuff.io;
009    
010    import java.io.FilterOutputStream;
011    import java.io.IOException;
012    import java.io.OutputStream;
013    
014    /**
015     * Writes zero or more {@link OutputStream}s serialized inside an underlying {@link OutputStream}
016     * to be deserialized as distinct {@link java.io.InputStream}s on the other end by an {@link InputStreamReader}.
017     *
018     * <p>
019     * To use this class, invoke {@link #start} to start a new {@link OutputStream}, write to it by writing to this
020     * class normally, and then use {@link #stop} to close the current {@link OutputStream}. Another {@link OutputStream}
021     * is started by invoking {@link #start} again.
022     * </p>
023     *
024     * <p>
025     * Each {@link OutputStream} written in this way will be read as distinct {@link java.io.InputStream} by the
026     * {@link InputStreamReader} at the other end.
027     * </p>
028     *
029     * @see InputStreamReader
030     */
031    public class OutputStreamWriter extends FilterOutputStream {
032    
033        private final RandomEscape randomEscape = new RandomEscape();
034    
035        private int escape;
036        private boolean started;
037        private boolean closed;
038    
039        /**
040         * Constructor.
041         *
042         * @param output the underlying {@link OutputStream} that will carry nested {@link OutputStream}s within it
043         */
044        public OutputStreamWriter(OutputStream output) {
045            super(output);
046            this.escape = this.randomEscape.next();
047        }
048    
049        /**
050         * Start a new {@link OutputStream}.
051         *
052         * @throws IOException if this instance is closed
053         * @throws IOException if an {@link OutputStream} is already started
054         */
055        public synchronized void start() throws IOException {
056            if (this.closed)
057                throw new IOException("this instance is closed");
058            if (this.started)
059                throw new IOException("already started");
060            this.started = true;
061        }
062    
063        /**
064         * End the current {@link OutputStream}. This flushes the underlying output.
065         * A new {@link OutputStream} will be created upon the next invocation of {@link #start}.
066         *
067         * @throws IOException if this instance is closed
068         * @throws IOException if no {@link OutputStream} is currently started
069         * @throws IOException if the underlying {@link OutputStream} throws an exception
070         */
071        public synchronized void stop() throws IOException {
072            if (this.closed)
073                throw new IOException("this instance is closed");
074            if (!this.started)
075                throw new IOException("not started");
076            this.started = false;
077            this.writeControl(InputStreamReader.CONTROL_SEPARATOR);
078            this.flush();
079        }
080    
081        /**
082         * Close this instance. Does nothing if already closed.
083         * If there an {@link OutputStream} is already started when this method is invoked, it will be implicitly
084         * {@linkplain #stop stopped}.
085         *
086         * <p>
087         * This ends the current {@link OutputStream} and closes the underlying input.
088         * </p>
089         *
090         * @throws IOException if an there is an error closing the underlying {@link OutputStream}
091         */
092        @Override
093        public synchronized void close() throws IOException {
094            if (this.closed)
095                return;
096            if (this.started) {
097                this.started = false;
098                this.writeControl(InputStreamReader.CONTROL_SEPARATOR);
099            }
100            this.closed = true;
101            this.out.close();
102        }
103    
104        @Override
105        public synchronized void write(int ch) throws IOException {
106            if (this.closed)
107                throw new IOException("this instance is closed");
108            if (!this.started)
109                this.started = true;
110            if ((ch & 0xff) == this.escape) {
111                this.writeControl(InputStreamReader.CONTROL_ESCAPE);
112                return;
113            }
114            this.out.write(ch);
115        }
116    
117        private synchronized void writeControl(int control) throws IOException {
118            this.out.write(this.escape);
119            this.out.write(this.escape ^ control);
120            this.escape = this.randomEscape.next();
121        }
122    }
