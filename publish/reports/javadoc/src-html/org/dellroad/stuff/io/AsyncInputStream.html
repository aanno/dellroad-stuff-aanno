008    package org.dellroad.stuff.io;
009    
010    import java.io.IOException;
011    import java.io.InputStream;
012    
013    import org.slf4j.Logger;
014    import org.slf4j.LoggerFactory;
015    
016    /**
017     * Performs asynchonous reads on an {@link InputStream} and notifies of input events.
018     */
019    public class AsyncInputStream {
020    
021        private static final int BUFFER_SIZE = 4096;
022    
023        protected final Logger log = LoggerFactory.getLogger(getClass());
024    
025        private final InputStream input;
026        private final String name;
027        private final Listener listener;
028    
029        private boolean closed;                     // this instance has been close()'d
030    
031        /**
032         * Constructor.
033         *
034         * @param input     underlying input stream
035         * @param name      name for this instance; used to create the name of the background thread
036         * @param listener  callback object for input events
037         * @throws IllegalArgumentException if any parameter is null
038         */
039        public AsyncInputStream(InputStream input, String name, Listener listener) {
040            if (input == null)
041                throw new IllegalArgumentException("null input");
042            if (name == null)
043                throw new IllegalArgumentException("name input");
044            if (listener == null)
045                throw new IllegalArgumentException("null listener");
046            this.input = input;
047            this.name = name;
048            this.listener = listener;
049            new Thread(this.name) {
050    
051                @Override
052                public void run() {
053                    AsyncInputStream.this.threadMain();
054                }
055            }.start();
056        }
057    
058        /**
059         * Close this instance.
060         */
061        public synchronized void close() {
062            if (this.closed)
063                return;
064            try {
065                this.input.close();
066            } catch (IOException e) {
067                // ignore; we assume main thread will awake in any case
068            }
069            this.closed = true;
070        }
071    
072        /**
073         * Writer thread main entry point.
074         */
075        private void threadMain() {
076            try {
077                this.runLoop();
078            } catch (Throwable t) {
079                synchronized (this) {
080                    if (this.closed)
081                        return;
082                }
083                try {
084                    this.listener.handleException(t);
085                } catch (Exception e) {
086                    this.log.error(this.name + ": caught unexpected exception", e);
087                }
088            }
089        }
090    
091        /**
092         * Async reader thread main loop.
093         */
094        private void runLoop() throws IOException {
095            byte[] buf = new byte[BUFFER_SIZE];
096            while (true) {
097                int r = this.input.read(buf);
098                if (r == -1) {
099                    this.listener.handleEOF();
100                    break;
101                }
102                this.listener.handleInput(buf, 0, r);
103            }
104        }
105    
106        /**
107         * Callback interface required by {@link AsyncInputStream}.
108         */
109        public interface Listener {
110    
111            /**
112             * Handle new data read from the underlying input.
113             * This method must not write to buffer bytes outside of the defined region.
114             *
115             * @param buf data buffer
116             * @param off starting offset of data in buffer
117             * @param len number of bytes of data
118             */
119            void handleInput(byte[] buf, int off, int len);
120    
121            /**
122             * Handle an exception detected on the underlying input.
123             * No further events will be delivered.
124             *
125             * @param e the exception received (usually {@link IOException} but could also be any other {@link RuntimeException})
126             */
127            void handleException(Throwable e);
128    
129            /**
130             * Handle end-of-file detected on the underlying input.
131             * No further events will be delivered.
132             */
133            void handleEOF();
134        }
135    }
