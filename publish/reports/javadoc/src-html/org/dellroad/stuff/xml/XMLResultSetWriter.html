030    /**
031     * Outputs the result of an SQL query as XML.
032     *
033     * <p>
034     * For example, the query <code>SELECT ID, LAST_NAME, FIRST_NAME FROM EMPLOYEE</code> might become:
035     * <pre>
036     *  &lt;result-set&gt;
037     *      &lt;query&gt;&lt;![CDATA[SELECT ID, LAST_NAME, FIRST_NAME FROM EMPLOYEE]]&gt;&lt;/query&gt;
038     *      &lt;columns&gt;
039     *          &lt;column index="1" name="ID" precision="20" type="BIGINT" typeName="BIGINT" nullable="false"/&gt;
040     *          &lt;column index="2" name="LAST_NAME" precision="255" type="VARCHAR" typeName="VARCHAR" nullable="false"/&gt;
041     *          &lt;column index="3" name="FIRST_NAME" precision="255" type="VARCHAR" typeName="VARCHAR" nullable="false"/&gt;
042     *      &lt;/columns&gt;
043     *      &lt;data&gt;
044     *          &lt;row&gt;
045     *              &lt;column index="1"&gt;1302&lt;/column&gt;
046     *              &lt;column index="2"&gt;Washington&lt;/column&gt;
047     *              &lt;column index="3"&gt;George&lt;/column&gt;
048     *          &lt;/row&gt;
049     *          &lt;row&gt;
050     *              &lt;column index="1"&gt;1303&lt;/column&gt;
051     *              &lt;column index="2"&gt;Lincoln&lt;/column&gt;
052     *              &lt;column index="3"&gt;Abraham&lt;/column&gt;
053     *          &lt;/row&gt;
054     *          ...
055     *      &lt;/data&gt;
056     *  &lt;/result-set&gt;
057     * </pre>
058     * </p>
059     */
060    public class XMLResultSetWriter {
061    
062        /**
063         * Mapping from result set {@link Types} values to their corresponding names.
064         */
065        public static final Map<Integer, String> TYPE_NAMES;
066        static {
067            HashMap<Integer, String> map = new HashMap<Integer, String>();
068            for (Field field : Types.class.getDeclaredFields()) {
069                if (field.getType() != Integer.TYPE || field.getModifiers() != (Modifier.PUBLIC | Modifier.STATIC | Modifier.FINAL))
070                    continue;
071                try {
072                    map.put(field.getInt(null), field.getName());
073                } catch (IllegalAccessException e) {
074                    throw new RuntimeException(e);
075                }
076            }
077            TYPE_NAMES = Collections.unmodifiableMap(map);
078        }
079    
080        private static final String DATE_FORMAT = "yyyy-MM-dd";
081        private static final String TIMESTAMP_FORMAT = DATE_FORMAT + " HH:mm:ss";
082    
083        private final XMLStreamWriter writer;
084        private final String indentSpace;
085    
086        private State state = State.LAST_CLOSE;
087        private int nesting;
088    
089        // Tracks state of the output
090        private enum State {
091            LAST_OPEN,
092            LAST_CLOSE,
093            LAST_TEXT
094        };
095    
096        /**
097         * Constructor.
098         *
099         * @param writer where to write the XML
100         * @param indent number of spaces for indentation
101         * @throws IllegalArgumentException if {@code writer} is null
102         */
103        public XMLResultSetWriter(XMLStreamWriter writer, int indent) {
104            if (writer == null)
105                throw new IllegalArgumentException("null writer");
106            this.writer = writer;
107            char[] spaces = new char[indent];
108            Arrays.fill(spaces, ' ');
109            this.indentSpace = new String(spaces);
110        }
111    
112        /**
113         * Write the given {@link ResultSet} out within a <code>&lt;result-set&gt;</code> XML element.
114         *
115         * <p>
116         * This is a convenience method, equivalent to:
117         *  <blockquote><code>
118         *      write(null, resultSet)
119         *  </code></blockquote>
120         * element is included.
121         */
122        public void write(ResultSet resultSet) throws SQLException, XMLStreamException {
123            this.write(null, resultSet);
124        }
125    
126        /**
127         * Write the given {@link ResultSet} out within a <code>&lt;result-set&gt;</code> XML element including the query.
128         * The given SQL query will be present as a <code>&lt;query&gt;</code> first child element,
129         * followed by a <code>&lt;columns&gt;</code> element containing column meta-data, and finally
130         * the <code>&lt;data&gt;</code> element containing each returned row.
131         *
132         * <p>
133         * This method just outputs an XML element. To output an entire XML document, you would do something like this:
134         * <pre>
135         *  XMLResultSetWriter xmlResultSetWriter = new XMLResultSetWriter(writer, 4);
136         *  writer.writeStartDocument("UTF-8", "1.0");
137         *  xmlResultSetWriter.write(query, resultSet);
138         *  writer.writeEndDocument();
139         * </pre>
140         * </p>
141         *
142         * <p>
143         * The result set is iterated in a forward direction only.
144         *
145         * @param query the SQL query, or null to omit
146         * @param resultSet result of the query
147         * @throws XMLStreamException if there is an error writing out the XML
148         * @throws SQLException if there is an error reading from {@code resultSet}
149         */
150        public void write(String query, ResultSet resultSet) throws SQLException, XMLStreamException {
151    
152            // Start result set
153            this.openTag("result-set");
154    
155            // Emit query
156            if (query != null) {
157                this.openTag("query");
158                this.emitCData(query);
159                this.closeTag();
160            }
161    
162            // Emit column info
163            final ResultSetMetaData metaData = resultSet.getMetaData();
164            final int numColumns = metaData.getColumnCount();
165            this.openTag("columns");
166            for (int col = 1; col <= numColumns; col++) {
167                this.openTag("column");
168                this.writer.writeAttribute("index", "" + col);
169                this.writer.writeAttribute("name", metaData.getColumnName(col));
170                int precision = metaData.getPrecision(col);
171                if (precision != 0)
172                    this.writer.writeAttribute("precision", "" + precision);
173                int scale = metaData.getScale(col);
174                if (scale != 0)
175                    this.writer.writeAttribute("scale", "" + scale);
176                int type = metaData.getColumnType(col);
177                final String typeName = XMLResultSetWriter.TYPE_NAMES.get(type);
178                this.writer.writeAttribute("type", typeName != null ? typeName : "" + type);
179                this.writer.writeAttribute("typeName", metaData.getColumnTypeName(col));
180                String nullableDesc;
181                switch (metaData.isNullable(col)) {
182                case ResultSetMetaData.columnNullable:
183                    nullableDesc = "true";
184                    break;
185                case ResultSetMetaData.columnNoNulls:
186                    nullableDesc = "false";
187                    break;
188                default:
189                    nullableDesc = "unknown";
190                    break;
191                }
192                this.writer.writeAttribute("nullable", nullableDesc);
193                this.closeTag();
194            }
195            this.closeTag();
196    
197            // Emit result data
198            this.openTag("data");
199            while (resultSet.next()) {
200                this.openTag("row");
201                for (int col = 1; col <= numColumns; col++)
202                    this.writeDataColumn(resultSet, metaData.getColumnType(col), col);
203                this.closeTag();
204            }
205            this.closeTag();
206    
207            // End result set
208            this.closeTag();
209        }
210    
211    // Writing a data row
212    
213        private void writeDataColumn(ResultSet resultSet, int columnType, int col) throws SQLException, XMLStreamException {
214            Object value = null;
215            switch (columnType) {
216            case Types.BIT:
217            case Types.BOOLEAN:
218            {
219                boolean b = resultSet.getBoolean(col);
220                if (!resultSet.wasNull())
221                    value = "" + b;
222                break;
223            }
224            case Types.CLOB:
225            case Types.NCLOB:
226            case Types.LONGVARCHAR:
227            case Types.LONGNVARCHAR:
228            case Types.VARCHAR:
229            case Types.NVARCHAR:
230            case Types.CHAR:
231            case Types.NCHAR:
232            {
233                Reader reader = resultSet.getCharacterStream(col);
234                if (reader == null)
235                    break;
236                int len;
237                char[] buf = new char[1024];
238                this.openColumnTag(col);
239                try {
240                    while ((len = reader.read(buf)) != -1)
241                        this.emitText(buf, 0, len);
242                    reader.close();
243                } catch (IOException e) {
244                    throw (SQLException)new SQLException("error reading from CLOB").initCause(e);
245                }
246                this.closeTag();
247                return;
248            }
249            case Types.BIGINT:
250            case Types.DECIMAL:
251            case Types.DOUBLE:
252            case Types.FLOAT:
253            case Types.REAL:
254            case Types.NUMERIC:
255                value = resultSet.getBigDecimal(col);
256                break;
257            case Types.INTEGER:
258            case Types.TINYINT:
259            case Types.SMALLINT:
260                value = resultSet.getInt(col);
261                break;
262            case Types.DATE:
263            {
264                Date date = resultSet.getDate(col);
265                if (date != null)
266                    value = new SimpleDateFormat(DATE_FORMAT).format(date);
267                break;
268            }
269            case Types.TIME:
270                value = resultSet.getTime(col);
271                break;
272            case Types.TIMESTAMP:
273            {
274                Timestamp timestamp = resultSet.getTimestamp(col);
275                if (timestamp != null)
276                    value = new SimpleDateFormat(TIMESTAMP_FORMAT).format(timestamp);
277                break;
278            }
279            case Types.BLOB:
280            case Types.BINARY:
281            case Types.VARBINARY:
282            case Types.LONGVARBINARY:
283            {
284                InputStream input = resultSet.getBinaryStream(col);
285                if (input == null)
286                    break;
287                this.openColumnTag(col);
288                try {
289                    this.writeBinary(input);
290                    input.close();
291                } catch (IOException e) {
292                    throw (SQLException)new SQLException("error reading from BLOB").initCause(e);
293                }
294                this.closeTag();
295                return;
296            }
297            case Types.ROWID:
298            {
299                try {
300                    value = resultSet.getRowId(col);
301                } catch (AbstractMethodError e) {
302                    value = resultSet.getObject(col);
303                }
304                break;
305            }
306            // Unhandled cases...
307            case Types.ARRAY:
308            case Types.DATALINK:
309            case Types.DISTINCT:
310            case Types.JAVA_OBJECT:
311            case Types.NULL:
312            case Types.OTHER:
313            case Types.REF:
314            case Types.SQLXML:
315            case Types.STRUCT:
316            default:
317                value = resultSet.getObject(col);
318                break;
319            }
320    
321            // Handle null by omitting the tag entirely
322            if (value == null || resultSet.wasNull())
323                return;
324    
325            // Emit value
326            this.openColumnTag(col);
327            this.emitText(value.toString());
328            this.closeTag();
329        }
330    
331        private void writeBinary(InputStream input) throws IOException, XMLStreamException {
332            int len;
333            byte[] buf = new byte[1024];
334            char[] ch = new char[2];
335            while ((len = input.read(buf)) != -1) {
336                for (int i = 0; i < len; i++) {
337                    int val = buf[i] & 0xff;
338                    ch[0] = Character.forDigit(val >> 4, 16);
339                    ch[1] = Character.forDigit(val & 0xf, 16);
340                    this.emitText(ch, 0, 2);
341                }
342            }
343        }
344    
345    // XML output formatting
346    
347        private void openColumnTag(int col) throws XMLStreamException {
348            this.openTag("column");
349            this.emitAttr("index", "" + col);
350        }
351    
352        private void openTag(String name) throws XMLStreamException {
353            switch (this.state) {
354            case LAST_OPEN:
355                this.writer.writeCharacters("\n");
356                break;
357            case LAST_CLOSE:
358                break;
359            case LAST_TEXT:
360            default:
361                throw new RuntimeException("mixed element content not supported");
362            }
363            for (int i = 0; i < this.nesting; i++)
364                this.writer.writeCharacters(this.indentSpace);
365            this.writer.writeStartElement(name);
366            this.nesting++;
367            this.state = State.LAST_OPEN;
368        }
369    
370        private void emitAttr(String name, String value) throws XMLStreamException {
371            this.writer.writeAttribute(name, value);
372        }
373    
374        private void emitText(String content) throws XMLStreamException {
375            this.writer.writeCharacters(content);
376            this.state = State.LAST_TEXT;
377        }
378    
379        private void emitText(char[] buf, int off, int len) throws XMLStreamException {
380            this.writer.writeCharacters(buf, off, len);
381            this.state = State.LAST_TEXT;
382        }
383    
384        private void emitCData(String content) throws XMLStreamException {
385            this.writer.writeCData(content);
386            this.state = State.LAST_TEXT;
387        }
388    
389        private void closeTag() throws XMLStreamException {
390            this.nesting--;
391            if (this.state == State.LAST_CLOSE) {
392                for (int i = 0; i < this.nesting; i++)
393                    this.writer.writeCharacters(this.indentSpace);
394            }
395            this.writer.writeEndElement();
396            if (this.nesting > 0)
397                this.writer.writeCharacters("\n");
398            this.state = State.LAST_CLOSE;
399        }
