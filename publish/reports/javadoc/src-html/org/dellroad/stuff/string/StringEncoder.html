008    package org.dellroad.stuff.string;
009    
010    /**
011     * Encodes/decodes Java strings, escaping control and XML-invalid characters.
012     */
013    public final class StringEncoder {
014    
015        private static final char[] HEXDIGITS = "0123456789abcdef".toCharArray();
016    
017        private StringEncoder() {
018        }
019    
020        /**
021         * Encode a string, escaping control and XML-invalid characters.
022         * Whether tab, newline, and carriage return are escaped is optional;
023         * these are the three control characters that are valid inside XML documents.
024         * <p/>
025         * <p/>
026         * Characters are escaped using <code>&#92;uNNNN</code> notation like Java unicode characters,
027         * e.g., <code>0x001f</code> would appear in the encoded string as <code>&#92;u001f</code>.
028         * Normal Java backslash escapes are used for tab, newline, carriage return, backspace, and formfeed.
029         * Backslash characters are themselves encoded with a double backslash.
030         *
031         * @param value         string to encode (possibly null)
032         * @param escapeTABNLCR escape tab, newline, and carriage return characters as well
033         * @return the encoded version of {@code value}, or {@code null} if {@code value} was {@code null}
034         * @see #decode
035         * @see #isValidXMLChar
036         */
037        public static String encode(String value, boolean escapeTABNLCR) {
038            if (value == null)
039                return value;
040            StringBuilder buf = new StringBuilder(value.length() + 4);
041            final int limit = value.length();
042            for (int i = 0; i < limit; i++) {
043                final char ch = value.charAt(i);
044    
045                // Handle special escapes
046                switch (ch) {
047                    case '\\':
048                        buf.append('\\').append('\\');
049                        continue;
050                    case '\b':
051                        buf.append('\\').append('b');
052                        continue;
053                    case '\f':
054                        buf.append('\\').append('f');
055                        continue;
056                    case '\t':
057                        if (escapeTABNLCR) {
058                            buf.append('\\').append('t');
059                            continue;
060                        }
061                        break;
062                    case '\n':
063                        if (escapeTABNLCR) {
064                            buf.append('\\').append('n');
065                            continue;
066                        }
067                        break;
068                    case '\r':
069                        if (escapeTABNLCR) {
070                            buf.append('\\').append('r');
071                            continue;
072                        }
073                        break;
074                    default:
075                        break;
076                }
077    
078                // If character is an otherwise valid XML character, pass it through unchanged
079                if (isValidXMLChar(ch)) {
080                    buf.append(ch);
081                    continue;
082                }
083    
084                // Escape it using 4 digit hex
085                buf.append('\\');
086                buf.append('u');
087                for (int shift = 12; shift >= 0; shift -= 4)
088                    buf.append(HEXDIGITS[(ch >> shift) & 0x0f]);
089            }
090            return buf.toString();
091        }
092    
093        /**
094         * Decode a string encoded by {@link #encode}.
095         * <p/>
096         * <p/>
097         * The parsing is strict; any ill-formed backslash escape sequence (i.e., not of the form
098         * <code>&#92;uNNNN</code>, <code>\b</code>, <code>\t</code>, <code>\n</code>, <code>\f</code>, <code>\r</code>
099         * or <code>\\</code>) will cause an exception to be thrown.
100         *
101         * @param text string to decode (possibly null)
102         * @return the decoded version of {@code text}, or {@code null} if {@code text} was {@code null}
103         * @throws IllegalArgumentException if {@code text} contains an invalid escape sequence
104         * @see #encode
105         */
106        public static String decode(String text) {
107            if (text == null)
108                return null;
109            StringBuilder buf = new StringBuilder(text.length());
110            final int limit = text.length();
111            for (int i = 0; i < limit; i++) {
112                char ch = text.charAt(i);
113    
114                // Handle unescaped characters
115                if (ch != '\\') {
116                    buf.append(ch);
117                    continue;
118                }
119    
120                // Get next char
121                if (++i >= limit)
122                    throw new IllegalArgumentException("illegal trailing '\\' in encoded string");
123                ch = text.charAt(i);
124    
125                // Check for special escapes
126                switch (ch) {
127                    case '\\':
128                        buf.append('\\');
129                        continue;
130                    case 'b':
131                        buf.append('\b');
132                        continue;
133                    case 't':
134                        buf.append('\t');
135                        continue;
136                    case 'n':
137                        buf.append('\n');
138                        continue;
139                    case 'f':
140                        buf.append('\f');
141                        continue;
142                    case 'r':
143                        buf.append('\r');
144                        continue;
145                    default:
146                        break;
147                }
148    
149                // Must be unicode escape
150                if (ch != 'u')
151                    throw new IllegalArgumentException("illegal escape sequence '\\" + ch + "' in encoded string");
152    
153                // Decode hex value
154                int value = 0;
155                for (int j = 0; j < 4; j++) {
156                    if (++i >= limit)
157                        throw new IllegalArgumentException("illegal truncated '\\u' escape sequence in encoded string");
158                    int nibble = Character.digit(text.charAt(i), 16);
159                    if (nibble == -1) {
160                        throw new IllegalArgumentException(
161                          "illegal escape sequence '" + text.substring(i - j - 2, i - j + 4) + "' in encoded string");
162                    }
163                    assert nibble >= 0 && nibble <= 0xf;
164                    value = (value << 4) | nibble;
165                }
166    
167                // Append decodec character
168                buf.append((char)value);
169            }
170            return buf.toString();
171        }
172    
173        /**
174         * Enquote a string. Functions like {@link #encode encode(string, true)} but in addition the resulting
175         * string is surrounded by double quotes and double quotes in the string are backslash-escaped.
176         * <p/>
177         * <p>
178         * Note: the strings returned by this method are not suitable for decoding by {@link #decode}.
179         * </p>
180         */
181        public static String enquote(String string) {
182            return '"' + encode(string, true).replaceAll("\"", "\\\"") + '"';
183        }
184    
185        /**
186         * Enquote bytes, treating them as an ASCII string.
187         *
188         * @see #enquote(String)
189         */
190        public static String enquote(byte[] data, int off, int len) {
191            char[] chars = new char[len];
192            for (int i = 0; i < len; i++)
193                chars[i] = (char)(data[i] & 0xff);
194            return enquote(new String(chars));
195        }
196    
197        /**
198         * Determine if the given character is a valid XML character according to the XML 1.0 specification.
199         * <p/>
200         * <p>
201         * Valid characters are tab, newline, carriage return, and characters in the ranges
202         * <code>&#92;u0020 - &#92;ud7ff</code> and <code>&#92;ue000 - &#92;fffdf</code> (inclusive).
203         * </p>
204         *
205         * @see <a href="http://www.w3.org/TR/REC-xml/#charsets">The XML 1.0 Specification</a>
206         */
207        public static boolean isValidXMLChar(char ch) {
208            return (ch >= '\u0020' && ch <= '\ud7ff') || ch == '\n' || ch == '\r' || ch == '\t' || (ch >= '\ue000' && ch <= '\ufffd');
209        }
210    }
