008    package org.dellroad.stuff.string;
009    
010    import java.math.BigInteger;
011    import java.util.Comparator;
012    
013    /**
014     * Comparator for version number parts.
015     *
016     * <p>
017     * This comparator puts numerical parts before non-numerical parts.
018     * Then, numerical parts are compared numerically, and non-numerical parts are compared lexically.
019     * </p>
020     *
021     * @see VersionComparator
022     */
023    public class VersionPartComparator implements Comparator<String> {
024    
025        @Override
026        public int compare(String part1, String part2) {
027            BigInteger num1 = part1.matches("[0-9]+") ? new BigInteger(part1, 10) : null;
028            BigInteger num2 = part2.matches("[0-9]+") ? new BigInteger(part2, 10) : null;
029            if (num1 != null && num2 != null)
030                return num1.compareTo(num2);
031            else if (num1 != null)
032                return 1;
033            else if (num2 != null)
034                return -1;
035            else
036                return part1.compareTo(part2);
037        }
038    }
