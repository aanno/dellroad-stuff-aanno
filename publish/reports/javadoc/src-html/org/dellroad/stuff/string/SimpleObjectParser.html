008    package org.dellroad.stuff.string;
009    
010    import java.beans.BeanInfo;
011    import java.beans.IndexedPropertyDescriptor;
012    import java.beans.IntrospectionException;
013    import java.beans.Introspector;
014    import java.beans.PropertyDescriptor;
015    import java.lang.reflect.Method;
016    import java.util.Collections;
017    import java.util.HashMap;
018    import java.util.Map;
019    import java.util.regex.Matcher;
020    import java.util.regex.Pattern;
021    import java.util.regex.PatternSyntaxException;
022    
023    import org.dellroad.stuff.java.Primitive;
024    
025    /**
026     * Parses strings using regular expressions into new instances of some class by parsing substrings.
027     * Primitive and String values are handled automatically. Other property types can be handled by
028     * overriding {@link #setProperty}.
029     *
030     * <a name="namedgroup"/>
031     * <p>
032     * This class supports parsing using a <i>named group regular expression</i>, which is pattern string
033     * using normal {@link Pattern} regular expression syntax with one additional grouping construct of the
034     * form <code>({property}...)</code>, allowing the Java bean property name to be specified inside the
035     * curly braces at the start of a grouped subexpression.
036     *
037     * <p>
038     * Instances of this class are immutable and thread-safe.
039     */
040    public class SimpleObjectParser<T> {
041    
042        private final Class<T> targetClass;
043        private final HashMap<String, PropertyDescriptor> propertyMap = new HashMap<String, PropertyDescriptor>();
044    
045        /**
046         * Constructor.
047         *
048         * @param targetClass class whose instances we will create
049         * @throws IllegalArgumentException if targetClass has no default constructor
050         */
051        public SimpleObjectParser(Class<T> targetClass) {
052            this.targetClass = targetClass;
053            try {
054                targetClass.newInstance();
055            } catch (Exception e) {
056                throw new IllegalArgumentException(targetClass + " has no usable default constructor");
057            }
058            buildPropertyMap();
059        }
060    
061        /**
062         * Get the target class.
063         */
064        public Class<T> getTargetClass() {
065            return this.targetClass;
066        }
067    
068        /**
069         * Parse the given text using the provided <i>named group regular expression</i>.
070         *
071         * <p>
072         * This method assumes the following about {@code regex}:
073         * <ul>
074         * <li>All instances of an opening parenthesis not preceded by a backslash are actual grouped sub-expressions
075         * <li>In particular, all instances of substrings like <code>({foo}</code> are actual named group sub-expressions
076         * </ul>
077         *
078         * @param text                string to parse
079         * @param regex               named group regular expression containing object property names
080         * @param allowSubstringMatch if false, entire text must match, otherwise only a (the first) substring need match
081         * @return parsed object or null if parse fails
082         * @throws PatternSyntaxException if the regular expression with the named group property names removed is invalid
083         * @throws PatternSyntaxException if this method cannot successfully parse the regular expression
084         * @throws IllegalArgumentException if a named group specfiies a property that is not a parseable
085         *                                  property of this instance's target class
086         *
087         * @since 1.0.85
088         */
089        public T parse(String text, String regex, boolean allowSubstringMatch) {
090    
091            // Scan regular expression for named sub-groups and parse them out
092            HashMap<Integer, String> patternMap = new HashMap<Integer, String>();
093            StringBuilder buf = new StringBuilder(regex.length());
094            Pattern namedGroup = Pattern.compile("\\(\\{(\\p{javaJavaIdentifierStart}\\p{javaJavaIdentifierPart}*)\\}");
095            Matcher matcher = namedGroup.matcher(regex);
096            int pos = 0;
097            int groupCount = 0;
098            while (true) {
099                int match = matcher.find(pos) ? matcher.start() : regex.length();
100                String chunk = regex.substring(pos, match);
101                for (int i = 0; i < chunk.length(); i++) {
102                    if (chunk.charAt(i) == '(' && (i == 0 || chunk.charAt(i - 1) != '\\'))
103                        groupCount++;
104                }
105                buf.append(chunk);
106                if (match == regex.length())
107                    break;
108                buf.append('(');
109                patternMap.put(++groupCount, matcher.group(1));
110                pos = matcher.end();
111            }
112    
113            // Sanity check our parse attempt
114            Pattern pattern = Pattern.compile(buf.toString());
115            if (pattern.matcher("").groupCount() != groupCount)
116                throw new PatternSyntaxException("can't parse named group regular expression", regex, 0);
117    
118            // Proceed
119            return this.parse(text, pattern, patternMap, allowSubstringMatch);
120        }
121    
122        /**
123         * Parse the given text using the provided pattern and mapping from pattern sub-group to Java bean property name.
124         *
125         * @param text                string to parse
126         * @param pattern             pattern with substring matching groups that match object properties
127         * @param patternMap          mapping from pattern substring group index to object property name
128         * @param allowSubstringMatch if false, entire text must match, otherwise only a (the first) substring need match
129         * @return parsed object or null if parse fails
130         * @throws IllegalArgumentException if the map contains a property that is not a parseable
131         *                                  property of this instance's target class
132         * @throws IllegalArgumentException if a subgroup index key in patternMap is out of bounds
133         */
134        public T parse(String text, Pattern pattern, Map<Integer, String> patternMap, boolean allowSubstringMatch) {
135    
136            // Compose given map with target class' property map
137            HashMap<Integer, PropertyDescriptor> subgroupMap = new HashMap<Integer, PropertyDescriptor>();
138            for (Map.Entry<Integer, String> entry : patternMap.entrySet()) {
139                String propName = entry.getValue();
140                PropertyDescriptor property = this.propertyMap.get(propName);
141                if (property == null)
142                    throw new IllegalArgumentException("parseable property \"" + propName + "\" not found in " + this.targetClass);
143                subgroupMap.put(entry.getKey(), property);
144            }
145    
146            // Attempt to match the string
147            Matcher matcher = pattern.matcher(text);
148            boolean matches = allowSubstringMatch ? matcher.find() : matcher.matches();
149            if (!matches)
150                return null;
151    
152            // Create instance
153            T obj;
154            try {
155                obj = targetClass.newInstance();
156            } catch (Exception e) {
157                throw new RuntimeException("unexpected exception", e);
158            }
159    
160            // Set fields based on matching substrings
161            for (Map.Entry<Integer, PropertyDescriptor> entry : subgroupMap.entrySet()) {
162    
163                // Get substring
164                String substring;
165                try {
166                    substring = matcher.group(entry.getKey());
167                } catch (IndexOutOfBoundsException e) {
168                    throw new IllegalArgumentException(
169                      "regex subgroup " + entry.getKey() + " does not exist in pattern `" + pattern + "'");
170                }
171    
172                // If substring was not matched, don't set property
173                if (substring == null)
174                    continue;
175    
176                // Set property from substring
177                this.setProperty(obj, entry.getValue(), substring);
178            }
179    
180            // Post-process
181            this.postProcess(obj);
182    
183            // Done
184            return obj;
185        }
186    
187        /**
188         * Get the mapping from property name to setter method.
189         */
190        public Map<String, PropertyDescriptor> getPropertyMap() {
191            return Collections.unmodifiableMap(this.propertyMap);
192        }
193    
194        /**
195         * Set a property value.
196         * <p/>
197         * <p>
198         * The implementation in {@link SimpleObjectParser} simply invokes {@link #setSimpleProperty}.
199         * Other property types can be handled by overriding this method.
200         * </p>
201         *
202         * @param obj       newly created instance
203         * @param property  descriptor for the property being set
204         * @param substring matched substring
205         * @throws IllegalArgumentException if substring cannot be successfully parsed
206         * @throws IllegalArgumentException if an exception is thrown attempting to set the property
207         */
208        protected void setProperty(T obj, PropertyDescriptor property, String substring) {
209            setSimpleProperty(obj, property, substring);
210        }
211    
212        /**
213         * Set a primitive or string property value.
214         * <p/>
215         * <p>
216         * The implementation in {@link SimpleObjectParser} handles primitives using the corresponding
217         * {@code valueOf} method; String values are handled by setting the value directly.
218         * </p>
219         *
220         * @throws IllegalArgumentException if property is not a primitive or String property.
221         * @throws IllegalArgumentException if substring cannot be successfully parsed (if primitive)
222         * @throws IllegalArgumentException if an exception is thrown attempting to set the property
223         */
224        protected void setSimpleProperty(T obj, PropertyDescriptor property, String substring) {
225    
226            // Parse substring
227            Object value;
228            if (property.getPropertyType() == String.class)
229                value = substring;
230            else {
231                Primitive prim = Primitive.get(property.getPropertyType());
232                if (prim == null) {
233                    throw new IllegalArgumentException(
234                      "property `" + property.getName() + "' of " + this.targetClass + " is not a primitive or String");
235                }
236                value = prim.parseValue(substring);
237            }
238    
239            // Set value
240            try {
241                property.getWriteMethod().invoke(obj, value);
242            } catch (Exception e) {
243                throw new IllegalArgumentException("can't set property `" + property.getName() + "' of " + this.targetClass, e);
244            }
245        }
246    
247        /**
248         * Post-process newly created instances. The instance's properties will have already been
249         * set by a successful parse.
250         * <p/>
251         * <p>
252         * The implementation in {@link SimpleObjectParser} does nothing. Subclasses may override if needed.
253         * </p>
254         */
255        protected void postProcess(T obj) {
256        }
257    
258        private void buildPropertyMap() {
259    
260            // Introspect target class
261            BeanInfo beanInfo;
262            try {
263                beanInfo = Introspector.getBeanInfo(this.targetClass);
264            } catch (IntrospectionException e) {
265                throw new RuntimeException(e);
266            }
267    
268            // Build map from property name -> setter method
269            for (PropertyDescriptor property : beanInfo.getPropertyDescriptors()) {
270                if (property instanceof IndexedPropertyDescriptor)
271                    continue;
272                Method setter = property.getWriteMethod();
273                if (setter == null)
274                    continue;
275                Class<?> type = property.getPropertyType();
276                this.propertyMap.put(property.getName(), property);
277            }
278        }
279    }
