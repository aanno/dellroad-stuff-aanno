008    package org.dellroad.stuff.pobj;
009    
010    import java.io.IOException;
011    import java.io.InputStream;
012    import java.util.Properties;
013    
014    import javax.xml.transform.Transformer;
015    import javax.xml.transform.TransformerException;
016    import javax.xml.transform.TransformerFactory;
017    import javax.xml.transform.stream.StreamSource;
018    
019    import org.dellroad.stuff.xml.TransformErrorListener;
020    import org.slf4j.LoggerFactory;
021    import org.springframework.core.io.Resource;
022    
023    /**
024     * {@link SpringPersistentObjectSchemaUpdate} that applies a configured XSL transform to the XML form of the persistent object.
025     *
026     * <p>
027     * The {@link #setTransform transform} property is required.
028     *
029     * <p>
030     * See {@link SpringPersistentObjectSchemaUpdater} for a Spring configuration example.
031     *
032     * @param <T> type of the persistent object
033     */
034    public class SpringXSLPersistentObjectSchemaUpdate<T> extends SpringPersistentObjectSchemaUpdate<T> {
035    
036        private Resource transformResource;
037        private Properties parameters;
038    
039        @Override
040        public void afterPropertiesSet() throws Exception {
041            super.afterPropertiesSet();
042            if (this.transformResource == null)
043                throw new Exception("no transform configured");
044        }
045    
046        /**
047         * Configure the XSLT transform as a resource.
048         */
049        public void setTransform(Resource transformResource) {
050            this.transformResource = transformResource;
051        }
052    
053        /**
054         * Configure XSLT parameters. This is an optional property.
055         */
056        public void setParameters(Properties properties) {
057            this.parameters = parameters;
058        }
059    
060        /**
061         * Apply this update to the given transaction.
062         */
063        @Override
064        public void apply(PersistentFileTransaction transaction) {
065    
066            // Get transform source
067            InputStream input;
068            try {
069                input = this.transformResource.getInputStream();
070            } catch (IOException e) {
071                throw new RuntimeException(e);
072            }
073            try {
074    
075                // Setup transformer
076                Transformer transformer = TransformerFactory.newInstance().newTransformer(
077                  new StreamSource(input, this.transformResource.getURI().toString()));
078                transformer.setErrorListener(new TransformErrorListener(LoggerFactory.getLogger(this.getClass()), true));
079                if (this.parameters != null) {
080                    for (String name : this.parameters.stringPropertyNames())
081                        transformer.setParameter(name, this.parameters.getProperty(name));
082                }
083    
084                // Do the transform
085                transaction.transform(transformer);
086    
087            } catch (IOException e) {
088                throw new RuntimeException(e);
089            } catch (TransformerException e) {
090                throw new RuntimeException(e);
091            } catch (RuntimeException e) {
092                if (e.getCause() instanceof TransformerException && e.getCause().getCause() instanceof RuntimeException)
093                    e = (RuntimeException)e.getCause().getCause();
094                throw e;
095            } finally {
096                try {
097                    input.close();
098                } catch (IOException e) {
099                    // ignore
100                }
101            }
102        }
103    }
