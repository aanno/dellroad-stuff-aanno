008    package org.dellroad.stuff.pobj;
009    
010    import java.io.IOException;
011    import java.util.Set;
012    
013    import javax.validation.ConstraintViolation;
014    import javax.xml.transform.Result;
015    import javax.xml.transform.Source;
016    
017    /**
018     * Delegate interface required for {@link PersistentObject}s.
019     * Instances provide methods for converting to/from XML, validation, etc.
020     *
021     * @param <T> type of the root persistent object
022     * @see PersistentObject
023     */
024    public interface PersistentObjectDelegate<T> {
025    
026        /**
027         * Serialize a root object into XML.
028         *
029         * @param obj object to serialize; must not be modified
030         * @param result XML destination
031         * @throws PersistentObjectException if an error occurs
032         */
033        void serialize(T obj, Result result) throws IOException;
034    
035        /**
036         * Deserialize a root object from XML.
037         *
038         * @param source XML source
039         * @return deserialized object
040         * @throws PersistentObjectException if an error occurs
041         */
042        T deserialize(Source source) throws IOException;
043    
044        /**
045         * Make a deep copy of the given object.
046         *
047         * @throws IllegalArgumentException if {@code original} is null
048         * @throws PersistentObjectException if an error occurs
049         */
050        T copy(T original);
051    
052        /**
053         * Validate the given object.
054         *
055         * @throws IllegalArgumentException if {@code obj} is null
056         * @throws PersistentObjectException if validation fails
057         */
058        Set<ConstraintViolation<T>> validate(T obj);
059    
060        /**
061         * Handle an exception thrown during a delayed write-back attempt. {@link ThreadDeath} exceptions are not
062         * passed to this method, but all others are.
063         *
064         * @param pobj the instance that encountered the exception
065         * @param t the exception thrown
066         */
067        void handleWritebackException(PersistentObject<T> pobj, Throwable t);
068    }
