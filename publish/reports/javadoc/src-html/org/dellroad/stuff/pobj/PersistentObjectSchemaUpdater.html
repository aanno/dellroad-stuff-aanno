032    /**
033     * Support superclass for {@link PersistentObject} schema updaters.
034     *
035     * <p>
036     * This class holds a nested {@link PersistentObject} and ensures that it's up to date when started.
037     * Use {@link #getPersistentObject} to access it.
038     *
039     * <p>
040     * Updates are tracked by "secretly" inserting <code>{@link #UPDATES_ELEMENT_NAME &lt;pobj:updates&gt;}</code>
041     * elements into the serialized XML document; these updates are transparently removed when the document is read back.
042     * In this way the document and its set of applied updates always travel together. For example:
043     *
044     * <blockquote><pre>
045     *  &lt;MyConfig&gt;
046     *      <b>&lt;pobj:updates xmlns:pobj="http://dellroad-stuff.googlecode.com/ns/persistentObject"&gt;
047     *          &lt;pobj:update>some-update-name-1&lt;/pobj:update&gt;
048     *          &lt;pobj:update>some-update-name-2&lt;/pobj:update&gt;
049     *      &lt;/pobj:updates&gt;</b>
050     *      &lt;username&gt;admin&lt;/username&gt;
051     *      &lt;password&gt;secret&lt;/password&gt;
052     *  &lt;/MyConfig&gt;
053     * </pre></blockquote>
054     *
055     * <p>
056     * Subclasses will typically override {@link #getInitialValue} for when there is no persistent file yet.
057     *
058     * @param <T> type of the root persistent object
059     */
060    public class PersistentObjectSchemaUpdater<T> extends AbstractSchemaUpdater<File, PersistentFileTransaction> {
061    
062        /**
063         * XML namespace URI used for nested update elements.
064         */
065        public static final String NAMESPACE_URI = "http://dellroad-stuff.googlecode.com/ns/persistentObject";
066    
067        /**
068         * Preferred XML namespace prefix for {@link #NAMESPACE_URI} elements.
069         */
070        public static final String XML_PREFIX = "pobj";
071    
072        /**
073         * XML element name for the updates list.
074         */
075        public static final QName UPDATES_ELEMENT_NAME = new QName(NAMESPACE_URI, "updates", XML_PREFIX);
076    
077        /**
078         * XML element name for a single update.
079         */
080        public static final QName UPDATE_ELEMENT_NAME = new QName(NAMESPACE_URI, "update", XML_PREFIX);
081    
082        /**
083         * XML namespace URI used for namespace declarations.
084         */
085        public static final QName XMLNS_ATTRIBUTE_NAME = new QName("http://www.w3.org/2000/xmlns/", XML_PREFIX, "xmlns");
086    
087        /**
088         * Default check interval for "out-of-band" updates to the persistent file ({@value}ms).
089         */
090        public static final long DEFAULT_CHECK_INTERVAL = 1000;
091    
092        protected File file;
093        protected long writeDelay;
094        protected long checkInterval = DEFAULT_CHECK_INTERVAL;
095        protected int numBackups;
096        protected boolean allowEmptyStart;
097        protected boolean allowEmptyStop;
098        protected PersistentObjectDelegate<T> delegate;
099    
100        private ArrayList<String> updateNames;
101        private PersistentObject<T> persistentObject;
102    
103        /**
104         * Configure the file used to store this object persistently. Required property.
105         */
106        public void setFile(File file) {
107            this.file = file;
108        }
109    
110        /**
111         * Configure the maximum delay after an update operation before a write-back to the persistent file
112         * must be initiated. Default is zero.
113         */
114        public void setWriteDelay(long writeDelay) {
115            this.writeDelay = writeDelay;
116        }
117    
118        /**
119         * Configure the check interval for "out-of-band" updates to the persistent file.
120         * Default is {@link #DEFAULT_CHECK_INTERVAL}.
121         */
122        public void setCheckInterval(long checkInterval) {
123            this.checkInterval = checkInterval;
124        }
125    
126        /**
127         * Configure the {@link PersistentObjectDelegate}. Required property.
128         */
129        public void setDelegate(PersistentObjectDelegate<T> delegate) {
130            this.delegate = delegate;
131        }
132    
133        /**
134         * Configure the number of backups to make of the persistent file.
135         *
136         * @see PersistentObject#getNumBackups
137         */
138        public void setNumBackups(int numBackups) {
139            this.numBackups = numBackups;
140        }
141    
142        /**
143         * Configure whether to all "empty starts". Default is false.
144         *
145         * @see PersistentObject
146         */
147        public void setAllowEmptyStart(boolean allowEmptyStart) {
148            this.allowEmptyStart = allowEmptyStart;
149        }
150    
151        /**
152         * Configure whether to all "empty stops". Default is false.
153         *
154         * @see PersistentObject
155         */
156        public void setAllowEmptyStop(boolean allowEmptyStop) {
157            this.allowEmptyStop = allowEmptyStop;
158        }
159    
160        /**
161         * Start this instance. Does nothing if already started.
162         *
163         * @throws IllegalArgumentException if an invalid file, write delay, or delegate is configured
164         * @throws PersistentObjectException if an error occurs
165         */
166        public synchronized void start() {
167    
168            // Already started?
169            if (this.persistentObject != null)
170                return;
171    
172            // Sanity check
173            if (this.file == null)
174                throw new IllegalArgumentException("no file configured");
175            if (this.writeDelay < 0)
176                throw new IllegalArgumentException("negative writeDelay configured");
177            if (this.delegate == null)
178                throw new IllegalArgumentException("no delegate configured");
179            if (this.numBackups < 0)
180                throw new IllegalArgumentException("negative numBackups configured");
181    
182            // Create persistent object
183            this.persistentObject = new PersistentObject<T>(new UpdaterDelegate(), this.file, this.writeDelay, this.checkInterval);
184            this.persistentObject.setNumBackups(this.numBackups);
185            this.persistentObject.setAllowEmptyStart(this.allowEmptyStart);
186            this.persistentObject.setAllowEmptyStop(this.allowEmptyStop);
187    
188            // Do schema updates
189            boolean success = false;
190            try {
191                this.initializeAndUpdateDatabase(this.file);
192                success = true;
193            } catch (RuntimeException e) {
194                throw e;
195            } catch (Exception e) {
196                throw new PersistentObjectException(e);
197            } finally {
198                if (!success)
199                    this.persistentObject = null;
200            }
201    
202            // Start persistent object
203            this.persistentObject.start();
204        }
205    
206        /**
207         * Stop this instance. Does nothing if already stopped.
208         *
209         * @throws PersistentObjectException if a delayed write back is pending and error occurs during writing
210         */
211        public synchronized void stop() {
212    
213            // Already stopped?
214            if (this.persistentObject == null)
215                return;
216    
217            // Stop
218            this.persistentObject.stop();
219            this.persistentObject = null;
220        }
221    
222        /**
223         * Get the {@link PersistentObject}.
224         *
225         * @throws IllegalStateException if this instance is not started
226         */
227        public synchronized PersistentObject<T> getPersistentObject() {
228            if (this.persistentObject == null)
229                throw new IllegalStateException("not started");
230            return this.persistentObject;
231        }
232    
233        /**
234         * Get the initial value for the persistent object when no persistent file is found.
235         *
236         * <p>
237         * The implementation in {@link PersistentObjectSchemaUpdater} just returns null, which leaves the
238         * initial root object unset. Subclasses should override as desired to provide an initial value.
239         *
240         * <p>
241         * The returned value must properly validate.
242         */
243        protected T getInitialValue() {
244            return null;
245        }
246    
247        @Override
248        protected boolean databaseNeedsInitialization(PersistentFileTransaction transaction) throws Exception {
249            return transaction.getData() == null;
250        }
251    
252        @Override
253        @SuppressWarnings("unchecked")
254        protected void initializeDatabase(PersistentFileTransaction transaction) throws Exception {
255    
256            // Get initial value
257            T initialValue = this.getInitialValue();
258            if (initialValue == null)
259                return;
260    
261            // Validate it
262            Set<ConstraintViolation<T>> violations = this.delegate.validate(initialValue);
263            if (!violations.isEmpty())
264                throw new PersistentObjectValidationException((Set<ConstraintViolation<?>>)(Object)violations);
265    
266            // Serialize it
267            ByteArrayOutputStream buffer = new ByteArrayOutputStream(PersistentFileTransaction.FILE_BUFFER_SIZE);
268            StreamResult result = new StreamResult(buffer);
269            this.delegate.serialize(initialValue, result);
270    
271            // Set it in the transaction
272            transaction.setData(buffer.toByteArray());
273        }
274    
275        @Override
276        protected PersistentFileTransaction openTransaction(File file) throws Exception {
277            return new PersistentFileTransaction(file);
278        }
279    
280        @Override
281        protected void commitTransaction(PersistentFileTransaction transaction) throws Exception {
282            this.updateNames = new ArrayList<String>(transaction.getUpdates());
283            transaction.commit();
284        }
285    
286        @Override
287        protected void rollbackTransaction(PersistentFileTransaction transaction) throws Exception {
288            transaction.rollback();
289        }
290    
291        @Override
292        protected Set<String> getAppliedUpdateNames(PersistentFileTransaction transaction) throws Exception {
293            return new HashSet<String>(transaction.getUpdates());
294        }
295    
296        @Override
297        protected void recordUpdateApplied(PersistentFileTransaction transaction, String name) throws Exception {
298            transaction.addUpdate(name);
299        }
300    
301        // Our PersistentObjectDelegate that hides the updates when (de)serializing
302        private class UpdaterDelegate extends FilterDelegate<T> {
303    
304            private final XMLInputFactory xmlInputFactory = XMLInputFactory.newFactory();
305            private final XMLOutputFactory xmlOutputFactory = XMLOutputFactory.newFactory();
306    
307            UpdaterDelegate() {
308                super(PersistentObjectSchemaUpdater.this.delegate);
309            }
310    
311            /**
312             * Serialize object to XML, adding update list.
313             */
314            @Override
315            public void serialize(T obj, Result result) throws IOException {
316                try {
317                    XMLEventWriter eventWriter = this.xmlOutputFactory.createXMLEventWriter(result);
318                    UpdatesXMLEventWriter updatesWriter = new UpdatesXMLEventWriter(eventWriter,
319                      PersistentObjectSchemaUpdater.this.updateNames);
320                    super.serialize(obj, new StAXResult(updatesWriter));
321                    updatesWriter.close();
322                } catch (IOException e) {
323                    throw e;
324                } catch (XMLStreamException e) {
325                    throw new PersistentObjectException(e);
326                }
327            }
328    
329            /**
330             * Deserialize object from XML, removing update list.
331             */
332            @Override
333            public T deserialize(Source source) throws IOException {
334                try {
335                    XMLEventReader eventReader = this.xmlInputFactory.createXMLEventReader(source);
336                    UpdatesXMLEventReader updatesReader = new UpdatesXMLEventReader(eventReader);
337                    return super.deserialize(new StAXSource(updatesReader));
338                } catch (IOException e) {
339                    throw e;
340                } catch (XMLStreamException e) {
341                    throw new PersistentObjectException(e);
342                }
343            }
344        }
345    }
