008    package org.dellroad.stuff.pobj;
009    
010    import java.io.ByteArrayInputStream;
011    import java.io.IOException;
012    import java.util.HashSet;
013    import java.util.List;
014    import java.util.Set;
015    
016    import javax.validation.ConstraintViolation;
017    import javax.xml.namespace.QName;
018    import javax.xml.stream.XMLEventWriter;
019    import javax.xml.stream.XMLOutputFactory;
020    import javax.xml.transform.Result;
021    import javax.xml.transform.Source;
022    import javax.xml.transform.stax.StAXResult;
023    import javax.xml.transform.stream.StreamSource;
024    
025    import org.dellroad.stuff.schema.AbstractSchemaUpdater;
026    
027    /**
028     * A {@link PersistentObjectDelegate} that is also a {@link AbstractSchemaUpdater} that automatically
029     * applies needed updates to the persistent XML file.
030     *
031     * <p>
032     * To use this class, provide your normal delegate to the constructor and use this one instead. This will augment
033     * the serialization and deserialization process to keep track of which updates have been applied to the XML structure,
034     * and apply any needed updates during deserialization.
035     * </p>
036     *
037     * <p>
038     * Updates are tracked by inserting an <code>{@link #UPDATES_ELEMENT_NAME &lt;pobj:updates&gt;}</code> element
039     * into the serialized XML document; this update list is transparently removed when the document is read back,
040     * and any missing updates are applied automatically.
041     * In this way the document and its set of applied updates always travel together. For example:
042     *
043     * <blockquote><pre>
044     *  &lt;MyConfig&gt;
045     *      <b>&lt;pobj:updates xmlns:pobj="http://dellroad-stuff.googlecode.com/ns/persistentObject"&gt;
046     *          &lt;pobj:update>some-update-name-1&lt;/pobj:update&gt;
047     *          &lt;pobj:update>some-update-name-2&lt;/pobj:update&gt;
048     *      &lt;/pobj:updates&gt;</b>
049     *      &lt;username&gt;admin&lt;/username&gt;
050     *      &lt;password&gt;secret&lt;/password&gt;
051     *  &lt;/MyConfig&gt;
052     * </pre></blockquote>
053     *
054     * @param <T> type of the root persistent object
055     */
056    public class PersistentObjectSchemaUpdater<T> extends AbstractSchemaUpdater<PersistentFileTransaction, PersistentFileTransaction>
057      implements PersistentObjectDelegate<T> {
058    
059        /**
060         * XML namespace URI used for nested update elements.
061         */
062        public static final String NAMESPACE_URI = "http://dellroad-stuff.googlecode.com/ns/persistentObject";
063    
064        /**
065         * Preferred XML namespace prefix for {@link #NAMESPACE_URI} elements.
066         */
067        public static final String XML_PREFIX = "pobj";
068    
069        /**
070         * XML element name for the updates list.
071         */
072        public static final QName UPDATES_ELEMENT_NAME = new QName(NAMESPACE_URI, "updates", XML_PREFIX);
073    
074        /**
075         * XML element name for a single update.
076         */
077        public static final QName UPDATE_ELEMENT_NAME = new QName(NAMESPACE_URI, "update", XML_PREFIX);
078    
079        /**
080         * XML namespace URI used for namespace declarations.
081         */
082        public static final QName XMLNS_ATTRIBUTE_NAME = new QName("http://www.w3.org/2000/xmlns/", XML_PREFIX, "xmlns");
083    
084        protected final PersistentObjectDelegate<T> delegate;
085    
086        private final XMLOutputFactory xmlOutputFactory = XMLOutputFactory.newFactory();
087    
088        private List<String> updateNames;
089    
090    // PersistentObjectDelegate methods
091    
092        /**
093         * Constructor.
094         *
095         * <p>
096         * Callers provide a {@link PersistentObjectDelegate} that will be used for all operations, with the exception
097         * that {@linkplain #serialize serialization} and {@linkplain #deserialize deserialization} will add/remove
098         * the upate list, and {@linkplain #deserialize deserialization} will apply any needed updates as necessary.
099         *
100         * @param delegate delegate that will be wrapped by this instance
101         * @throws IllegalArgumentException if {@code delegate} is null
102         */
103        public PersistentObjectSchemaUpdater(PersistentObjectDelegate<T> delegate) {
104            if (delegate == null)
105                throw new IllegalArgumentException("null delegate");
106            this.delegate = delegate;
107        }
108    
109        /**
110         * Make a deep copy of the given object.
111         *
112         * <p>
113         * The implementation in {@link PersistentObjectSchemaUpdater} delegates to the delegate provided in the constructor.
114         *
115         * @throws IllegalArgumentException {@inheritDoc}
116         * @throws PersistentObjectException {@inheritDoc}
117         */
118        @Override
119        public T copy(T original) {
120            return this.delegate.copy(original);
121        }
122    
123        /**
124         * Compare two object graphs.
125         *
126         * <p>
127         * The implementation in {@link PersistentObjectSchemaUpdater} delegates to the delegate provided in the constructor.
128         */
129        @Override
130        public boolean isSameGraph(T root1, T root2) {
131            return this.delegate.isSameGraph(root1, root2);
132        }
133    
134        /**
135         * Validate the given instance.
136         *
137         * <p>
138         * The implementation in {@link PersistentObjectSchemaUpdater} delegates to the delegate provided in the constructor.
139         *
140         * @throws IllegalArgumentException {@inheritDoc}
141         */
142        @Override
143        public Set<ConstraintViolation<T>> validate(T obj) {
144            return this.delegate.validate(obj);
145        }
146    
147        /**
148         * Handle an exception thrown during a delayed write-back attempt.
149         *
150         * <p>
151         * The implementation in {@link PersistentObjectSchemaUpdater} delegates to the delegate provided in the constructor.
152         */
153        @Override
154        public void handleWritebackException(PersistentObject<T> pobj, Throwable t) {
155            this.delegate.handleWritebackException(pobj, t);
156        }
157    
158        /**
159         * Get the default value for the root object graph.
160         *
161         * <p>
162         * The implementation in {@link PersistentObjectSchemaUpdater} delegates to the delegate provided in the constructor.
163         */
164        @Override
165        public T getDefaultValue() {
166            return this.delegate.getDefaultValue();
167        }
168    
169        /**
170         * Serialize object to XML.
171         *
172         * <p>
173         * The implementation in {@link PersistentObjectSchemaUpdater} delegates to the delegate provided in the constructor
174         * but also adds the update list as the first XML tag.
175         *
176         * @throws PersistentObjectException {@inheritDoc}
177         */
178        @Override
179        public void serialize(T obj, Result result) throws IOException {
180            try {
181    
182                // Get update names
183                List<String> updateNameList = this.updateNames != null ? this.updateNames : this.getAllUpdateNames();
184    
185                // Wrap result with a writer that adds the update list
186                XMLEventWriter eventWriter = this.xmlOutputFactory.createXMLEventWriter(result);
187                UpdatesXMLEventWriter updatesWriter = new UpdatesXMLEventWriter(eventWriter, updateNameList);
188    
189                // Serialize using the provided delegate
190                this.delegate.serialize(obj, new StAXResult(updatesWriter));
191                updatesWriter.close();
192            } catch (RuntimeException e) {
193                throw e;
194            } catch (IOException e) {
195                throw e;
196            } catch (Exception e) {
197                throw new PersistentObjectException(e);
198            }
199        }
200    
201        /**
202         * Deserialize object from XML.
203         *
204         * <p>
205         * The implementation in {@link PersistentObjectSchemaUpdater} delegates to the delegate provided in the constructor
206         * but also removes the update list as the first XML tag and applies any needed updates.
207         *
208         * @throws PersistentObjectException {@inheritDoc}
209         */
210        @Override
211        public T deserialize(Source source) throws IOException {
212            try {
213    
214                // Create a temporary in-memory "database" containing the XML content
215                PersistentFileTransaction transaction = new PersistentFileTransaction(source);
216    
217                // Apply schema updates as necessary to update the XML structure
218                this.initializeAndUpdateDatabase(transaction);
219    
220                // Save updates names so we can preserve their order
221                this.updateNames = transaction.getUpdates();
222    
223                // Sanity check that all updates were applied
224                final HashSet<String> unappliedUpdates = new HashSet<String>(this.getAllUpdateNames());
225                unappliedUpdates.removeAll(this.updateNames);
226                if (!unappliedUpdates.isEmpty())
227                    throw new PersistentObjectException("internal inconsistency: unapplied updates remain: " + unappliedUpdates);
228    
229                // Deserialize the now up-to-date XML structure using the provided delegate
230                StreamSource updatedSource = new StreamSource(new ByteArrayInputStream(transaction.getData()));
231                updatedSource.setSystemId(transaction.getSystemId());
232                return this.delegate.deserialize(updatedSource);
233            } catch (RuntimeException e) {
234                throw e;
235            } catch (IOException e) {
236                throw e;
237            } catch (Exception e) {
238                throw new PersistentObjectException(e);
239            }
240        }
241    
242    // AbstractSchemaUpdater methods
243    
244        @Override
245        protected boolean databaseNeedsInitialization(PersistentFileTransaction transaction) throws Exception {
246            return false;
247        }
248    
249        @Override
250        @SuppressWarnings("unchecked")
251        protected void initializeDatabase(PersistentFileTransaction transaction) throws Exception {
252            throw new UnsupportedOperationException();
253        }
254    
255        @Override
256        protected PersistentFileTransaction openTransaction(PersistentFileTransaction transaction) throws Exception {
257            return transaction;
258        }
259    
260        @Override
261        protected void commitTransaction(PersistentFileTransaction transaction) throws Exception {
262            // nothing to do
263        }
264    
265        @Override
266        protected void rollbackTransaction(PersistentFileTransaction transaction) throws Exception {
267            // nothing to do
268        }
269    
270        @Override
271        protected Set<String> getAppliedUpdateNames(PersistentFileTransaction transaction) throws Exception {
272            return new HashSet<String>(transaction.getUpdates());
273        }
274    
275        @Override
276        protected void recordUpdateApplied(PersistentFileTransaction transaction, String name) throws Exception {
277            transaction.getUpdates().add(name);
278        }
279    }
