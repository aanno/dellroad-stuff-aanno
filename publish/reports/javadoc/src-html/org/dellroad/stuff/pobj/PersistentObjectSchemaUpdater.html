008    package org.dellroad.stuff.pobj;
009    
010    import java.io.ByteArrayOutputStream;
011    import java.io.File;
012    import java.io.IOException;
013    import java.util.ArrayList;
014    import java.util.HashSet;
015    import java.util.Set;
016    
017    import javax.xml.stream.XMLEventReader;
018    import javax.xml.stream.XMLEventWriter;
019    import javax.xml.stream.XMLInputFactory;
020    import javax.xml.stream.XMLOutputFactory;
021    import javax.xml.stream.XMLStreamException;
022    import javax.xml.transform.Result;
023    import javax.xml.transform.Source;
024    import javax.xml.transform.stax.StAXResult;
025    import javax.xml.transform.stax.StAXSource;
026    import javax.xml.transform.stream.StreamResult;
027    
028    import org.dellroad.stuff.schema.AbstractSchemaUpdater;
029    
030    /**
031     * Support superclass for {@link PersistentObject} schema updaters.
032     *
033     * <p>
034     * This class holds a nested {@link PersistentObject} and ensures that it's up to date when started.
035     * Use {@link #getPersistentObject} to access it.
036     *
037     * <p>
038     * Updates are tracked by "secretly" inserting <code>{@link XMLConstants#UPDATES_ELEMENT_NAME &lt;pobj:updates&gt;}</code>
039     * elements into the serialized XML document; these updates are transparently removed when the document is read back.
040     * In this way the document and its set of applied updates always travel together.
041     *
042     * <p>
043     * Subclasses will typically override {@link #getInitialValue} for when there is no persistent file yet.
044     *
045     * @param <T> type of the root persistent object
046     */
047    public class PersistentObjectSchemaUpdater<T> extends AbstractSchemaUpdater<File, PersistentFileTransaction> {
048    
049        protected File file;
050        protected long writeDelay;
051        protected PersistentObjectDelegate<T> delegate;
052    
053        private ArrayList<String> updateNames;
054        private PersistentObject<T> persistentObject;
055    
056        /**
057         * Configure the file used to store this object persistently. Required property.
058         */
059        public void setFile(File file) {
060            this.file = file;
061        }
062    
063        /**
064         * Configure the maximum delay after an update operation before a write-back to the persistent file
065         * must be initiated. Default is zero.
066         */
067        public void setWriteDelay(long writeDelay) {
068            this.writeDelay = writeDelay;
069        }
070    
071        /**
072         * Configure the {@link PersistentObjectDelegate}. Required property.
073         */
074        public void setDelegate(PersistentObjectDelegate<T> delegate) {
075            this.delegate = delegate;
076        }
077    
078        /**
079         * Start this instance. Does nothing if already started.
080         *
081         * @throws IllegalArgumentException if an invalid file, write delay, or delegate is configured
082         * @throws PersistentObjectException if an error occurs
083         */
084        public synchronized void start() {
085    
086            // Already started?
087            if (this.persistentObject != null)
088                return;
089    
090            // Sanity check
091            if (this.file == null)
092                throw new IllegalArgumentException("no file configured");
093            if (this.writeDelay < 0)
094                throw new IllegalArgumentException("negative writeDelay file configured");
095            if (this.delegate == null)
096                throw new IllegalArgumentException("no delegate configured");
097    
098            // Do schema updates
099            try {
100                this.initializeAndUpdateDatabase(this.file);
101            } catch (RuntimeException e) {
102                throw e;
103            } catch (Exception e) {
104                throw new PersistentObjectException(e);
105            }
106    
107            // Create and start persistent object
108            this.persistentObject = new PersistentObject<T>(new UpdaterDelegate(), this.file, this.writeDelay);
109            this.persistentObject.start();
110        }
111    
112        /**
113         * Stop this instance. Does nothing if already stopped.
114         *
115         * @throws PersistentObjectException if a delayed write back is pending and error occurs during writing
116         */
117        public synchronized void stop() {
118    
119            // Already stopped?
120            if (this.persistentObject == null)
121                return;
122    
123            // Stop
124            this.persistentObject.stop();
125            this.persistentObject = null;
126        }
127    
128        /**
129         * Get the {@link PersistentObject}.
130         *
131         * @throws IllegalStateException if this instance is not started
132         */
133        public synchronized PersistentObject<T> getPersistentObject() {
134            if (this.persistentObject == null)
135                throw new IllegalStateException("not started");
136            return this.persistentObject;
137        }
138    
139        /**
140         * Get the initial value for the persistent object when no persistent file is found.
141         *
142         * <p>
143         * The implementation in {@link PersistentObjectSchemaUpdater} just returns null.
144         * Subclasses should override as desired.
145         */
146        protected T getInitialValue() {
147            return null;
148        }
149    
150        @Override
151        protected boolean databaseNeedsInitialization(PersistentFileTransaction transaction) throws Exception {
152            return transaction.getData() == null;
153        }
154    
155        @Override
156        protected void initializeDatabase(PersistentFileTransaction transaction) throws Exception {
157    
158            // Get initial value
159            T initialValue = this.getInitialValue();
160            if (initialValue == null)
161                return;
162    
163            // Serialize it
164            ByteArrayOutputStream buffer = new ByteArrayOutputStream(PersistentFileTransaction.FILE_BUFFER_SIZE);
165            StreamResult result = new StreamResult(buffer);
166            this.delegate.serialize(initialValue, result);
167    
168            // Set it in the transaction
169            transaction.setData(buffer.toByteArray());
170        }
171    
172        @Override
173        protected PersistentFileTransaction openTransaction(File file) throws Exception {
174            return new PersistentFileTransaction(file);
175        }
176    
177        @Override
178        protected void commitTransaction(PersistentFileTransaction transaction) throws Exception {
179            this.updateNames = new ArrayList<String>(transaction.getUpdates());
180            transaction.commit();
181        }
182    
183        @Override
184        protected void rollbackTransaction(PersistentFileTransaction transaction) throws Exception {
185            transaction.rollback();
186        }
187    
188        @Override
189        protected Set<String> getAppliedUpdateNames(PersistentFileTransaction transaction) throws Exception {
190            return new HashSet<String>(transaction.getUpdates());
191        }
192    
193        @Override
194        protected void recordUpdateApplied(PersistentFileTransaction transaction, String name) throws Exception {
195            transaction.addUpdate(name);
196        }
197    
198        // Our PersistentObjectDelegate that hides the updates when (de)serializing
199        private class UpdaterDelegate extends FilterDelegate<T> {
200    
201            private final XMLInputFactory xmlInputFactory = XMLInputFactory.newFactory();
202            private final XMLOutputFactory xmlOutputFactory = XMLOutputFactory.newFactory();
203    
204            UpdaterDelegate() {
205                super(PersistentObjectSchemaUpdater.this.delegate);
206            }
207    
208            /**
209             * Serialize object to XML, adding update list.
210             */
211            @Override
212            public void serialize(T obj, Result result) throws IOException {
213                try {
214                    XMLEventWriter eventWriter = this.xmlOutputFactory.createXMLEventWriter(result);
215                    UpdatesXMLEventWriter updatesWriter = new UpdatesXMLEventWriter(eventWriter,
216                      PersistentObjectSchemaUpdater.this.updateNames);
217                    super.serialize(obj, new StAXResult(updatesWriter));
218                    updatesWriter.close();
219                } catch (IOException e) {
220                    throw e;
221                } catch (XMLStreamException e) {
222                    throw new PersistentObjectException(e);
223                }
224            }
225    
226            /**
227             * Deserialize object from XML, removing update list.
228             */
229            @Override
230            public T deserialize(Source source) throws IOException {
231                try {
232                    XMLEventReader eventReader = this.xmlInputFactory.createXMLEventReader(source);
233                    UpdatesXMLEventReader updatesReader = new UpdatesXMLEventReader(eventReader);
234                    return super.deserialize(new StAXSource(updatesReader));
235                } catch (IOException e) {
236                    throw e;
237                } catch (XMLStreamException e) {
238                    throw new PersistentObjectException(e);
239                }
240            }
241        }
242    }
