032    /**
033     * Main class for Simple XML Persistence Objects (POBJ).
034     *
035     * <p>
036     * The model is that there is a root Java object which, together with the tree of objects is points to,
037     * represents the current "database" state. XML (de)serializers provide correspondence between the in-memory
038     * Java object tree and an XML document, and define what the "tree" is. Just like the persistent XML file that
039     * backs it, the object tree is read and written in its entirety, atomically, and so is always treated as "one thing".
040     * In other words, the object tree is passed by value.
041     *
042     * <p>
043     * When object tree is updated, it must pass validation checks, and then the persistent file is updated and
044     * listener notifications are sent out.
045     *
046     * <p>
047     * Support for optimistic locking is included. There is a current version number which is incremented each
048     * time the object tree is updated.
049     *
050     * <p>
051     * Support for delayed write-back is included. This allows modifications that occur in rapid succession to be consolidated
052     * into a single filesystem write operation.
053     *
054     * <p>
055     * File system writes use the rename operation for atomicity.
056     *
057     * <p>
058     * Instances must be supplied with a {@link PersistentObjectDelegate} that knows how to validate the root object
059     * and perform the XML conversions.
060     *
061     * @param <T> type of the root persistent object
062     * @see org.dellroad.stuff.pobj
063     */
064    public class PersistentObject<T> {
065    
066        protected final Logger log = LoggerFactory.getLogger(this.getClass());
067    
068        private final HashSet<PersistentObjectListener<T>> listeners = new HashSet<PersistentObjectListener<T>>();
069        private final PersistentObjectDelegate<T> delegate;
070        private final File persistentFile;
071        private final long writeDelay;
072    
073        private T root;
074        private T sharedRoot;
075        private ScheduledExecutorService writebackExecutor;
076        private ExecutorService notifyExecutor;
077        private ScheduledFuture pendingWrite;
078        private long version;
079        private boolean started;
080    
081        /**
082         * Constructor.
083         *
084         * <p>
085         * The {@code writeDelay} is the maximum delay after an update operation before a write-back to the persistent file
086         * must be initiated.
087         *
088         * @param delegate delegate supplying required operations
089         * @param file the file used to persist
090         * @param writeDelay write delay in milliseconds, or zero for immediate write-back
091         * @throws IllegalArgumentException if {@code writeDelay} is negative
092         */
093        protected PersistentObject(PersistentObjectDelegate<T> delegate, File file, long writeDelay) {
094            if (delegate == null)
095                throw new IllegalArgumentException("null delegate");
096            if (file == null)
097                throw new IllegalArgumentException("null file");
098            if (writeDelay < 0)
099                throw new IllegalArgumentException("negative writeDelay");
100            this.delegate = delegate;
101            this.persistentFile = file;
102            this.writeDelay = writeDelay;
103        }
104    
105        /**
106         * Get the persistent file containing the XML form of the persisted object.
107         */
108        public File getPersistentFile() {
109            return this.persistentFile;
110        }
111    
112        /**
113         * Get the maximum delay after an update operation before a write-back to the persistent file
114         * must be initiated.
115         *
116         * @return write delay in milliseconds, or zero for immediate write-back
117         */
118        public long getWriteDelay() {
119            return this.writeDelay;
120        }
121    
122        /**
123         * Get the version of the current root.
124         *
125         * <p>
126         * This returns a value which increases monotonically with each update.
127         */
128        public synchronized long getVersion() {
129            return this.version;
130        }
131    
132        /**
133         * Start this instance. Does nothing if already started.
134         *
135         * @throws PersistentObjectException if an error occurs
136         */
137        public synchronized void start() {
138    
139            // Already started?
140            if (this.started)
141                return;
142    
143            // Read file (if it exists)
144            this.log.info(this + ": starting");
145            if (this.persistentFile.exists()) {
146                this.root = this.read();
147                this.version++;
148            } else
149                this.log.info(this + ": persistent file `" + this.persistentFile + "' does not exist yet");
150    
151            // Create executor services
152            this.writebackExecutor = Executors.newSingleThreadScheduledExecutor();
153            this.notifyExecutor = Executors.newSingleThreadExecutor();
154            this.started = true;
155        }
156    
157        /**
158         * Stop this instance. Does nothing if already stopped.
159         *
160         * @throws PersistentObjectException if a delayed write back is pending and error occurs in {@link #write}
161         */
162        public synchronized void stop() {
163    
164            // Already stopped?
165            if (!this.started)
166                return;
167    
168            // Perform any lingering pending save now
169            if (this.pendingWrite != null) {
170                this.pendingWrite.cancel(false);
171                this.pendingWrite = null;
172                this.write(this.root);
173            }
174    
175            // Stop executor services
176            this.log.info(this + ": shutting down");
177            this.writebackExecutor.shutdown();
178            this.notifyExecutor.shutdown();
179    
180            // Reset
181            this.writebackExecutor = null;
182            this.notifyExecutor = null;
183            this.root = null;
184            this.started = false;
185        }
186    
187        /**
188         * Atomically read the root object.
189         *
190         * <p>
191         * If there is no persistent file and no value has been set, null will be returned.
192         * This condition is checked on each invocation of this method, so the persistent file
193         * is allowed to appear at any time. However, this is the only case where modifications
194         * to the underlying file are allowed while this instance is started.
195         *
196         * <p>
197         * This returns a deep copy of the current root object; any subsequent modifications are not written back.
198         *
199         * @return the current root instance, or be null if no value has ever been written
200         * @throws IllegalStateException if this instance is not started
201         * @throws PersistentObjectException if an error occurs
202         */
203        public synchronized T getRoot() {
204    
205            // Sanity check
206            if (!this.started)
207                throw new IllegalStateException("not started");
208    
209            // Try again to read the file (if it exists)
210            if (this.root == null && this.persistentFile.exists()) {
211                this.root = this.read();
212                this.version++;
213            }
214    
215            // Copy root
216            return this.root != null ? this.delegate.copy(this.root) : null;
217        }
218    
219        /**
220         * Get a shared copy of the root object.
221         *
222         * <p>
223         * This returns a copy of the root object, but it returns the same copy each time (until the next change).
224         * This method is more efficient than {@link #getRoot} when used by callers who mutually agree not to
225         * modify the returned object.
226         */
227        public synchronized T getSharedRoot() {
228            if (this.sharedRoot == null)
229                this.sharedRoot = this.getRoot();
230            return this.sharedRoot;
231        }
232    
233        /**
234         * Atomically update the root object.
235         *
236         * <p>
237         * The given object is deep-copied and the copy replaces the current root.
238         *
239         * <p>
240         * If {@code version} is non-zero, then if the current version is not equal to it,
241         * a {@link PersistentObjectVersionException} exception is thrown. This mechanism
242         * can be used for optimistic locking.
243         *
244         * @param newRoot new persistent object
245         * @param expectedVersion expected current version number, or zero for none
246         * @throws IllegalArgumentException if {@code newRoot} is null
247         * @throws IllegalArgumentException if {@code version} is negative
248         * @throws IllegalStateException if this instance is not started
249         * @throws PersistentObjectException if an error occurs
250         * @throws PersistentObjectVersionException if {@code version} is non-zero and not equal to the current version
251         * @throws PersistentObjectValidationException if the new root has validation errors
252         */
253        @SuppressWarnings("unchecked")
254        public final synchronized void setRoot(T newRoot, long expectedVersion) {
255    
256            // Sanity check
257            if (newRoot == null)
258                throw new IllegalArgumentException("null newRoot");
259            if (this.root == null)
260                throw new IllegalStateException("not started");
261            if (expectedVersion < 0)
262                throw new IllegalStateException("negative expectedVersion");
263    
264            // Check version number
265            if (expectedVersion != 0 && this.version != expectedVersion)
266                throw new PersistentObjectVersionException(this.version, expectedVersion);
267    
268            // Validate the new root
269            Set<ConstraintViolation<T>> violations = this.delegate.validate(newRoot);
270            if (!violations.isEmpty())
271                throw new PersistentObjectValidationException((Set<ConstraintViolation<?>>)(Object)violations);
272    
273            // Do the update
274            final T oldRoot = this.root;
275            this.root = this.delegate.copy(newRoot);
276            this.version++;
277            this.sharedRoot = null;
278    
279            // Perform write-back, either now or later
280            if (this.writeDelay == 0)
281                this.write(this.root);
282            else if (this.pendingWrite == null) {
283                this.pendingWrite = this.writebackExecutor.schedule(new Runnable() {
284                    @Override
285                    public void run() {
286                        PersistentObject.this.writeTimeout();
287                    }
288                }, this.writeDelay, TimeUnit.MILLISECONDS);
289            }
290    
291            // Notify listeners
292            this.notifyListeners(this.version, oldRoot, newRoot);
293        }
294    
295        /**
296         * Atomically update the root object.
297         *
298         * <p>
299         * The is a convenience method, equivalent to:
300         *  <blockquote>
301         *  <code>{@link #setRoot(Object, long) setRoot}(newRoot, 0)</code>
302         *  </blockquote>
303         *
304         * <p>
305         * This method cannot throw {@link PersistentObjectVersionException}.
306         */
307        public final synchronized void setRoot(T newRoot) {
308            this.setRoot(newRoot, 0);
309        }

314         * @throws IllegalArgumentException if {@code listener} is null

316        public void addListener(PersistentObjectListener<T> listener) {
317            if (listener == null)
318                throw new IllegalArgumentException("null listener");

322        }

325         * Remove a listener.
326         */

328            synchronized (this.listeners) {
329                this.listeners.remove(listener);
330            }
331        }

348        /**
349         * Read the persistent file.
350         *
351         * @throws PersistentObjectException if an error occurs
352         */
353        protected T read() {
354    
355            // Open file
356            this.log.info(this + ": reading persistent file `" + this.persistentFile + "'");
357            BufferedInputStream input;
358            try {
359                input = new BufferedInputStream(new FileInputStream(this.persistentFile));
360            } catch (IOException e) {
361                throw new PersistentObjectException("error opening persistent file", e);
362            }
363    
364            // Parse XML
365            T obj;
366            try {
367                StreamSource source = new StreamSource(input);
368                source.setSystemId(this.persistentFile);
369                try {
370                    obj = this.delegate.deserialize(source);
371                } catch (IOException e) {
372                    throw new PersistentObjectException("error reading persistent file", e);
373                }
374            } finally {
375                try {
376                    input.close();
377                } catch (IOException e) {
378                    // ignore
379                }
380            }
381    
382            // Check result
383            if (obj == null)
384                throw new PersistentObjectException("null object returned by delegate.deserialize()");
385    
386            // Done
387            return obj;
388        }
389    
390        /**
391         * Write the persistent file.
392         *
393         * <p>
394         * A temporary file is created in the same directory and then renamed to provide for an atomic update.
395         *
396         * @throws IllegalArgumentException if {@code obj} is null
397         * @throws PersistentObjectException if an error occurs
398         */
399        protected final synchronized void write(T obj) {
400    
401            // Sanity check
402            if (obj == null)
403                throw new IllegalArgumentException("null obj");
404    
405            // Create temporary file
406            this.log.info(this + ": writing persistent file `" + this.persistentFile + "'");
407            File tempFile;
408            try {
409                tempFile = File.createTempFile(this.persistentFile.getName(), null, this.persistentFile.getParentFile());
410            } catch (IOException e) {
411                throw new PersistentObjectException("error creating temporary file", e);
412            }
413            try {
414    
415                // Open temporary file
416                BufferedOutputStream output;
417                try {
418                    output = new BufferedOutputStream(new FileOutputStream(tempFile));
419                } catch (IOException e) {
420                    throw new PersistentObjectException("error opening to temporary file", e);
421                }
422    
423                // Serialize to XML
424                this.log.info(this + ": saving persistent object");
425                try {
426                    StreamResult result = new StreamResult(output);
427                    result.setSystemId(tempFile);
428                    try {
429                        this.delegate.serialize(obj, result);
430                    } catch (IOException e) {
431                        throw new PersistentObjectException("error writing persistent file", e);
432                    }
433                    try {
434                        output.close();
435                    } catch (IOException e) {
436                        throw new PersistentObjectException("error closing temporary file", e);
437                    }
438                    output = null;
439                } finally {
440                    try {
441                        if (output != null)
442                            output.close();
443                    } catch (IOException e) {
444                        // ignore
445                    }
446                }
447    
448                // Rename file
449                if (!tempFile.renameTo(this.persistentFile)) {
450                    throw new PersistentObjectException("error renaming temporary file `"
451                      + tempFile.getName() + "' to `" + this.persistentFile.getName() + "'");
452                }
453                tempFile = null;
454            } finally {
455                if (tempFile != null)
456                    tempFile.delete();
457            }
458        }

460        /**
461         * Notify listeners.
462         *
463         * @param newVersion the version number associated with the new root
464         */
465        protected void notifyListeners(long newVersion, T oldRoot, T newRoot) {
466    
467            // Snapshot listeners
468            final ArrayList<PersistentObjectListener<T>> listenersCopy = new ArrayList<PersistentObjectListener<T>>();
469            synchronized (this.listeners) {
470                listenersCopy.addAll(this.listeners);
471            }
472    
473            // Notify them
474            final PersistentObjectEvent<T> event = new PersistentObjectEvent<T>(this, newVersion, oldRoot, newRoot);
475            this.notifyExecutor.submit(new Runnable() {
476                @Override
477                public void run() {
478                    PersistentObject.this.doNotifyListeners(listenersCopy, event);
479                }
480            });
481        }
482    
483        // Handle a write-back timeout
484        private synchronized void writeTimeout() {
485    
486            // Check for cancel race
487            if (this.pendingWrite == null)
488                return;
489            this.pendingWrite = null;
490    
491            // Write it
492            try {
493                this.write(this.root);
494            } catch (ThreadDeath t) {
495                throw t;
496            } catch (Throwable t) {
497                this.delegate.handleWritebackException(this, t);
498            }
499        }
500    
501        // Notify listeners. This is invoked in a separate thread.
502        private void doNotifyListeners(ArrayList<PersistentObjectListener<T>> list, PersistentObjectEvent<T> event) {
503            for (PersistentObjectListener<T> listener : list) {
504                try {
505                    listener.handleEvent(event);
506                } catch (ThreadDeath t) {
507                    throw t;
508                } catch (Throwable t) {
509                    this.log.error(this + ": error notifying listener " + listener, t);
510                }
511            }
512        }
