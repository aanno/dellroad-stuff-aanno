008    package org.dellroad.stuff.vaadin;
009    
010    /**
011     * Cacheable list of Java objects backing an {@link AbstractQueryContainer}.
012     * Instances contain a list of Java objects (or some portion thereof), and may become invalid over time.
013     *
014     * @see AbstractQueryContainer
015     */
016    public interface QueryList<T> {
017    
018        /**
019         * Get the total size of this list.
020         */
021        long size();
022    
023        /**
024         * Get an item in the list, or throw an exception if this instance is no longer valid.
025         *
026         * @param index index of the item (zero-based)
027         * @throws IndexOutOfBoundsException if {@code index} is less than zero or greater than {@link #size}
028         * @throws InvalidQueryListException if this list has become invalid or cannot provide the item
029         */
030        T get(long index) throws InvalidQueryListException;
031    }
