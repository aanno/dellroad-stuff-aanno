008    package org.dellroad.stuff.vaadin;
009    
010    import java.util.List;
011    
012    /**
013     * Simple {@link QueryList} implementation using a normal {@link List}.
014     *
015     * @see AbstractQueryContainer
016     */
017    public class SimpleQueryList<T> implements QueryList<T> {
018    
019        private final List<? extends T> list;
020    
021        /**
022         * Constructor.
023         *
024         * @param list backing list
025         * @throws IllegalArgumentException if {@code list} is null
026         */
027        public SimpleQueryList(List<? extends T> list) {
028            if (list == null)
029                throw new IllegalArgumentException("null list");
030            this.list = list;
031        }
032    
033        @Override
034        public long size() {
035            return this.list.size();
036        }
037    
038        @Override
039        public T get(long index) {
040            if (index > Integer.MAX_VALUE || index < Integer.MIN_VALUE)
041                throw new IndexOutOfBoundsException("index = " + index);
042            return this.list.get((int)index);
043        }
044    }
