008    package org.dellroad.stuff.vaadin;
009    
010    import com.vaadin.data.Container;
011    import com.vaadin.data.Item;
012    import com.vaadin.data.Property;
013    import com.vaadin.data.util.ObjectProperty;
014    import com.vaadin.ui.Table;
015    
016    /**
017     * Defines a Vaadin property, having a name, which is also the property ID, and its type.
018     * This class provides a mechanism for the explicit naming and identification of Vaadin properties.
019     *
020     * <p>
021     * For a given instance, <code>def.{@link #getPropertyId getPropertyId()}</code> is the property ID
022     * and you can access the value using <code>def.{@link #read read(item)}</code>.
023     *
024     * <p>
025     * Example:
026     * <blockquote><pre>
027     *  PropertyDef&lt;Integer&gt; def = new PropertyDef&lt;Integer&gt;("age", Integer.class, -1);
028     *  def.addTo(container);
029     *  def.addTo(item, property);
030     *  ...
031     *  int age = def.read(item);
032     *  ...
033     *  Property prop = this.get(container, itemId);
034     * </pre></blockquote>
035     *
036     * @param <T> property's value type
037     */
038    public final class PropertyDef<T> {
039    
040        private final String name;
041        private final Class<T> type;
042        private final T defaultValue;
043    
044        /**
045         * Convenience contructor.
046         *
047         * <p>
048         * Equivalent to:
049         *  <blockquote><pre>
050         *  PropertyDef(name, type, null);
051         *  </pre></blockquote>
052         * </p>
053         */
054        public PropertyDef(String name, Class<T> type) {
055            this(name, type, null);
056        }
057    
058        /**
059         * Primary constructor.
060         *
061         * @param name property name; also serves as the property ID
062         * @param type property type
063         * @param defaultValue default value for the property; may be null
064         */
065        public PropertyDef(String name, Class<T> type, T defaultValue) {
066            if (name == null)
067                throw new IllegalArgumentException("null name");
068            if (type == null)
069                throw new IllegalArgumentException("null type");
070            this.name = name;
071            this.type = type;
072            this.defaultValue = defaultValue;
073        }
074    
075        /**
076         * Get the name of this property. This is also used as the property ID.
077         *
078         * @return property name, never null
079         */
080        public String getName() {
081            return this.name;
082        }
083    
084        /**
085         * Get the ID of this property. Returns the same thing as {@link #getName()}.
086         *
087         * @return property name, never null
088         */
089        public Object getPropertyId() {
090            return this.getName();
091        }
092    
093        /**
094         * Get the type of the property value that this instance represents.
095         *
096         * @return property type, never null
097         */
098        public Class<T> getType() {
099            return this.type;
100        }
101    
102        /**
103         * Get the default value for this property.
104         *
105         * @return property default value
106         */
107        public T getDefaultValue() {
108            return this.defaultValue;
109        }
110    
111        /**
112         * Create a simple {@link ObjectProperty} using the given value.
113         *
114         * @param value property value
115         * @param readOnly whether property should be read-only
116         * @return new property
117         */
118        public ObjectProperty createProperty(T value, boolean readOnly) {
119            return new ObjectProperty<T>(value, this.getType(), readOnly);
120        }
121    
122        /**
123         * Create a read/write {@link ObjectProperty} using the given value.
124         *
125         * <p>
126         * Equivalent to:
127         *  <blockquote><pre>
128         *  {@link #createProperty(T, boolean) createProperty()}(value, false);
129         *  </pre></blockquote>
130         * </p>
131         *
132         * @param value property value
133         * @return new property
134         */
135        public ObjectProperty createProperty(T value) {
136            return this.createProperty(value, false);
137        }
138    
139        /**
140         * Create a read/write {@link ObjectProperty} using the default value.
141         *
142         * <p>
143         * Equivalent to:
144         *  <blockquote><pre>
145         *  {@link #createProperty(T) createProperty()}(def.getDefaultValue());
146         *  </pre></blockquote>
147         * </p>
148         *
149         * @param value property value
150         * @return new property
151         */
152        public ObjectProperty createProperty() {
153            return this.createProperty(this.getDefaultValue());
154        }
155    
156        /**
157         * Get the property that this instance represents from the given {@link Item}.
158         *
159         * @return property, or null if not found
160         */
161        public Property get(Item item) {
162            return item.getItemProperty(this.getPropertyId());
163        }
164    
165        /**
166         * Get the property that this instance represents from the given {@link Container}.
167         *
168         * @param container the container containing the items
169         * @param itemId the ID of the item containing the property
170         * @return property, or null if not found
171         */
172        public Property get(Container container, Object itemId) {
173            return container.getContainerProperty(itemId, this.getPropertyId());
174        }
175    
176        /**
177         * Add a property represented by this instance to the given {@link Container}.
178         *
179         * @param container the container to add the property to
180         * @return true if the operation succeeded, false if not
181         * @throws UnsupportedOperationException if the operation is not supported
182         */
183        public boolean addTo(Container container) {
184            return container.addContainerProperty(this.getPropertyId(), this.getType(), this.getDefaultValue());
185        }
186    
187        /**
188         * Add a property represented by this instance to the given {@link Item}.
189         *
190         * @param item the item to add the property to
191         * @param property the property to be added to the item and identified by this instance's name
192         * @return true if the operation succeeded, false if not
193         * @throws UnsupportedOperationException if the operation is not supported
194         */
195        public boolean addTo(Item item, Property property) {
196            return item.addItemProperty(this.getPropertyId(), property);
197        }
198    
199        /**
200         * Add a property represented by this instance to the given {@link Table}.
201         *
202         * @param table the table to add the property to
203         * @return true if the operation succeeded, false if not
204         * @throws UnsupportedOperationException if the operation is not supported
205         */
206        public boolean addTo(Table table) {
207            return table.addContainerProperty(this.getPropertyId(), this.getType(), this.getDefaultValue());
208        }
209    
210        /**
211         * Read the property that this instance represents from the given {@link Item}.
212         *
213         * @return property value, or null if not found
214         * @throws ClassCastException if the property in {@code item} has the wrong type
215         */
216        public T read(Item item) {
217            Property property = this.get(item);
218            if (property == null)
219                return null;
220            return this.type.cast(property.getValue());
221        }
222    
223        @Override
224        public int hashCode() {
225            return this.name.hashCode()
226              ^ this.type.hashCode()
227              ^ (this.defaultValue != null ? this.defaultValue.hashCode() : 0);
228        }
229    
230        @Override
231        public boolean equals(Object obj) {
232            if (!(obj instanceof PropertyDef))
233                return false;
234            PropertyDef<?> that = (PropertyDef<?>)obj;
235            return this.name.equals(that.name)
236              && this.type == that.type
237              && (this.defaultValue != null ? this.defaultValue.equals(that.defaultValue) : that.defaultValue == null);
238        }
239    }
