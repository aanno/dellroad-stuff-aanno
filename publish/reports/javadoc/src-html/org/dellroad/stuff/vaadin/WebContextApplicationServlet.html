008    package org.dellroad.stuff.vaadin;
009    
010    import com.vaadin.Application;
011    import com.vaadin.terminal.gwt.server.ApplicationServlet;
012    
013    import javax.servlet.ServletConfig;
014    import javax.servlet.ServletException;
015    import javax.servlet.http.HttpServletRequest;
016    
017    import org.slf4j.Logger;
018    import org.slf4j.LoggerFactory;
019    import org.springframework.beans.BeansException;
020    import org.springframework.web.context.WebApplicationContext;
021    import org.springframework.web.context.support.WebApplicationContextUtils;
022    
023    /**
024     * Vaadin {@link Application} servlet that acquires {@link Application} instances from an
025     * associated Spring {@link WebApplicationContext}.
026     * This allows a Vaadin application to be configured as a normal bean inside
027     * a Spring application context, where annotations such as
028     * <code>{@link org.springframework.beans.factory.annotation.Autowired @Autowired}</code>,
029     * <code>{@link org.springframework.beans.factory.annotation.Required @Required}</code>, etc.
030     * and interfaces such as {@link org.springframework.beans.factory.BeanFactoryAware BeanFactoryAware},
031     * etc. will work.
032     * </p>
033     *
034     * <p>
035     * Important: The application bean must be declared in the associated {@link WebApplicationContext}
036     * with <code>scope="prototype"</code> (or <code>scope="session"</code> if you prefer), and it must be
037     * the only instance of the configured application class.
038     * </p>
039     *
040     * <p>
041     * An example of "direct" use of this servlet in conjunction with Spring's
042     * {@link org.springframework.web.context.ContextLoaderListener ContextLoaderListener}:
043     * <blockquote><pre>
044     *  &lt;!-- Spring context loader --&gt;
045     *  &lt;listener&gt;
046     *      &lt;listener-class&gt;org.springframework.web.context.ContextLoaderListener&lt;/listener-class&gt;
047     *  &lt;/listener&gt;
048     *
049     *  &lt;!-- Spring web request listener: makes the current web request available in a private ThreadLocal --&gt;
050     *  &lt;listener&gt;
051     *      &lt;listener-class&gt;org.springframework.web.context.request.RequestContextListener&lt;/listener-class&gt;
052     *  &lt;/listener&gt;
053     *
054     *  &lt;!-- Vaadin servlet --&gt;
055     *  &lt;servlet&gt;
056     *      &lt;servlet-name&gt;myapp&lt;/servlet-name&gt;
057     *      &lt;servlet-class&gt;org.dellroad.stuff.vaadin.WebContextApplicationServlet&lt;/servlet-class&gt;
058     *      &lt;init-param&gt;
059     *          &lt;param-name&gt;application&lt;/param-name&gt;
060     *          &lt;param-value&gt;some.spring.configured.Application&lt;/param-value&gt;
061     *      &lt;/init-param&gt;
062     *      &lt;init-param&gt;
063     *          &lt;param-name&gt;productionMode&lt;/param-name&gt;
064     *          &lt;param-value&gt;true&lt;/param-value&gt;
065     *      &lt;/init-param&gt;
066     *  &lt;/servlet&gt;
067     *  &lt;servlet-mapping&gt;
068     *      &lt;servlet-name&gt;myapp&lt;/servlet-name&gt;
069     *      &lt;url-pattern&gt;/*&lt;/url-pattern&gt;
070     *  &lt;/servlet-mapping&gt;
071     * </pre></blockquote>
072     * with this application context:
073     * <blockquote><pre>
074     *  &lt;!-- Activate Spring annotation support --&gt;
075     *  &lt;context:annotation-config/&gt;
076     *
077     *  &lt;!-- Define Vaadin application bean --&gt;
078     *  &lt;bean class="some.spring.configured.Application" scope="prototype"/&gt;
079     *
080     *  &lt;!-- Define other beans... --&gt;
081     * </pre></blockquote>
082     * </p>
083     *
084     * <p>
085     * An example that creates a Spring MVC "controller" bean for use with Spring's
086     * {@link org.springframework.web.servlet.DispatcherServlet DispatcherServlet}:
087     * <blockquote><pre>
088     *  &lt;!-- Activate Spring annotation support --&gt;
089     *  &lt;context:annotation-config/&gt;
090     *
091     *  &lt;!-- Define controller bean for Vaadin application --&gt;
092     *  &lt;bean id="applicationController" class="org.springframework.web.servlet.mvc.ServletWrappingController"
093     *     p:servletClass="org.dellroad.stuff.vaadin.WebContextApplicationServlet"&gt;
094     *      &lt;property name="initParameters"&gt;
095     *          &lt;props&gt;
096     *              &lt;prop key="application"&gt;some.spring.configured.Application&lt;/prop&gt;
097     *              &lt;prop key="productionMode"&gt;true&lt;/prop&gt;
098     *          &lt;/props&gt;
099     *      &lt;/property&gt;
100     *  &lt;/bean&gt;
101     *
102     *  &lt;!-- Define Vaadin application bean --&gt;
103     *  &lt;bean class="some.spring.configured.Application" scope="prototype"/&gt;
104     *
105     *  &lt;!-- Define other beans... --&gt;
106     * </pre></blockquote>
107     *
108     * <p>
109     * Note: a nicer approach is to use Spring's {@link org.springframework.beans.factory.annotation.Configurable @Configurable}
110     * annotation in combination with build-time (or runtime) AOP bytecode weaving. With that approach, all of your widget classes
111     * can be autowired, not just the {@link Application} class as with this class.
112     * </p>
113     *
114     * @see org.springframework.web.servlet.mvc.ContextLoaderListener
115     * @see org.springframework.web.servlet.DispatcherServlet
116     * @see org.springframework.web.servlet.mvc.ServletWrappingController
117     * @see WebApplicationContext
118     * @since 1.0.50
119     */
120    @SuppressWarnings("serial")
121    public class WebContextApplicationServlet extends ApplicationServlet {
122    
123        protected final Logger log = LoggerFactory.getLogger(getClass());
124    
125        private WebApplicationContext webApplicationContext;
126    
127        /**
128         * Initialize this servlet.
129         *
130         * @throws ServletException if there is no {@link WebApplicationContext} associated with this servlet's context
131         */
132        @Override
133        public void init(ServletConfig config) throws ServletException {
134            super.init(config);
135            log.debug("finding containing WebApplicationContext");
136            try {
137                this.webApplicationContext = WebApplicationContextUtils.getRequiredWebApplicationContext(config.getServletContext());
138            } catch (IllegalStateException e) {
139                throw new ServletException("could not locate containing WebApplicationContext");
140            }
141        }
142    
143        /**
144         * Get the containing Spring {@link WebApplicationContext}.
145         * This only works after the servlet has been initialized (via {@link #init init()}).
146         *
147         * @throws ServletException if the operation fails
148         */
149        protected final WebApplicationContext getWebApplicationContext() throws ServletException {
150            if (this.webApplicationContext == null)
151                throw new ServletException("can't retrieve WebApplicationContext before init() is invoked");
152            return this.webApplicationContext;
153        }
154    
155        /**
156         * Create and configure a new instance of the configured application class.
157         *
158         * <p>
159         * The implementation in {@link WebContextApplicationServlet} invokes
160         * {@link org.springframework.beans.factory.BeanFactory#getBean(Class) BeanFactory.getBean()}
161         * on the associated {@link WebApplicationContext} class to acquire an instance of the class
162         * configured via the {@code application} servlet parameter.
163         * </p>
164         *
165         * @param request the triggering {@link HttpServletRequest}
166         * @throws ServletException if bean creation fails
167         */
168        @Override
169        protected Application getNewApplication(HttpServletRequest request) throws ServletException {
170            Class<? extends Application> applicationClass;
171            try {
172                applicationClass = this.getApplicationClass();
173            } catch (ClassNotFoundException e) {
174                throw new ServletException("failed to create new instance of application class", e);
175            }
176            this.log.debug("acquiring new instance of " + applicationClass + " from " + this.webApplicationContext);
177            try {
178                return this.getWebApplicationContext().getBean(applicationClass);
179            } catch (BeansException e) {
180                throw new ServletException("failed to acquire new instance of " + applicationClass, e);
181            }
182        }
183    }
