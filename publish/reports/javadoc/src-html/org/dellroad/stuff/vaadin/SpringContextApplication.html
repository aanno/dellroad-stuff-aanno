032    /**
033     * Vaadin application implementation that manages an associated Spring {@link WebApplicationContext}.
034     *
035     * <h3>Overview</h3>
036     *
037     * <p>
038     * Each Vaadin application instance is given its own Spring application context, and all such
039     * application contexts share the same parent context, which is the one associated with the overal servlet web context
040     * (i.e., the one created by Spring's {@link org.springframework.web.context.ContextLoaderListener ContextLoaderListener}).
041     * A context is created when a new Vaadin application instance is initialized, and destroyed when it is closed.
042     * </p>
043     *
044     * <p>
045     * This setup is analogous to how Spring's {@link org.springframework.web.servlet.DispatcherServlet DispatcherServlet}
046     * creates per-servlet application contexts that are children of the overall servlet web context.
047     * </p>
048     *
049     * <p>
050     * For each Vaadin application {@code com.example.FooApplication} that subclasses this class, there should exist an XML
051     * file named {@code FooApplication.xml} in the {@code WEB-INF/} directory that defines the per-Vaadin application Spring
052     * application context (this naming scheme {@linkplain #getApplicationName can be overriden}).
053     * </p>
054     *
055     * <h3>Vaadin Application as BeanFactory singleton</h3>
056     *
057     * <p>
058     * This {@link SpringContextApplication} instance can itself be exposed in, and configured by, the associated Spring
059     * application context. Simply create a bean definition that invokes {@link ContextApplication#get}:
060     * <blockquote><pre>
061     *  &lt;bean id="myVaadinApplication" class="org.dellroad.stuff.vaadin.ContextApplication" factory-method="get"/&gt;
062     * </pre></blockquote>
063     *
064     * <p>
065     * This then allows you to autowire the {@link SpringContextApplication} and other UI components together, e.g.:
066     * <blockquote><pre>
067     *  public class MyApplication extends SpringContextApplication {
068     *
069     *      &#64;Autowired
070     *      private MainPanel mainPanel;
071     *
072     *      &#64;Override
073     *      public void initSpringApplication(ConfigurableWebApplicationContext context) {
074     *          Window mainWindow = new Window("MyApplication", this.mainPanel);
075     *          this.setMainWindow(mainWindow);
076     *      }
077     *
078     *      ...
079     *  }
080     *
081     *  &#64;Component
082     *  public class MainPanel extends VerticalLayout {
083     *
084     *      &#64;Autowired
085     *      private MyApplication application;
086     *
087     *      ...
088     *  }
089     * </pre></blockquote>
090     * </p>
091     *
092     * <p>
093     * Even if you don't explicitly define the {@link SpringContextApplication} bean in your Spring application context,
094     * it will still be available as a dependency for autowiring into other beans (this is accomplished using
095     * {@link ConfigurableListableBeanFactory#registerResolvableDependency
096     * ConfigurableListableBeanFactory.registerResolvableDependency()}). Of course, in this case the
097     * {@link SpringContextApplication} bean won't itself be autowired or configured.
098     * </p>
099     *
100     * <h3><code>@VaadinConfigurable</code> Beans</h3>
101     *
102     * <p>
103     * It is also possible to configure beans outside of this application context using AOP, so that any invocation of
104     * {@code new FooBar()}, where the class {@code FooBar} is marked {@link VaadinConfigurable @VaadinConfigurable},
105     * will automagically cause the new {@code FooBar} object to be configured by the application context associated with
106     * the {@linkplain ContextApplication#get() currently running application instance}. In effect, this does for
107     * Vaadin application beans what Spring's {@link org.springframework.beans.factory.annotation.Configurable @Configurable}
108     * does for regular beans.
109     * </p>
110     *
111     * <p>
112     * Note however that Spring {@linkplain org.springframework.beans.factory.DisposableBean#destroy destroy methods}
113     * will not be invoked on application close for these beans, since their lifecycle is controlled outside of the
114     * Spring application context (this is also the case with
115     * {@link org.springframework.beans.factory.annotation.Configurable @Configurable} beans). Instead, these beans
116     * can register as a {@link ContextApplication.CloseListener} for shutdown notification.
117     * </p>
118     *
119     * <p>
120     * For the this annotation to do anything, {@link VaadinConfigurable @VaadinConfigurable} classes must be woven
121     * (either at build time or runtime) using the
122     * <a href="http://www.eclipse.org/aspectj/doc/released/faq.php#compiler">AspectJ compiler</a> with the
123     * {@code VaadinConfigurableAspect} aspect (included in the <code>dellroad-stuff</code> JAR file).
124     * </p>
125     *
126     * @see ContextApplication#get
127     * @see <a href="https://github.com/archiecobbs/dellroad-stuff-vaadin-spring-demo3">Example Code on GitHub</a>
128     */
129    @SuppressWarnings("serial")
130    public abstract class SpringContextApplication extends ContextApplication {
131    
132        private static final AtomicLong UNIQUE_INDEX = new AtomicLong();
133    
134        private transient ConfigurableWebApplicationContext context;
135    
136        /**
137         * Get this instance's associated Spring application context.
138         */
139        public ConfigurableWebApplicationContext getApplicationContext() {
140            return this.context;
141        }
142    
143        /**
144         * Get the {@link SpringContextApplication} instance associated with the current thread or throw an exception if there is none.
145         *
146         * <p>
147         * Works just like {@link ContextApplication#get()} but returns this narrower type.
148         * </p>
149         *
150         * @return the {@link SpringContextApplication} associated with the current thread
151         * @throws IllegalStateException if the current thread is not servicing a Vaadin web request
152         *  or the current Vaadin {@link com.vaadin.Application} is not a {@link SpringContextApplication}
153         */
154        public static SpringContextApplication get() {
155            return ContextApplication.get(SpringContextApplication.class);
156        }
157    
158        /**
159         * Initializes the associated {@link ConfigurableWebApplicationContext}.
160         */
161        protected final void initApplication() {
162    
163            // Load the context
164            this.loadContext();
165    
166            // Initialize subclass
167            this.initSpringApplication(context);
168        }
169    
170        /**
171         * Initialize the application. Sub-classes of {@link SpringContextApplication} must implement this method.
172         *
173         * @param context the associated {@link WebApplicationContext} just created and refreshed
174         * @see #destroySpringApplication
175         */
176        protected abstract void initSpringApplication(ConfigurableWebApplicationContext context);
177    
178        /**
179         * Perform any application-specific shutdown work. This will be invoked at shutdown after this Vaadin application and the
180         * associated {@link WebApplicationContext} have both been closed.
181         *
182         * <p>
183         * The implementation in {@link SpringContextApplication} does nothing. Subclasses may override as necessary.
184         * </p>
185         *
186         * <p>
187         * Note that if a {@link SpringContextApplication} instance is exposed in the application context and configured
188         * with a Spring {@linkplain org.springframework.beans.factory.DisposableBean#destroy destroy method}, then that
189         * method will also be invoked when the application is closed. In such cases overriding this method is not necessary.
190         * </p>
191         *
192         * @see #initSpringApplication
193         */
194        protected void destroySpringApplication() {
195        }
196    
197        /**
198         * Post-process the given {@link WebApplicationContext} after initial creation but before the initial
199         * {@link org.springframework.context.ConfigurableApplicationContext#refresh refresh()}.
200         *
201         * <p>
202         * The implementation in {@link SpringContextApplication} does nothing. Subclasses may override as necessary.
203         * </p>
204         *
205         * @param context the associated {@link WebApplicationContext} just refreshed
206         * @see #onRefresh
207         * @see ConfigurableWebApplicationContext#refresh()
208         */
209        protected void postProcessWebApplicationContext(ConfigurableWebApplicationContext context) {
210        }
211    
212        /**
213         * Perform any application-specific work after a successful application context refresh.
214         *
215         * <p>
216         * The implementation in {@link SpringContextApplication} does nothing. Subclasses may override as necessary.
217         * </p>
218         *
219         * @param context the associated {@link WebApplicationContext} just refreshed
220         * @see #postProcessWebApplicationContext
221         * @see org.springframework.context.ConfigurableApplicationContext#refresh
222         */
223        protected void onRefresh(ApplicationContext context) {
224        }
225    
226        /**
227         * Get the name for this application. This is used as the name of the XML file in {@code WEB-INF/} that
228         * defines the Spring application context associated with this instance.
229         *
230         * <p>
231         * The implementation in {@link SpringContextApplication} returns this instance's class'
232         * {@linkplain Class#getSimpleName simple name}.
233         * </p>
234         */
235        protected String getApplicationName() {
236            return this.getClass().getSimpleName();
237        }
238    
239    // ApplicationContext setup
240    
241        private void loadContext() {
242    
243            // Logging
244            this.log.info("loading application context for Vaadin application " + this.getApplicationName());
245    
246            // Sanity check
247            if (this.context != null)
248                throw new IllegalStateException("context already loaded");
249    
250            // Find the application context associated with the servlet; it will be the parent
251            ServletContext servletContext;
252            HttpServletRequest request = ContextApplication.currentRequest();
253            try {
254                // getServletContext() is a servlet AIP 3.0 method, so don't freak out if it's not there
255                servletContext = (ServletContext)HttpServletRequest.class.getMethod("getServletContext").invoke(request);
256            } catch (Exception e) {
257                servletContext = ContextLoader.getCurrentWebApplicationContext().getServletContext();
258            }
259            WebApplicationContext parent = WebApplicationContextUtils.getWebApplicationContext(servletContext);
260    
261            // Create and configure a new application context for this Application instance
262            this.context = new XmlWebApplicationContext();
263            this.context.setId(ConfigurableWebApplicationContext.APPLICATION_CONTEXT_ID_PREFIX
264              + servletContext.getContextPath() + "/" + this.getApplicationName() + "-"
265              + SpringContextApplication.UNIQUE_INDEX.incrementAndGet());
266            this.context.setParent(parent);
267            this.context.setServletContext(servletContext);
268            //context.setServletConfig(??);
269            this.context.setNamespace(this.getApplicationName());
270    
271            // Register listener so we can notify subclass on refresh events
272            this.context.addApplicationListener(new SourceFilteringListener(this.context, new RefreshListener()));
273    
274            // Register this instance as an implicitly resolvable dependency
275            this.context.addBeanFactoryPostProcessor(new BeanFactoryPostProcessor() {
276                @Override
277                public void postProcessBeanFactory(ConfigurableListableBeanFactory beanFactory) {
278                    beanFactory.registerResolvableDependency(Application.class, SpringContextApplication.this);
279                }
280            });
281    
282            // Invoke any subclass setup
283            this.postProcessWebApplicationContext(context);
284    
285            // Refresh context
286            this.context.refresh();
287    
288            // Get notified of application shutdown so we can shut down the context as well
289            this.addListener(new ContextCloseListener());
290        }
291    
292    // Serialization
293    
294        private void readObject(ObjectInputStream input) throws IOException, ClassNotFoundException {
295            input.defaultReadObject();
296            this.loadContext();
297        }
298    
299    // Nested classes
300    
301        // My refresh listener
302        private class RefreshListener implements ApplicationListener<ContextRefreshedEvent>, Serializable {
303            @Override
304            public void onApplicationEvent(ContextRefreshedEvent event) {
305                SpringContextApplication.this.onRefresh(event.getApplicationContext());
306            }
307        }
308    
309        // My close listener
310        private class ContextCloseListener implements CloseListener, Serializable {
311            @Override
312            public void applicationClosed(CloseEvent closeEvent) {
313                SpringContextApplication.this.log.info("closing application context associated with Vaadin application "
314                  + SpringContextApplication.this.getApplicationName());
315                SpringContextApplication.this.context.close();
316                SpringContextApplication.this.destroySpringApplication();
317            }
318        }
319    }
