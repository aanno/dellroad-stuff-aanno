008    package org.dellroad.stuff.vaadin;
009    
010    import com.vaadin.Application;
011    import com.vaadin.terminal.gwt.server.ApplicationServlet;
012    
013    import javax.servlet.ServletConfig;
014    import javax.servlet.ServletException;
015    import javax.servlet.http.HttpServletRequest;
016    
017    import org.slf4j.Logger;
018    import org.slf4j.LoggerFactory;
019    import org.springframework.beans.BeansException;
020    import org.springframework.beans.factory.config.AutowireCapableBeanFactory;
021    import org.springframework.web.context.WebApplicationContext;
022    import org.springframework.web.context.support.WebApplicationContextUtils;
023    
024    /**
025     * {@link ApplicationServlet} that autowires and configures the {@link Application}
026     * objects it creates using the associated Spring {@link WebApplicationContext}.
027     * This allows a Vaadin application to be configured normally as a Spring bean.
028     *
029     * <p>
030     * For example, annotations such as
031     * <code>{@link org.springframework.beans.factory.annotation.Autowired @Autowired}</code>,
032     * <code>{@link org.springframework.beans.factory.annotation.Required @Required}</code>, etc.
033     * and interfaces such as {@link org.springframework.beans.factory.BeanFactoryAware BeanFactoryAware},
034     * etc. will work on your {@link Application} instances.
035     * </p>
036     *
037     * <p>
038     * An example of "direct" use of this servlet in conjunction with Spring's
039     * {@link org.springframework.web.context.ContextLoaderListener ContextLoaderListener}:
040     * <blockquote><pre>
041     *  &lt;!-- Spring context loader --&gt;
042     *  &lt;listener&gt;
043     *      &lt;listener-class&gt;org.springframework.web.context.ContextLoaderListener&lt;/listener-class&gt;
044     *  &lt;/listener&gt;
045     *
046     *  &lt;!-- Vaadin servlet --&gt;
047     *  &lt;servlet&gt;
048     *      &lt;servlet-name&gt;myapp&lt;/servlet-name&gt;
049     *      &lt;servlet-class&gt;com.example.AutowiringApplicationServlet&lt;/servlet-class&gt;
050     *      &lt;init-param&gt;
051     *          &lt;param-name&gt;application&lt;/param-name&gt;
052     *          &lt;param-value&gt;some.spring.configured.Application&lt;/param-value&gt;
053     *      &lt;/init-param&gt;
054     *      &lt;init-param&gt;
055     *          &lt;param-name&gt;productionMode&lt;/param-name&gt;
056     *          &lt;param-value&gt;true&lt;/param-value&gt;
057     *      &lt;/init-param&gt;
058     *  &lt;/servlet&gt;
059     *  &lt;servlet-mapping&gt;
060     *      &lt;servlet-name&gt;myapp&lt;/servlet-name&gt;
061     *      &lt;url-pattern&gt;/*&lt;/url-pattern&gt;
062     *  &lt;/servlet-mapping&gt;
063     * </pre></blockquote>
064     * </p>
065     *
066     * <p>
067     * An example that creates a Spring MVC "controller" bean for use with Spring's
068     * {@link org.springframework.web.servlet.DispatcherServlet DispatcherServlet}:
069     * <blockquote><pre>
070     *  &lt;!-- Activate Spring annotation support --&gt;
071     *  &lt;context:annotation-config/&gt;
072     *
073     *  &lt;!-- Define controller bean for Vaadin application --&gt;
074     *  &lt;bean id="applicationController" class="org.springframework.web.servlet.mvc.ServletWrappingController"
075     *     p:servletClass="com.example.AutowiringApplicationServlet"&gt;
076     *      &lt;property name="initParameters"&gt;
077     *          &lt;props&gt;
078     *              &lt;prop key="application"&gt;some.spring.configured.Application&lt;/prop&gt;
079     *              &lt;prop key="productionMode"&gt;true&lt;/prop&gt;
080     *          &lt;/props&gt;
081     *      &lt;/property&gt;
082     *  &lt;/bean&gt;
083     * </pre></blockquote>
084     *
085     * @see org.springframework.web.servlet.mvc.ContextLoaderListener
086     * @see org.springframework.web.servlet.DispatcherServlet
087     * @see org.springframework.web.servlet.mvc.ServletWrappingController
088     * @see AutowireCapableBeanFactory
089     */
090    @SuppressWarnings("serial")
091    public class AutowiringApplicationServlet extends ApplicationServlet {
092    
093        protected final Logger log = LoggerFactory.getLogger(getClass());
094    
095        private WebApplicationContext webApplicationContext;
096    
097        /**
098         * Initialize this servlet.
099         *
100         * @throws ServletException if there is no {@link WebApplicationContext} associated with this servlet's context
101         */
102        @Override
103        public void init(ServletConfig config) throws ServletException {
104            super.init(config);
105            log.debug("finding containing WebApplicationContext");
106            try {
107                this.webApplicationContext = WebApplicationContextUtils.getRequiredWebApplicationContext(config.getServletContext());
108            } catch (IllegalStateException e) {
109                throw new ServletException("could not locate containing WebApplicationContext");
110            }
111        }
112    
113        /**
114         * Get the containing Spring {@link WebApplicationContext}.
115         * This only works after the servlet has been initialized (via {@link #init init()}).
116         *
117         * @throws ServletException if the operation fails
118         */
119        protected final WebApplicationContext getWebApplicationContext() throws ServletException {
120            if (this.webApplicationContext == null)
121                throw new ServletException("can't retrieve WebApplicationContext before init() is invoked");
122            return this.webApplicationContext;
123        }
124    
125        /**
126         * Get the {@link AutowireCapableBeanFactory} associated with the containing Spring {@link WebApplicationContext}.
127         * This only works after the servlet has been initialized (via {@link #init init()}).
128         *
129         * @throws ServletException if the operation fails
130         */
131        protected final AutowireCapableBeanFactory getAutowireCapableBeanFactory() throws ServletException {
132            try {
133                return getWebApplicationContext().getAutowireCapableBeanFactory();
134            } catch (IllegalStateException e) {
135                throw new ServletException("containing context " + getWebApplicationContext() + " is not autowire-capable", e);
136            }
137        }
138    
139        /**
140         * Create and configure a new instance of the configured application class.
141         *
142         * <p>
143         * The implementation in {@link AutowiringApplicationServlet} delegates to
144         * {@link #getAutowireCapableBeanFactory getAutowireCapableBeanFactory()}, then invokes
145         * {@link AutowireCapableBeanFactory#createBean AutowireCapableBeanFactory.createBean()}
146         * using the configured {@link Application} class.
147         * </p>
148         *
149         * @param request the triggering {@link HttpServletRequest}
150         * @throws ServletException if creation or autowiring fails
151         */
152        @Override
153        protected Application getNewApplication(HttpServletRequest request) throws ServletException {
154            Class<? extends Application> cl;
155            try {
156                cl = getApplicationClass();
157            } catch (ClassNotFoundException e) {
158                throw new ServletException("failed to create new instance of application class", e);
159            }
160            log.debug("creating new instance of " + cl);
161            AutowireCapableBeanFactory beanFactory = getAutowireCapableBeanFactory();
162            try {
163                return beanFactory.createBean(cl);
164            } catch (BeansException e) {
165                throw new ServletException("failed to create new instance of " + cl, e);
166            }
167        }
168    }
