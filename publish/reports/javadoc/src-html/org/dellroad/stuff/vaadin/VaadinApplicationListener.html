008    package org.dellroad.stuff.vaadin;
009    
010    import org.springframework.context.ApplicationEvent;
011    import org.springframework.context.event.ApplicationEventMulticaster;
012    import org.springframework.context.event.SmartApplicationListener;
013    
014    /**
015     * A Spring {@link org.springframework.context.ApplicationListener} support superclass customized for use by
016     * listeners that are part of a Vaadin application when listening to non-Vaadin application event sources.
017     *
018     * <p>
019     * Listeners that are part of a Vaadin application should use this superclass if they are going to be registered
020     * with non-Vaadin event multicasters. This will ensure that events are delivered {@linkplain ContextApplication#invoke
021     * in the proper Vaadin application context} and memory leaks are avoided.
022     * </p>
023     *
024     * @param <E> The type of the event
025     * @see VaadinExternalListener
026     * @see ContextApplication#invoke
027     * @see VaadinApplicationScope
028     * @see SpringContextApplication
029     */
030    public abstract class VaadinApplicationListener<E extends ApplicationEvent>
031      extends VaadinExternalListener<ApplicationEventMulticaster> implements SmartApplicationListener {
032    
033        private final Class<E> eventType;
034    
035        /**
036         * Convenience constructor. Equivalent to:
037         * <blockquote>
038         *  {@link #VaadinApplicationListener(ApplicationEventMulticaster, Class, ContextApplication)
039         *      VaadinApplicationListener(multicaster, eventType, ContextApplication.get())}
040         * </blockquote>
041         *
042         * @param multicaster the application event multicaster on which this listener will register
043         * @throws IllegalArgumentException if {@code eventType} is null
044         * @throws IllegalStateException if there is no {@link ContextApplication} associated with the current thread
045         */
046        public VaadinApplicationListener(ApplicationEventMulticaster multicaster, Class<E> eventType) {
047            this(multicaster, eventType, ContextApplication.get());
048        }
049    
050        /**
051         * Primary constructor. This results in this instance being registered as a listener on {@code multicaster}.
052         *
053         * @param multicaster the application event multicaster on which this listener will register
054         * @param eventType type of events this instance should receive (others will be ignored)
055         * @param application the associated Vaadin application
056         * @throws IllegalArgumentException if either parameter is null
057         */
058        public VaadinApplicationListener(ApplicationEventMulticaster multicaster, Class<E> eventType, ContextApplication application) {
059            super(multicaster, application);
060            if (eventType == null)
061                throw new IllegalArgumentException("null eventType");
062            this.eventType = eventType;
063        }
064    
065        /**
066         * Get the event type that this listener listens for.
067         */
068        public final Class<E> getEventType() {
069            return this.eventType;
070        }
071    
072        @Override
073        protected void register(ApplicationEventMulticaster eventSource) {
074            this.getEventSource().addApplicationListener(this);
075        }
076    
077        @Override
078        protected void unregister(ApplicationEventMulticaster eventSource) {
079            this.getEventSource().removeApplicationListener(this);
080        }
081    
082        @Override
083        public final void onApplicationEvent(ApplicationEvent event) {
084            E castEvent;
085            try {
086                castEvent = this.eventType.cast(event);
087            } catch (ClassCastException e) {
088                // should not happen
089                return;
090            }
091            final E castEvent2 = castEvent;
092            this.handleEvent(new Runnable() {
093                @Override
094                public void run() {
095                    VaadinApplicationListener.this.onApplicationEventInternal(castEvent2);
096                }
097            });
098        }
099    
100        /**
101         * Handle a listener event. When this method is invoked, it will already be within the context
102         * of the {@link ContextApplication} with which this listener is associated.
103         * The current {@link ContextApplication} is also available via {@link ContextApplication#get}.
104         *
105         * @see ContextApplication#get
106         */
107        protected abstract void onApplicationEventInternal(E event);
108    
109        /**
110         * Determine whether this listener actually supports the given event type.
111         *
112         * <p>
113         * The implementation in {@link VaadinApplicationListener} tests whether {@code eventType}
114         * is assignable to the type given in the constructor. Subclasses may override as desired.
115         */
116        @Override
117        public boolean supportsEventType(Class<? extends ApplicationEvent> eventType) {
118            return this.eventType.isAssignableFrom(eventType);
119        }
120    
121        /**
122         * Determine whether this listener actually supports the given source type.
123         *
124         * <p>
125         * The implementation in {@link VaadinApplicationListener} always returns true. Subclasses may override as desired.
126         */
127        @Override
128        public boolean supportsSourceType(Class<?> sourceType) {
129            return true;
130        }
131    
132        /**
133         * Get ordering value.
134         *
135         * <p>
136         * The implementation in {@link VaadinApplicationListener} always returns zero. Subclasses may override as desired.
137         */
138        @Override
139        public int getOrder() {
140            return 0;
141        }
142    }
