008    package org.dellroad.stuff.vaadin;
009    
010    import org.springframework.context.ApplicationEvent;
011    import org.springframework.context.event.ApplicationEventMulticaster;
012    import org.springframework.context.event.SmartApplicationListener;
013    
014    /**
015     * A Spring {@link org.springframework.context.ApplicationListener} support superclass customized for use by
016     * listeners that are part of a Vaadin application when listening to non-Vaadin application event sources.
017     *
018     * <p>
019     * Listeners that are part of a Vaadin application should use this superclass if they are going to be registered
020     * with non-Vaadin event sources. This will ensure that events are delivered {@linkplain ContextApplication#invoke
021     * in the proper Vaadin application context}.
022     * </p>
023     *
024     * <p>
025     * Note: to avoid memory leaks, listeners must be explicitly unregistered when the associated Vaadin application closes.
026     * This is done automatically when {@link #addAsListenerTo addAsListenerTo()} is used to register this instance as a listener.
027     * Alternately, you can explicitly unregister this listener in a Spring
028     * {@linkplain org.springframework.beans.factory.DisposableBean#destroy destroy-method} associated with a
029     * bean that has {@link VaadinApplicationScope scope="vaadinApplication"} or lives in a {@link SpringContextApplication}
030     * application context (so that the bean's destroy method will be invoked when the Vaadin application closes), etc.
031     * However, Spring does not provide any {@code removeApplicationListener()} method in the
032     * {@link org.springframework.context.ConfigurableApplicationContext} class itself, so an explicitly-defined
033     * {@link ApplicationEventMulticaster} must be used for this to work.
034     * </p>
035     *
036     * <p>
037     * Note: when listening to event sources that are scoped to specific Vaadin application instances and originate events
038     * within the proper Vaadin application context, then the use of this listener superclass is not necessary.
039     * </p>
040     *
041     * @see ContextApplication#invoke
042     * @see VaadinApplicationScope
043     * @see SpringContextApplication
044     */
045    public abstract class VaadinApplicationListener<E extends ApplicationEvent> implements SmartApplicationListener {
046    
047        private final Class<E> eventType;
048        private final ContextApplication application;
049    
050        /**
051         * Convenience constructor. Equivalent to:
052         * <blockquote>
053         *  {@link #VaadinApplicationListener(Class, ContextApplication) VaadinApplicationListener(eventType, ContextApplication.get())}
054         * </blockquote>
055         *
056         * @throws IllegalArgumentException if {@code eventType} is null
057         * @throws IllegalStateException if there is no {@link ContextApplication} associated with the current thread
058         */
059        public VaadinApplicationListener(Class<E> eventType) {
060            this(eventType, ContextApplication.get());
061        }
062    
063        /**
064         * Primary constructor.
065         *
066         * @param eventType type of events this instance should receive (others will be ignored)
067         * @param application the associated Vaadin application
068         * @throws IllegalArgumentException if either parameter is null
069         */
070        public VaadinApplicationListener(Class<E> eventType, ContextApplication application) {
071            if (eventType == null)
072                throw new IllegalArgumentException("null eventType");
073            if (application == null)
074                throw new IllegalArgumentException("null application");
075            this.eventType = eventType;
076            this.application = application;
077        }
078    
079        /**
080         * Register as a listener on the given {@link ApplicationEventMulticaster} and also register a
081         * {@link ContextApplication.CloseListener} on the {@linkplain #getApplication configured Vaadin application}
082         * so that when the application closes we can unregister this instance from the provided event multicaster
083         * to avoid a memory leak.
084         *
085         * <p>
086         * This method (or some other means) must be used to avoid a memory leak when the Vaadin application closes.
087         *
088         * @param eventMulticaster object with which to register as a listener and from which to unregister
089         *  when the Vaadin application closes
090         * @throws IllegalArgumentException if {@code eventMulticaster} is null
091         */
092        public void addAsListenerTo(final ApplicationEventMulticaster eventMulticaster) {
093    
094            // Sanity check
095            if (eventMulticaster == null)
096                throw new IllegalArgumentException("null eventMulticaster");
097    
098            // Register myself as a listener
099            eventMulticaster.addApplicationListener(this);
100    
101            // Unregister myself on Application close
102            this.getApplication().addListener(new ContextApplication.CloseListener() {
103                @Override
104                public void applicationClosed(ContextApplication.CloseEvent closeEvent) {
105                    eventMulticaster.removeApplicationListener(VaadinApplicationListener.this);
106                }
107            });
108        }
109    
110        /**
111         * Get the event type that this listener listens for.
112         */
113        public final Class<E> getEventType() {
114            return this.eventType;
115        }
116    
117        /**
118         * Get the {@link ContextApplication} that this instance will set as the "current Vaadin appliction"
119         * when {@link #onApplicationEventInternal} is invoked.
120         */
121        public final ContextApplication getApplication() {
122            return this.application;
123        }
124    
125        @Override
126        public final void onApplicationEvent(ApplicationEvent event) {
127            E castEvent;
128            try {
129                castEvent = this.eventType.cast(event);
130            } catch (ClassCastException e) {
131                // should not happen
132                return;
133            }
134            final E castEvent2 = castEvent;
135            this.application.invoke(new Runnable() {
136                @Override
137                public void run() {
138                    VaadinApplicationListener.this.onApplicationEventInternal(castEvent2);
139                }
140            });
141        }
142    
143        /**
144         * Handle a listener event within the context of the {@link ContextApplication} with which this listener is associated.
145         * The current {@link ContextApplication} is also available via {@link ContextApplication#get}.
146         *
147         * @see ContextApplication#get
148         */
149        protected abstract void onApplicationEventInternal(E event);
150    
151        /**
152         * Determine whether this listener actually supports the given event type.
153         *
154         * <p>
155         * The implementation in {@link VaadinApplicationListener} tests whether {@code eventType}
156         * is assignable to the type given in the constructor. Subclasses may override as desired.
157         */
158        @Override
159        public boolean supportsEventType(@SuppressWarnings("hiding") Class<? extends ApplicationEvent> eventType) {
160            return this.eventType.isAssignableFrom(eventType);
161        }
162    
163        /**
164         * Determine whether this listener actually supports the given source type.
165         *
166         * <p>
167         * The implementation in {@link VaadinApplicationListener} always returns true. Subclasses may override as desired.
168         */
169        @Override
170        public boolean supportsSourceType(Class<?> sourceType) {
171            return true;
172        }
173    
174        /**
175         * Get ordering value.
176         *
177         * <p>
178         * The implementation in {@link VaadinApplicationListener} always returns zero. Subclasses may override as desired.
179         */
180        @Override
181        public int getOrder() {
182            return 0;
183        }
184    }
