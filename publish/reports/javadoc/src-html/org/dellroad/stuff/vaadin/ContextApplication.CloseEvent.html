008    package org.dellroad.stuff.vaadin;
009    
010    import com.vaadin.Application;
011    import com.vaadin.terminal.Terminal;
012    import com.vaadin.terminal.gwt.server.HttpServletRequestListener;
013    import com.vaadin.ui.Window;
014    
015    import java.net.SocketException;
016    import java.util.EventObject;
017    import java.util.HashSet;
018    
019    import javax.servlet.http.HttpServletRequest;
020    import javax.servlet.http.HttpServletResponse;
021    
022    import org.slf4j.Logger;
023    import org.slf4j.LoggerFactory;
024    
025    /**
026     * {@link Application} subclass that provides subclasses with a {@link Logger}
027     * and logs and displays any exceptions thrown in an overlay error window,
028     * as well as static methods to access the current {@link ContextApplication} instance.
029     *
030     * @since 1.0.134
031     */
032    @SuppressWarnings("serial")
033    public abstract class ContextApplication extends Application implements HttpServletRequestListener {
034    
035        /**
036         * Default notification linger time for error notifications (in milliseconds): Value is {@value}ms.
037         */
038        public static final int DEFAULT_NOTIFICATION_DELAY = 30000;
039    
040        private static final ThreadLocal<ContextApplication> CURRENT_APPLICATION = new ThreadLocal<ContextApplication>();
041        private static final ThreadLocal<HttpServletRequest> CURRENT_REQUEST = new ThreadLocal<HttpServletRequest>();
042        private static final ThreadLocal<HttpServletResponse> CURRENT_RESPONSE = new ThreadLocal<HttpServletResponse>();
043    
044        protected final Logger log = LoggerFactory.getLogger(this.getClass());
045    
046        private final HashSet<CloseListener> closeListeners = new HashSet<CloseListener>();
047    
048    // Initialization
049    
050        /**
051         * Initialize the application.
052         *
053         * <p>
054         * The implementation in {@link ContextApplication} delegates to {@link #initApplication}.
055         * </p>
056         */
057        @Override
058        public final void init() {
059            CURRENT_APPLICATION.set(this);
060            this.initApplication();
061        }
062    
063        /**
064         * Initialize the application. Sub-classes of {@link ContextApplication} must implement this method.
065         */
066        protected abstract void initApplication();
067    
068    // Error handling
069    
070        /**
071         * Handle an uncaugt exception thrown by a Vaadin HTTP request.
072         *
073         * <p>
074         * The implementation in {@link ContextApplication} logs the error and displays in on the
075         * user's screen via {@link #showError(String, Throwable)}.
076         */
077        @Override
078        public void terminalError(Terminal.ErrorEvent event) {
079    
080            // Delegate to superclass
081            super.terminalError(event);
082    
083            // Get exception; ignore client "hangups"
084            final Throwable t = event.getThrowable();
085            if (t instanceof SocketException)
086                return;
087    
088            // Notify user and log it
089            this.showError("Internal Error", "" + t);
090            this.log.error("error within Vaadin operation", t);
091        }
092    
093        /**
094         * Display an error message to the user.
095         */
096        public void showError(String title, String description) {
097            Window.Notification notification = new Window.Notification(title, description, Window.Notification.TYPE_ERROR_MESSAGE);
098            notification.setStyleName("warning");
099            notification.setDelayMsec(this.getNotificationDelay());
100            this.getMainWindow().showNotification(notification);
101        }
102    
103        /**
104         * Display an error message to the user caused by an exception.
105         */
106        public void showError(String title, Throwable t) {
107            for (int i = 0; i < 100 && t.getCause() != null; i++)
108                t = t.getCause();
109            this.showError(title, this.getErrorMessage(t));
110        }
111    
112        /**
113         * Get the notification linger time for error notifications (in milliseconds).
114         *
115         * <p>
116         * The implementation in {@link ContextApplication} returns {@link #DEFAULT_NOTIFICATION_DELAY}.
117         */
118        protected int getNotificationDelay() {
119            return DEFAULT_NOTIFICATION_DELAY;
120        }
121    
122        /**
123         * Convert an exception into a displayable error message.
124         */
125        protected String getErrorMessage(Throwable t) {
126            return t.getClass().getSimpleName() + ": " + t.getMessage();
127        }
128    
129    // ThreadLocal stuff
130    
131        /**
132         * Set this instance as the "current application" (if not set already) while invoking the given callback.
133         *
134         * <p>
135         * This method is useful for situations in which non-Vaadin threads need to call into Vaadin code
136         * that expects to successfully retrieve the current application via {@link #currentApplication}.
137         * </p>
138         *
139         * <p>
140         * This method also synchronizes on this {@link ContextApplication} instance as required by Vaadin for thread safety.
141         * </p>
142         *
143         * @param action action to perform
144         * @throws IllegalStateException if a different {@link ContextApplication} is already set as the current application
145         *  associated with the current thread
146         */
147        public void invoke(Runnable action) {
148            final ContextApplication previous = ContextApplication.CURRENT_APPLICATION.get();
149            if (previous != null && previous != this)
150                throw new IllegalStateException("there is already a current application for this thread");
151            ContextApplication.CURRENT_APPLICATION.set(this);
152            try {
153                synchronized (this) {
154                    action.run();
155                }
156            } finally {
157                ContextApplication.CURRENT_APPLICATION.set(previous);
158            }
159        }
160    
161        /**
162         * Handle the start of a request.
163         *
164         * <p>
165         * The implementation in {@link ContextApplication} delegates to {@link #doOnRequestStart}.
166         * </p>
167         */
168        @Override
169        public final void onRequestStart(HttpServletRequest request, HttpServletResponse response) {
170            ContextApplication.CURRENT_APPLICATION.set(this);
171            ContextApplication.CURRENT_REQUEST.set(request);
172            ContextApplication.CURRENT_RESPONSE.set(response);
173            this.doOnRequestStart(request, response);
174        }
175    
176        /**
177         * Handle the end of a request.
178         *
179         * <p>
180         * The implementation in {@link ContextApplication} delegates to {@link #doOnRequestEnd}.
181         * </p>
182         */
183        @Override
184        public final void onRequestEnd(HttpServletRequest request, HttpServletResponse response) {
185            try {
186                this.doOnRequestEnd(request, response);
187            } finally {
188                ContextApplication.CURRENT_APPLICATION.remove();
189                ContextApplication.CURRENT_REQUEST.remove();
190                ContextApplication.CURRENT_RESPONSE.remove();
191            }
192        }
193    
194        /**
195         * Sub-class hook for handling the start of a request. This method is invoked by {@link #onRequestStart}.
196         *
197         * <p>
198         * The implementation in {@link ContextApplication} does nothing. Subclasses should override as necessary.
199         * </p>
200         */
201        protected void doOnRequestStart(HttpServletRequest request, HttpServletResponse response) {
202        }
203    
204        /**
205         * Sub-class hook for handling the end of a request. This method is invoked by {@link #onRequestEnd}.
206         *
207         * <p>
208         * The implementation in {@link ContextApplication} does nothing. Subclasses should override as necessary.
209         * </p>
210         */
211        protected void doOnRequestEnd(HttpServletRequest request, HttpServletResponse response) {
212        }
213    
214        /**
215         * Get the {@link ContextApplication} instance associated with the current thread.
216         *
217         * <p>
218         * If the current thread is handling a Vaadin HTTP request that is executing within an {@link ContextApplication} instance,
219         * or is executing within {@link #invoke}, then this method will return the associated {@link ContextApplication}.
220         * </p>
221         *
222         * @return the {@link ContextApplication} associated with the current thread, or {@code null} if the current thread
223         *  is not servicing a Vaadin web request or the current Vaadin {@link Application} is not an {@link ContextApplication}
224         *
225         * @see #invoke
226         */
227        public static ContextApplication currentApplication() {
228            return ContextApplication.CURRENT_APPLICATION.get();
229        }
230    
231        /**
232         * Get the {@link ContextApplication} associated with the current thread, cast to the desired type.
233         *
234         * @param type expected application type
235         * @return the {@link ContextApplication} associated with the current thread
236         * @see #currentApplication()
237         * @throws ClassCastException if the current application is not assignable to {@code type}
238         */
239        public static <A extends ContextApplication> A currentApplication(Class<A> type) {
240            return type.cast(ContextApplication.currentApplication());
241        }
242    
243        /**
244         * Get the {@link ContextApplication} instance associated with the current thread or throw an exception if there is none.
245         *
246         * <p>
247         * If the current thread is handling a Vaadin web request that is executing within an {@link ContextApplication} instance,
248         * or is executing within {@link #invoke}, then this method will return the associated {@link ContextApplication}.
249         * Otherwise, an exception is thrown.
250         * </p>
251         *
252         * @return the {@link ContextApplication} associated with the current thread
253         * @throws IllegalStateException if the current thread is not servicing a Vaadin web request
254         *  or the current Vaadin {@link Application} is not an {@link ContextApplication}
255         */
256        public static ContextApplication get() {
257            ContextApplication app = ContextApplication.currentApplication();
258            if (app != null)
259                return app;
260            throw new IllegalStateException("no current application found");
261        }
262    
263        /**
264         * Get the {@link ContextApplication} instance associated with the current thread, cast to the desired type,
265         * or throw an exception if there is none.
266         *
267         * @param type expected application type
268         * @return the {@link ContextApplication} associated with the current thread
269         * @throws IllegalStateException if the current {@link ContextApplication} is not found
270         * @see #get()
271         * @throws ClassCastException if the current application is not assignable to {@code type}
272         */
273        public static <A extends ContextApplication> A get(Class<A> type) {
274            A app = ContextApplication.currentApplication(type);
275            if (app != null)
276                return app;
277            throw new IllegalStateException("no current application found");
278        }
279    
280        /**
281         * Get the {@link HttpServletRequest} associated with the current thread.
282         *
283         * <p>
284         * If the current thread is handling a Vaadin web request for an instance of this class,
285         * this method will return the associated {@link HttpServletRequest}.
286         * </p>
287         *
288         * @return the {@link HttpServletRequest} associated with the current thread, or {@code null} if the current thread
289         *  is not servicing a Vaadin web request or the current Vaadin {@link Application} is not an instance of this class
290         */
291        public static HttpServletRequest currentRequest() {
292            return ContextApplication.CURRENT_REQUEST.get();
293        }
294    
295        /**
296         * Get the {@link HttpServletResponse} associated with the current thread.
297         *
298         * <p>
299         * If the current thread is handling a Vaadin web request for an instance of this class,
300         * this method will return the associated {@link HttpServletResponse}.
301         * </p>
302         *
303         * @return the {@link HttpServletResponse} associated with the current thread, or {@code null} if the current thread
304         *  is not servicing a Vaadin web request or the current Vaadin {@link Application} is not an instance of this class
305         */
306        public static HttpServletResponse currentResponse() {
307            return ContextApplication.CURRENT_RESPONSE.get();
308        }
309    
310    // Listener stuff
311    
312        /**
313         * Close this instance.
314         *
315         * <p>
316         * The implementation in {@link ContextApplication} first delegates to the superclass and then
317         * notifies any registered {@link CloseListener}s.
318         * </p>
319         */
320        @Override
321        public void close() {
322            super.close();
323            CloseEvent closeEvent = new CloseEvent(this);
324            for (CloseListener closeListener : this.getCloseListeners()) {
325                try {
326                    closeListener.applicationClosed(closeEvent);
327                } catch (ThreadDeath t) {
328                    throw t;
329                } catch (Throwable t) {
330                    this.log.error("exception thrown by CloseListener", t);
331                }
332            }
333        }
334    
335        /**
336         * Add a {@link CloseListener} to be notified when this instance is closed.
337         */
338        public void addListener(CloseListener listener) {
339            synchronized (this.closeListeners) {
340                this.closeListeners.add(listener);
341            }
342        }
343    
344        /**
345         * Remove a {@link CloseListener}.
346         */
347        public void removeListener(CloseListener listener) {
348            synchronized (this.closeListeners) {
349                this.closeListeners.remove(listener);
350            }
351        }
352    
353        private HashSet<CloseListener> getCloseListeners() {
354            synchronized (this.closeListeners) {
355                return new HashSet<CloseListener>(this.closeListeners);
356            }
357        }
358    
359        /**
360         * Implemented by listeners that wish to be notified when a {@link ContextApplication} is closed.
361         */
362        public interface CloseListener {
363    
364            /**
365             * Notification upon {@link ContextApplication} closing.
366             */
367            void applicationClosed(CloseEvent closeEvent);
368        }
369    
370        /**
371         * Event delivered to listeners when a {@link ContextApplication} is closed.
372         */
373        public static class CloseEvent extends EventObject {
374    
375            public CloseEvent(ContextApplication application) {
376                super(application);
377            }
378    
379            /**
380             * Get the {@link ContextApplication} that is being closed.
381             */
382            public ContextApplication getContextApplication() {
383                return (ContextApplication)super.getSource();
384            }
385        }
386    }
