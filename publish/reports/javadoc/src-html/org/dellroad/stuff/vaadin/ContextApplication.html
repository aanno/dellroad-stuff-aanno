008    package org.dellroad.stuff.vaadin;
009    
010    import com.vaadin.Application;
011    import com.vaadin.terminal.Terminal;
012    import com.vaadin.terminal.gwt.server.HttpServletRequestListener;
013    import com.vaadin.ui.Window;
014    
015    import java.net.SocketException;
016    
017    import javax.servlet.http.HttpServletRequest;
018    import javax.servlet.http.HttpServletResponse;
019    
020    import org.slf4j.Logger;
021    import org.slf4j.LoggerFactory;
022    
023    /**
024     * {@link Application} subclass that provides subclasses with a {@link Logger}
025     * and logs and displays any exceptions thrown in an overlay error window,
026     * as well as static methods to access the current {@link ContextApplication} instance.
027     *
028     * @since 1.0.134
029     */
030    @SuppressWarnings("serial")
031    public abstract class ContextApplication extends Application implements HttpServletRequestListener {
032    
033        /**
034         * Default notification linger time for error notifications (in milliseconds): Value is {@value}ms.
035         */
036        public static final int DEFAULT_NOTIFICATION_DELAY = 30000;
037    
038        private static final ThreadLocal<ContextApplication> CURRENT_APPLICATION = new ThreadLocal<ContextApplication>();
039        private static final ThreadLocal<HttpServletRequest> CURRENT_REQUEST = new ThreadLocal<HttpServletRequest>();
040        private static final ThreadLocal<HttpServletResponse> CURRENT_RESPONSE = new ThreadLocal<HttpServletResponse>();
041    
042        protected final Logger log = LoggerFactory.getLogger(getClass());
043    
044    // Initialization
045    
046        /**
047         * Initialize the application.
048         *
049         * <p>
050         * The implementation in {@link ContextApplication} delegates to {@link #initApplication}.
051         * </p>
052         */
053        @Override
054        public final void init() {
055            CURRENT_APPLICATION.set(this);
056            this.initApplication();
057        }
058    
059        /**
060         * Initialize the application. Sub-classes of {@link ContextApplication} must implement this method.
061         */
062        protected abstract void initApplication();
063    
064    // Error handling
065    
066        /**
067         * Handle an uncaugt exception thrown by a Vaadin HTTP request.
068         *
069         * <p>
070         * The implementation in {@link ContextApplication} logs the error and displays in on the
071         * user's screen via {@link #showError(String, Throwable)}.
072         */
073        @Override
074        public void terminalError(Terminal.ErrorEvent event) {
075    
076            // Delegate to superclass
077            super.terminalError(event);
078    
079            // Get exception; ignore client "hangups"
080            final Throwable t = event.getThrowable();
081            if (t instanceof SocketException)
082                return;
083    
084            // Notify user and log it
085            this.showError("Internal Error", "" + t);
086            this.log.error("error within Vaadin operation", t);
087        }
088    
089        /**
090         * Display an error message to the user.
091         */
092        public void showError(String title, String description) {
093            Window.Notification notification = new Window.Notification(title, description, Window.Notification.TYPE_ERROR_MESSAGE);
094            notification.setStyleName("warning");
095            notification.setDelayMsec(this.getNotificationDelay());
096            this.getMainWindow().showNotification(notification);
097        }
098    
099        /**
100         * Display an error message to the user caused by an exception.
101         */
102        public void showError(String title, Throwable t) {
103            for (int i = 0; i < 100 && t.getCause() != null; i++)
104                t = t.getCause();
105            this.showError(title, this.getErrorMessage(t));
106        }
107    
108        /**
109         * Get the notification linger time for error notifications (in milliseconds).
110         *
111         * <p>
112         * The implementation in {@link ContextApplication} returns {@link #DEFAULT_NOTIFICATION_DELAY}.
113         */
114        protected int getNotificationDelay() {
115            return DEFAULT_NOTIFICATION_DELAY;
116        }
117    
118        /**
119         * Convert an exception into a displayable error message.
120         */
121        protected String getErrorMessage(Throwable t) {
122            return t.getClass().getSimpleName() + ": " + t.getMessage();
123        }
124    
125    // ThreadLocal stuff
126    
127        /**
128         * Set this instance as the "current application" (if not set already) while invoking the given callback.
129         *
130         * <p>
131         * This method is useful for situations in which non-Vaadin threads need to call into Vaadin code
132         * that expects to successfully retrieve the current application via {@link #currentApplication}.
133         * </p>
134         *
135         * <p>
136         * This method also synchronizes on this {@link ContextApplication} instance as required by Vaadin for thread safety.
137         * </p>
138         *
139         * @param action action to perform
140         * @throws IllegalStateException if a different {@link ContextApplication} is already set as the current application
141         *  associated with the current thread
142         */
143        public void invoke(Runnable action) {
144            final ContextApplication previous = ContextApplication.CURRENT_APPLICATION.get();
145            if (previous != null && previous != this)
146                throw new IllegalStateException("there is already a current application for this thread");
147            ContextApplication.CURRENT_APPLICATION.set(this);
148            try {
149                synchronized (this) {
150                    action.run();
151                }
152            } finally {
153                ContextApplication.CURRENT_APPLICATION.set(previous);
154            }
155        }
156    
157        /**
158         * Handle the start of a request.
159         *
160         * <p>
161         * The implementation in {@link ContextApplication} delegates to {@link #doOnRequestStart}.
162         * </p>
163         */
164        @Override
165        public final void onRequestStart(HttpServletRequest request, HttpServletResponse response) {
166            ContextApplication.CURRENT_APPLICATION.set(this);
167            ContextApplication.CURRENT_REQUEST.set(request);
168            ContextApplication.CURRENT_RESPONSE.set(response);
169            this.doOnRequestStart(request, response);
170        }
171    
172        /**
173         * Handle the end of a request.
174         *
175         * <p>
176         * The implementation in {@link ContextApplication} delegates to {@link #doOnRequestEnd}.
177         * </p>
178         */
179        @Override
180        public final void onRequestEnd(HttpServletRequest request, HttpServletResponse response) {
181            try {
182                this.doOnRequestEnd(request, response);
183            } finally {
184                ContextApplication.CURRENT_APPLICATION.remove();
185                ContextApplication.CURRENT_REQUEST.remove();
186                ContextApplication.CURRENT_RESPONSE.remove();
187            }
188        }
189    
190        /**
191         * Sub-class hook for handling the start of a request. This method is invoked by {@link #onRequestStart}.
192         *
193         * <p>
194         * The implementation in {@link ContextApplication} does nothing. Subclasses should override as necessary.
195         * </p>
196         */
197        protected void doOnRequestStart(HttpServletRequest request, HttpServletResponse response) {
198        }
199    
200        /**
201         * Sub-class hook for handling the end of a request. This method is invoked by {@link #onRequestEnd}.
202         *
203         * <p>
204         * The implementation in {@link ContextApplication} does nothing. Subclasses should override as necessary.
205         * </p>
206         */
207        protected void doOnRequestEnd(HttpServletRequest request, HttpServletResponse response) {
208        }
209    
210        /**
211         * Get the {@link ContextApplication} instance associated with the current thread.
212         *
213         * <p>
214         * If the current thread is handling a Vaadin web request that is executing within an {@link ContextApplication} instance,
215         * then this method will return the associated {@link ContextApplication}.
216         * </p>
217         *
218         * @return the {@link ContextApplication} associated with the current thread, or {@code null} if the current thread
219         *  is not servicing a Vaadin web request or the current Vaadin {@link Application} is not an {@link ContextApplication}
220         */
221        public static ContextApplication currentApplication() {
222            return ContextApplication.CURRENT_APPLICATION.get();
223        }
224    
225        /**
226         * Get the {@link ContextApplication} associated with the current thread, cast to the desired type.
227         *
228         * @param type expected application type
229         * @return the {@link ContextApplication} associated with the current thread
230         * @see #currentApplication()
231         * @throws ClassCastException if the current application is not assignable to {@code type}
232         */
233        public static <A extends ContextApplication> A currentApplication(Class<A> type) {
234            return type.cast(ContextApplication.currentApplication());
235        }
236    
237        /**
238         * Get the {@link ContextApplication} instance associated with the current thread or throw an exception if there is none.
239         *
240         * <p>
241         * If the current thread is handling a Vaadin web request that is executing within an {@link ContextApplication} instance,
242         * then this method will return the associated {@link ContextApplication}. Otherwise, an exception is thrown.
243         * </p>
244         *
245         * @return the {@link ContextApplication} associated with the current thread
246         * @throws IllegalStateException if the current thread is not servicing a Vaadin web request
247         *  or the current Vaadin {@link Application} is not an {@link ContextApplication}
248         */
249        public static ContextApplication get() {
250            ContextApplication app = ContextApplication.currentApplication();
251            if (app != null)
252                return app;
253            throw new IllegalStateException("no current application found");
254        }
255    
256        /**
257         * Get the {@link ContextApplication} instance associated with the current thread, cast to the desired type,
258         * or throw an exception if there is none.
259         *
260         * @param type expected application type
261         * @return the {@link ContextApplication} associated with the current thread
262         * @throws IllegalStateException if the current {@link ContextApplication} is not found
263         * @see #get()
264         * @throws ClassCastException if the current application is not assignable to {@code type}
265         */
266        public static <A extends ContextApplication> A get(Class<A> type) {
267            A app = ContextApplication.currentApplication(type);
268            if (app != null)
269                return app;
270            throw new IllegalStateException("no current application found");
271        }
272    
273        /**
274         * Get the {@link HttpServletRequest} associated with the current thread.
275         *
276         * <p>
277         * If the current thread is handling a Vaadin web request for an instance of this class,
278         * this method will return the associated {@link HttpServletRequest}.
279         * </p>
280         *
281         * @return the {@link HttpServletRequest} associated with the current thread, or {@code null} if the current thread
282         *  is not servicing a Vaadin web request or the current Vaadin {@link Application} is not an instance of this class
283         */
284        public static HttpServletRequest currentRequest() {
285            return ContextApplication.CURRENT_REQUEST.get();
286        }
287    
288        /**
289         * Get the {@link HttpServletResponse} associated with the current thread.
290         *
291         * <p>
292         * If the current thread is handling a Vaadin web request for an instance of this class,
293         * this method will return the associated {@link HttpServletResponse}.
294         * </p>
295         *
296         * @return the {@link HttpServletResponse} associated with the current thread, or {@code null} if the current thread
297         *  is not servicing a Vaadin web request or the current Vaadin {@link Application} is not an instance of this class
298         */
299        public static HttpServletResponse currentResponse() {
300            return ContextApplication.CURRENT_RESPONSE.get();
301        }
302    }
