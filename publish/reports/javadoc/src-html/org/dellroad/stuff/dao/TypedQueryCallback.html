008    package org.dellroad.stuff.dao;
009    
010    import javax.persistence.EntityManager;
011    import javax.persistence.Query;
012    import javax.persistence.TypedQuery;
013    
014    /**
015     * Helper class for building and executing JPA typed queries.
016     *
017     * @param <T> persistent instance type
018     * @param <R> query return type
019     */
020    public abstract class TypedQueryCallback<T, R> extends QueryCallback<R> {
021    
022        /**
023         * Execute the query.
024         *
025         * <p>
026         * The implementation in {@link TypedQueryCallback} delegates to {@link #executeQuery(TypedQuery)}.
027         */
028        @SuppressWarnings("unchecked")
029        protected final R executeQuery(Query query) {
030            return this.executeQuery((TypedQuery<T>)query);
031        }
032    
033        /**
034         * Build the query as a {@link TypedQuery}.
035         */
036        @Override
037        protected abstract TypedQuery<T> buildQuery(EntityManager entityManager);
038    
039        /**
040         * Execute the query.
041         */
042        protected abstract R executeQuery(TypedQuery<T> query);
043    }
