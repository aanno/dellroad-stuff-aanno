008    package org.dellroad.stuff.validation;
009    
010    /**
011     * Converts values into some other object that has the desired uniqueness.
012     *
013     * @param <T> the type of object being validated
014     * @see Unique @Unique
015     */
016    public interface Uniquifier<T> {
017    
018        /**
019         * Get an object representing the unique value of the given instance.
020         * In other words, if and only two values are "the same", this method should return object(s) that
021         * match (according to {@link Object#equals equals()} and {@link Object#hashCode}).
022         *
023         * <p>
024         * This method may return a value of {@code null} to indicate that {@code value} does not in fact
025         * need to be unique.
026         * </p>
027         *
028         * @param value the value to be uniquified
029         * @return unique representative, or {@code null} to except {@code value} from the uniqueness requirement
030         * @throws ClassCastException if {@code value} is not of a supported type
031         */
032        Object getUniqued(T value);
033    }
