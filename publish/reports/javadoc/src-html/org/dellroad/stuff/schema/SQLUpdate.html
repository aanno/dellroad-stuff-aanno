008    package org.dellroad.stuff.schema;
009    
010    import java.sql.Connection;
011    import java.sql.SQLException;
012    import java.sql.Statement;
013    import java.util.regex.Pattern;
014    
015    /**
016     * Update that executes a provided SQL script.
017     *
018     * <p>
019     * The script may contain multiple SQL statements, in which case individual statements will be parsed out
020     * and executed individually in order. However,this requires proper configuration of a {@link #setSplitPattern split pattern}.
021     */
022    public class SQLUpdate extends AbstractSchemaUpdate {
023    
024        String sqlScript;
025        String splitPattern;
026    
027        /**
028         * Configure the SQL script. This is a required property.
029         *
030         * @param sqlScript script containing one or more SQL statements; if more than one statement is present,
031         *  a {@link #setSplitPattern split pattern} must also be configured
032         * @see #setSplitPattern setSplitPattern()
033         */
034        public void setSQLScript(String sqlScript) {
035            this.sqlScript = sqlScript;
036        }
037    
038        /**
039         * Set the <i>split pattern</i> used to split apart a script containing multiple SQL statements into individual statements.
040         * If this property is not set, the script is assumed to contain a single SQL statement.
041         *
042         * <p>
043         * For example, assuming statements are terminated by semi-colons and each SQL statement starts on a new line,
044         * a reasonable setting would be <code>";[ \t\r]*\n\s+"</code>.
045         *
046         * @throws java.util.regex.PatternSyntaxException if the pattern is not a valid Java regular expression
047         */
048        public void setSplitPattern(String splitPattern) {
049            Pattern.compile(splitPattern);
050            this.splitPattern = splitPattern;
051        }
052    
053        @Override
054        public void apply(Connection c) throws SQLException {
055            if (this.sqlScript == null)
056                throw new IllegalArgumentException("no SQL script configured");
057            String[] statements = this.splitPattern != null ? this.sqlScript.split(this.splitPattern) : new String[] { this.sqlScript };
058            for (int i = 0; i < statements.length; i++) {
059                String sql = statements[i].trim();
060                if (sql.length() == 0)
061                    continue;
062                String sep = sql.indexOf('\n') != -1 ? "\n" : " ";
063                this.log.info("applying SQL update:" + sep + sql);
064                Statement statement = c.createStatement();
065                try {
066                    statement.execute(sql);
067                } catch (SQLException e) {
068                    this.log.error("SQL update failed: " + sql, e);
069                    throw e;
070                } finally {
071                    statement.close();
072                }
073            }
074        }
075    }
