008    package org.dellroad.stuff.schema;
009    
010    import java.sql.Connection;
011    import java.sql.SQLException;
012    
013    /**
014     * Wraps an underlying {@link DatabaseAction} and applies it transactionally using serializable isolation.
015     */
016    public class TransactionalDatabaseAction implements DatabaseAction {
017    
018        private final DatabaseAction action;
019    
020        public TransactionalDatabaseAction(DatabaseAction action) {
021            if (action == null)
022                throw new IllegalArgumentException("null action");
023            this.action = action;
024        }
025    
026        /**
027         * Invoke this instance's wrapped {@link DatabaseAction} within an SQL transaction.
028         */
029        @Override
030        public void apply(Connection c) throws SQLException {
031            boolean success = false;
032            try {
033                c.setTransactionIsolation(Connection.TRANSACTION_SERIALIZABLE);
034                c.setAutoCommit(false);
035                action.apply(c);
036                c.commit();
037                success = true;
038            } finally {
039                if (!success)
040                    c.rollback();
041            }
042        }
043    }
