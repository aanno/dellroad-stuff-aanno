008    package org.dellroad.stuff.schema;
009    
010    import java.sql.Connection;
011    import java.sql.SQLException;
012    
013    import javax.sql.DataSource;
014    
015    /**
016     * Database utility methods.
017     */
018    public final class ActionUtils {
019    
020        private ActionUtils() {
021        }
022    
023        /**
024         * Apply a {@link DatabaseAction} to a {@link DataSource}.
025         *
026         * <p>
027         * This implementation simply invokes {@link DatabaseAction#apply} after acquiring
028         * a {@link Connection} from the given {@link DataSource} (and closing it in any event when done).
029         */
030        public static void applyAction(DataSource dataSource, DatabaseAction action) throws SQLException {
031            Connection c = dataSource.getConnection();
032            try {
033                action.apply(c);
034            } finally {
035                c.close();
036            }
037        }
038    
039        /**
040         * Apply a {@link DatabaseAction} to a {@link DataSource} within a transaction.
041         *
042         * <p>
043         * This implementation does the standard JDBC thing using serializable isolation.
044         *
045         * @see TransactionalDatabaseAction
046         */
047        public static void applyInTransaction(DataSource dataSource, DatabaseAction action) throws SQLException {
048            ActionUtils.applyAction(dataSource, new TransactionalDatabaseAction(action));
049        }
050    }
