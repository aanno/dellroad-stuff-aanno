008    package org.dellroad.stuff.schema;
009    
010    import java.util.HashSet;
011    
012    import org.springframework.beans.factory.BeanFactory;
013    import org.springframework.beans.factory.BeanFactoryAware;
014    import org.springframework.beans.factory.BeanNameAware;
015    import org.springframework.beans.factory.config.ConfigurableBeanFactory;
016    
017    /**
018     * Support superclass for {@link SchemaUpdate}s declared in Spring {@link BeanFactory}s that infer their update
019     * name and required predecessors from their Spring bean attributes.
020     *
021     * <p>
022     * Instances infer their {@link #getName name} and {@link #getRequiredPredecessors required predecessors} from
023     * their Spring bean name (specified by the <code>id</code> XML attribute) and Spring dependencies (specified by
024     * the <code>depends-on</code> XML attribute), respectively.
025     *
026     * <p>
027     * The use of <code>depends-on</code> is an abuse of Spring's dependency notation for convenience. Normally
028     * <code>depends-on</code> refers to bean intialization ordering, whereas this class uses it to refer to schema update ordering.
029     * Schema updates are not normally expected to have any initialization ordering requirements, so this abuse shouldn't matter.
030     * If they do, this class should not be used.
031     *
032     * <p>
033     * Also, the containing {@link BeanFactory} must be a {@link ConfigurableBeanFactory} (normally this is always the case).
034     */
035    public abstract class AbstractSpringSchemaUpdate extends AbstractSchemaUpdate implements BeanNameAware, BeanFactoryAware {
036    
037        @Override
038        public void setBeanName(String beanName) {
039            this.setName(beanName);
040        }
041    
042        @Override
043        public void setBeanFactory(BeanFactory beanFactory) {
044            AbstractSpringSchemaUpdate.setRequiredPredecessorsFromDependencies(this, beanFactory);
045        }
046    
047        /**
048         * Infer required predecessors from Spring dependencies.
049         *
050         * @param update the update in question; it's name must be configured and will be used to query for Spring dependencies
051         * @param beanFactory the bean factory containing the update and the update's depdendencies
052         * @throws IllegalArgumentException if {@code beanFactory} is not a {@link ConfigurableBeanFactory}
053         */
054        public static void setRequiredPredecessorsFromDependencies(AbstractSchemaUpdate update, BeanFactory beanFactory) {
055    
056            // Check factory type
057            if (!(beanFactory instanceof ConfigurableBeanFactory))
058                throw new IllegalArgumentException("BeanFactory is not a ConfigurableBeanFactory: " + beanFactory);
059            ConfigurableBeanFactory configurableBeanFactory = (ConfigurableBeanFactory)beanFactory;
060    
061            // Find required predecessors defined as Spring dependencies
062            String[] predecessorNames = configurableBeanFactory.getDependenciesForBean(update.getName());
063            HashSet<SchemaUpdate> predecessors = new HashSet<SchemaUpdate>(predecessorNames.length);
064            for (String predecessorName : predecessorNames)
065                predecessors.add(configurableBeanFactory.getBean(predecessorName, SchemaUpdate.class));
066            update.setRequiredPredecessors(predecessors);
067        }
068    }
