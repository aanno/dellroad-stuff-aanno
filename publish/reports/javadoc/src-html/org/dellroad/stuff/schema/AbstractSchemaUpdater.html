026    /**
027     * Handles the initialization and schema maintenance of a database.
028     *
029     * <p>
030     * In this class, a <b>database</b> is some stateful object whose structure and/or content may need to change over time.
031     * <b>Updates</b> are uniquely named objects capable of making such changes. Databases are also capable of storing the
032     * names of the already-applied updates.
033     *
034     * <p>
035     * Given a database and a set of current updates, this class will ensure that a database is initialized if necessary
036     * and up-to-date with respect to the updates.
037     *
038     * <p>
039     * The primary method is {@link #initializeAndUpdateDatabase initializeAndUpdateDatabase()}, which will:
040     * <ul>
041     * <li>Initialize an {@linkplain #databaseNeedsInitialization empty} database (if necessary);</li>
042     * <li>Apply any outstanding {@link SchemaUpdate}s as needed, ordered properly according to
043     *      their {@linkplain SchemaUpdate#getRequiredPredecessors predecessor constraints}; and</li>
044     * <li>Keep track of which {@link SchemaUpdate}s have already been applied across restarts.</li>
045     * </ul>
046     * </p>
047     *
048     * @param <D> database type
049     * @param <T> database transaction type
050     */
051    public abstract class AbstractSchemaUpdater<D, T> {
052    
053        protected final Logger log = LoggerFactory.getLogger(this.getClass());
054    
055        private Collection<? extends SchemaUpdate<T>> updates;
056        private boolean ignoreUnrecognizedUpdates;
057    
058        /**
059         * Get the configured updates. This property is required.
060         *
061         * @return configured updates
062         * @see #setUpdates setUpdates()
063         */
064        public Collection<? extends SchemaUpdate<T>> getUpdates() {
065            return this.updates;
066        }
067    
068        /**
069         * Configure the updates.
070         * This should be the set of all updates that may need to be applied to the database.
071         *
072         * <p>
073         * For any given application, ideally this set should be "write only" in the sense that once an update is added to the set
074         * and applied to one or more actual databases, the update and its name should thereafter never change. Otherwise,
075         * it would be possible for different databases to have inconsistent schemas even though the same updates were recorded.
076         *
077         * <p>
078         * Furthermore, if not configured to {@linkplain #setIgnoreUnrecognizedUpdates ignore unrecognized updates already applied}
079         * (the default behavior), then updates must never be removed from this set as the application evolves;
080         * see {@link #setIgnoreUnrecognizedUpdates} for more information on the rationale.
081         *
082         * @param updates all updates; each update must have a unique {@link SchemaUpdate#getName name}.
083         * @see #getUpdates
084         * @see #setIgnoreUnrecognizedUpdates
085         */
086        public void setUpdates(Collection<? extends SchemaUpdate<T>> updates) {
087            this.updates = updates;
088        }
089    
090        /**
091         * Determine whether unrecognized updates are ignored or cause an exception.
092         *
093         * @return true if unrecognized updates should be ignored, false if they should cause an exception to be thrown
094         * @see #setIgnoreUnrecognizedUpdates setIgnoreUnrecognizedUpdates()
095         */
096        public boolean isIgnoreUnrecognizedUpdates() {
097            return this.ignoreUnrecognizedUpdates;
098        }
099    
100        /**
101         * Configure behavior when an unknown update is registered as having already been applied in the database.
102         *
103         * <p>
104         * The default behavior is <code>false</code>, which results in an exception being thrown. This protects against
105         * accidental downgrades (i.e., running older code against a database with a newer schema), which are not supported.
106         * However, this also requires that all updates that might ever possibly have been applied to the database be
107         * present in the set of configured updates.
108         *
109         * <p>
110         * Setting this to <code>true</code> will result in unrecognized updates simply being ignored.
111         * This setting loses the downgrade protection but allows obsolete schema updates to be dropped over time.
112         *
113         * @param ignoreUnrecognizedUpdates whether to ignore unrecognized updates
114         * @see #isIgnoreUnrecognizedUpdates
115         */
116        public void setIgnoreUnrecognizedUpdates(boolean ignoreUnrecognizedUpdates) {
117            this.ignoreUnrecognizedUpdates = ignoreUnrecognizedUpdates;
118        }
119    
120        /**
121         * Perform database schema initialization and updates.
122         *
123         * <p>
124         * This method applies the following logic: if the {@linkplain #databaseNeedsInitialization database needs initialization},
125         * then {@linkplain #initializeDatabase initialize the database} and {@linkplain #recordUpdateApplied record} each update
126         * as having been applied; otherwise, {@linkplain #apply apply} any {@linkplain #getAppliedUpdateNames unapplied updates}
127         * as needed.
128         *
129         * <p>
130         * Note this implies the database initialization must initialize the database to its current, up-to-date state
131         * (with respect to the set of all available updates), not its original, pre-update state.
132         *
133         * <p>
134         * The database initialization step, and each of the update steps, is {@linkplain #applyInTransaction performed within
135         * its own transaction}.
136         *
137         * @param database the database to initialize (if necessary) and update
138         * @throws Exception if an update fails
139         * @throws IllegalStateException if this instance is not configured to {@linkplain #setIgnoreUnrecognizedUpdates ignore
140         * unrecognized updates} and an unrecognized update has already been applied
141         * @throws IllegalArgumentException if two configured updates have the same name
142         * @throws IllegalArgumentException if any configured update has a required predecessor which is not also a configured update
143         *  (i.e., if the updates are not transitively closed under predecessors)
144         */
145        public synchronized void initializeAndUpdateDatabase(final D database) throws Exception {
146    
147            // Log
148            this.log.info("verifying database " + database);
149    
150            // First, initialize if necessary
151            this.applyInTransaction(database, new DatabaseAction<T>() {
152                @Override
153                public void apply(T transaction) throws Exception {
154    
155                    // Already initialized?
156                    if (!AbstractSchemaUpdater.this.databaseNeedsInitialization(transaction)) {
157                        AbstractSchemaUpdater.this.log.debug("detected already-initialized database " + database);
158                        return;
159                    }
160    
161                    // Initialize database
162                    AbstractSchemaUpdater.this.log.info("uninitialized database detected; initializing " + database);
163                    AbstractSchemaUpdater.this.initializeDatabase(transaction);
164    
165                    // Record all schema updates as having already been applied
166                    for (String updateName : AbstractSchemaUpdater.this.getAllUpdateNames())
167                        AbstractSchemaUpdater.this.recordUpdateApplied(transaction, updateName);
168                }
169            });
170    
171            // Next, apply any new updates
172            this.applySchemaUpdates(database);
173    
174            // Done
175            this.log.info("database verification completed for " + database);
176        }
177    
178        /**
179         * Determine if the given schema update name is valid. Valid names are non-empty and
180         * have no leading or trailing whitespace.
181         */
182        public static boolean isValidUpdateName(String updateName) {
183            return updateName.length() > 0 && updateName.trim().length() == updateName.length();
184        }
185    
186        /**
187         * Determine if the database needs initialization.
188         *
189         * <p>
190         * If so, {@link #initializeDatabase} will eventually be invoked.
191         *
192         * @param transaction open transaction
193         * @throws Exception if an error occurs while accessing the database
194         */
195        protected abstract boolean databaseNeedsInitialization(T transaction) throws Exception;
196    
197        /**
198         * Initialize an uninitialized database. This should create and initialize the database schema and content,
199         * including whatever portion of that is used to track schema updates.
200         *
201         * @param transaction open transaction
202         * @throws Exception if an error occurs while accessing the database
203         */
204        protected abstract void initializeDatabase(T transaction) throws Exception;
205    
206        /**
207         * Begin a transaction on the given database.
208         * The transaction will always eventually either be
209         * {@linkplain #commitTransaction committed} or {@linkplain #rollbackTransaction rolled back}.
210         *
211         * @param database database
212         * @return transaction handle
213         * @throws Exception if an error occurs while accessing the database
214         */
215        protected abstract T openTransaction(D database) throws Exception;
216    
217        /**
218         * Commit a previously opened transaction.
219         *
220         * @param transaction open transaction previously returned from {@link #openTransaction openTransaction()}
221         * @throws Exception if an error occurs while accessing the database
222         */
223        protected abstract void commitTransaction(T transaction) throws Exception;
224    
225        /**
226         * Roll back a previously opened transaction.
227         * This method will also be invoked if {@link #commitTransaction commitTransaction()} throws an exception.
228         *
229         * @param transaction open transaction previously returned from {@link #openTransaction openTransaction()}
230         * @throws Exception if an error occurs while accessing the database
231         */
232        protected abstract void rollbackTransaction(T transaction) throws Exception;
233    
234        /**
235         * Determine which updates have already been applied to the database.
236         *
237         * @param transaction open transaction
238         * @throws Exception if an error occurs while accessing the database
239         */
240        protected abstract Set<String> getAppliedUpdateNames(T transaction) throws Exception;
241    
242        /**
243         * Record an update as having been applied to the database.
244         *
245         * @param transaction open transaction
246         * @param name update name
247         * @throws IllegalStateException if the update has already been recorded in the database
248         * @throws Exception if an error occurs while accessing the database
249         */
250        protected abstract void recordUpdateApplied(T transaction, String name) throws Exception;
251    
252        /**
253         * Determine the preferred ordering of two updates that do not have any predecessor constraints
254         * (including implied indirect constraints) between them.
255         *
256         * <p>
257         * The {@link Comparator} returned by the implementation in {@link AbstractSchemaUpdater} simply sorts updates by name.
258         * Subclasses may override if necessary.
259         *
260         * @return a {@link Comparator} that sorts incomparable updates in the order they should be applied
261         */
262        protected Comparator<SchemaUpdate<T>> getOrderingTieBreaker() {
263            return new UpdateByNameComparator();
264        }
265    
266        /**
267         * Generate the update name for one action within a multi-action update.
268         *
269         * <p>
270         * The implementation in {@link AbstractSchemaUpdater} just adds a suffix using {@code index + 1}, padded to
271         * 5 digits, producing names like {@code name-00001}, {@code name-00002}, etc.
272         *
273         * @param update the schema update
274         * @param index the index of the action (zero based)
275         * @see SchemaUpdate#isSingleAction
276         */
277        protected String generateMultiUpdateName(SchemaUpdate<T> update, int index) {
278            return String.format("%s-%05d", update.getName(), index + 1);
279        }
280    
281        /**
282         * Get the names of all updates including multi-action updates.
283         */
284        protected List<String> getAllUpdateNames() throws Exception {
285            ArrayList<SchemaUpdate<T>> updateList = new ArrayList<SchemaUpdate<T>>(this.getUpdates());
286            ArrayList<String> updateNameList = new ArrayList<String>(updateList.size());
287            Collections.sort(updateList, new UpdateByNameComparator());
288            for (SchemaUpdate<T> update : updateList)
289                updateNameList.addAll(this.getUpdateNames(update));
290            return updateNameList;
291        }
292    
293        /**
294         * Execute a database action within an existing transaction.
295         *
296         * <p>
297         * All database operations in {@link AbstractSchemaUpdater} are performed via this method;
298         * subclasses are encouraged to follow this pattern.
299         *
300         * <p>
301         * The implementation in {@link AbstractSchemaUpdater} simply invokes {@link DatabaseAction#apply action.apply()};
302         * subclasses may override if desired.
303         *
304         * @throws Exception if an error occurs while accessing the database
305         */
306        protected void apply(T transaction, DatabaseAction<T> action) throws Exception {
307            action.apply(transaction);
308        }
309    
310        /**
311         * Execute a database action. A new transaction will be created, used, and closed.
312         * Delegates to {@link #apply apply()} for the actual execution of the action.
313         *
314         * <p>
315         * If the action or {@link #commitTransaction commitTransaction()} fails, the transaction
316         * is {@linkplain #rollbackTransaction rolled back}.
317         *
318         * @throws Exception if an error occurs while accessing the database
319         */
320        protected void applyInTransaction(D database, DatabaseAction<T> action) throws Exception {
321            T transaction = this.openTransaction(database);
322            boolean success = false;
323            try {
324                this.apply(transaction, action);
325                this.commitTransaction(transaction);
326                success = true;
327            } finally {
328                if (!success)
329                    this.rollbackTransaction(transaction);
330            }
331        }

333        /**
334         * Apply schema updates to an initialized database.
335         */
336        private void applySchemaUpdates(D database) throws Exception {
337    
338            // Sanity check
339            if (this.getUpdates() == null)
340                throw new IllegalArgumentException("no updates configured");
341            final HashSet<SchemaUpdate<T>> allUpdates = new HashSet<SchemaUpdate<T>>(this.getUpdates());
342    
343            // Create mapping from update name to update; multiple updates will have multiple names
344            TreeMap<String, SchemaUpdate<T>> updateMap = new TreeMap<String, SchemaUpdate<T>>();
345            for (SchemaUpdate<T> update : allUpdates) {
346                for (String updateName : this.getUpdateNames(update)) {
347                    if (!isValidUpdateName(updateName))
348                        throw new IllegalArgumentException("illegal schema update name `" + updateName + "'");
349                    if (updateMap.put(updateName, update) != null)
350                        throw new IllegalArgumentException("duplicate schema update name `" + updateName + "'");
351                }
352            }
353            this.log.debug("these are all known schema updates: " + updateMap.keySet());
354    
355            // Verify updates are transitively closed under predecessor constraints
356            for (SchemaUpdate<T> update : allUpdates) {
357                for (SchemaUpdate<T> predecessor : update.getRequiredPredecessors()) {
358                    if (!allUpdates.contains(predecessor)) {
359                        throw new IllegalArgumentException("schema update `" + update.getName()
360                          + "' has a required predecessor named `" + predecessor.getName() + "' that is not a configured update");
361                    }
362                }
363            }
364    
365            // Sort updates in the order we should to apply them
366            List<SchemaUpdate<T>> updateList = new TopologicalSorter<SchemaUpdate<T>>(allUpdates,
367              new SchemaUpdateEdgeLister<T>(), this.getOrderingTieBreaker()).sortEdgesReversed();
368    
369            // Determine which updates have already been applied
370            final HashSet<String> appliedUpdateNames = new HashSet<String>();
371            this.applyInTransaction(database, new DatabaseAction<T>() {
372                @Override
373                public void apply(T transaction) throws Exception {
374                    appliedUpdateNames.addAll(AbstractSchemaUpdater.this.getAppliedUpdateNames(transaction));
375                }
376            });
377            this.log.debug("these are the already-applied schema updates: " + appliedUpdateNames);
378    
379            // Check whether any unknown updates have been applied
380            TreeSet<String> unknownUpdateNames = new TreeSet<String>(appliedUpdateNames);
381            unknownUpdateNames.removeAll(updateMap.keySet());
382            if (!unknownUpdateNames.isEmpty()) {
383                if (!this.isIgnoreUnrecognizedUpdates()) {
384                    throw new IllegalStateException(unknownUpdateNames.size()
385                      + " unrecognized update(s) have already been applied: " + unknownUpdateNames);
386                }
387                this.log.info("ignoring " + unknownUpdateNames.size()
388                  + " unrecognized update(s) already applied: " + unknownUpdateNames);
389            }
390    
391            // Remove the already-applied updates
392            updateMap.keySet().removeAll(appliedUpdateNames);
393            HashSet<SchemaUpdate<T>> remainingUpdates = new HashSet<SchemaUpdate<T>>(updateMap.values());
394            for (Iterator<SchemaUpdate<T>> i = updateList.iterator(); i.hasNext(); ) {
395                if (!remainingUpdates.contains(i.next()))
396                    i.remove();
397            }
398    
399            // Now are any updates needed?
400            if (updateList.isEmpty()) {
401                this.log.info("no schema updates are required");
402                return;
403            }
404    
405            // Log which updates we're going to apply
406            final LinkedHashSet<String> remainingUpdateNames = new LinkedHashSet<String>(updateMap.size());
407            for (SchemaUpdate<T> update : updateList) {
408                ArrayList<String> updateNames = this.getUpdateNames(update);
409                updateNames.removeAll(appliedUpdateNames);
410                remainingUpdateNames.addAll(updateNames);
411            }
412            this.log.info("applying " + remainingUpdateNames.size() + " schema update(s): " + remainingUpdateNames);
413    
414            // Apply and record each unapplied update
415            for (SchemaUpdate<T> nextUpdate : updateList) {
416                final RecordingUpdateHandler updateHandler = new RecordingUpdateHandler(nextUpdate, remainingUpdateNames);
417                this.applyInTransaction(database, new DatabaseAction<T>() {
418                    @Override
419                    public void apply(T transaction) throws Exception {
420                        updateHandler.process(transaction);
421                    }
422                });
423            }
424        }
425    
426        // Get all update names, expanding multi-updates as necessary
427        private ArrayList<String> getUpdateNames(SchemaUpdate<T> update) throws Exception {
428            final ArrayList<String> names = new ArrayList<String>();
429            UpdateHandler updateHandler = new UpdateHandler(update) {
430                @Override
431                protected void handleSingleUpdate(T transaction, DatabaseAction<T> action) {
432                    names.add(this.update.getName());
433                }
434                @Override
435                protected void handleMultiUpdate(T transaction, DatabaseAction<T> action, int index) {
436                    names.add(AbstractSchemaUpdater.this.generateMultiUpdateName(this.update, index));
437                }
438            };
439            updateHandler.process(null);
440            return names;
441        }
442    
443        // Apply and record an update, all within a single transaction
444        private void applyAndRecordUpdate(T transaction, String name, final DatabaseAction<T> action) throws Exception {
445            if (action != null) {
446                this.log.info("applying schema update `" + name + "'");
447                this.apply(transaction, action);
448            } else
449                this.log.info("recording empty schema update `" + name + "'");
450            this.recordUpdateApplied(transaction, name);
451        }
452    
453        private class RecordingUpdateHandler extends UpdateHandler {
454    
455            private final Set<String> remainingUpdateNames;
456    
457            public RecordingUpdateHandler(SchemaUpdate<T> update, Set<String> remainingUpdateNames) {
458                super(update);
459                this.remainingUpdateNames = remainingUpdateNames;
460            }
461    
462            @Override
463            protected void handleEmptyUpdate(T transaction) throws Exception {
464                assert this.remainingUpdateNames.contains(this.update.getName());
465                AbstractSchemaUpdater.this.applyAndRecordUpdate(transaction, this.update.getName(), null);
466            }
467    
468            @Override
469            protected void handleSingleUpdate(T transaction, DatabaseAction<T> action) throws Exception {
470                assert this.remainingUpdateNames.contains(this.update.getName());
471                AbstractSchemaUpdater.this.applyAndRecordUpdate(transaction, this.update.getName(), action);
472            }
473    
474            @Override
475            protected void handleSingleMultiUpdate(T transaction, final List<? extends DatabaseAction<T>> actions)
476              throws Exception {
477                assert this.remainingUpdateNames.contains(this.update.getName());
478                AbstractSchemaUpdater.this.applyAndRecordUpdate(transaction, this.update.getName(), new DatabaseAction<T>() {
479                    @Override
480                    public void apply(T transaction) throws Exception {
481                        for (DatabaseAction<T> action : actions)
482                            AbstractSchemaUpdater.this.apply(transaction, action);
483                    }
484                });
485            }
486    
487            @Override
488            protected void handleMultiUpdate(T transaction, DatabaseAction<T> action, int index) throws Exception {
489                String updateName = AbstractSchemaUpdater.this.generateMultiUpdateName(this.update, index);
490                if (!this.remainingUpdateNames.contains(updateName))                 // a partially completed multi-update
491                    return;
492                AbstractSchemaUpdater.this.applyAndRecordUpdate(transaction, updateName, action);
493            }
494        }

496        // Adapter class for handling updates of various types
497        private class UpdateHandler {
498    
499            protected final SchemaUpdate<T> update;
500    
501            private final List<? extends DatabaseAction<T>> actions;
502    
503            public UpdateHandler(SchemaUpdate<T> update) {
504                this.update = update;
505                this.actions = update.getDatabaseActions();
506            }
507    
508            public final void process(T transaction) throws Exception {
509                switch (this.actions.size()) {
510                case 0:
511                    this.handleEmptyUpdate(transaction);
512                    break;
513                case 1:
514                    this.handleSingleUpdate(transaction, this.actions.get(0));
515                    break;
516                default:
517                    if (update.isSingleAction()) {
518                        this.handleSingleMultiUpdate(transaction, actions);
519                        break;
520                    } else {
521                        int index = 0;
522                        for (DatabaseAction<T> action : this.actions)
523                            this.handleMultiUpdate(transaction, action, index++);
524                    }
525                    break;
526                }
527            }
528    
529            protected void handleEmptyUpdate(T transaction) throws Exception {
530                this.handleSingleUpdate(transaction, null);
531            }
532    
533            protected void handleSingleUpdate(T transaction, DatabaseAction<T> action) throws Exception {
534            }
535    
536            protected void handleSingleMultiUpdate(T transaction, List<? extends DatabaseAction<T>> actions) throws Exception {
537                this.handleSingleUpdate(transaction, null);
538            }
539    
540            protected void handleMultiUpdate(T transaction, DatabaseAction<T> action, int index) throws Exception {
541            }
542        }

544        // Sorts updates by name
545        private class UpdateByNameComparator implements Comparator<SchemaUpdate<T>> {
546    
547            @Override
548            public int compare(SchemaUpdate<T> update1, SchemaUpdate<T> update2) {
549                return update1.getName().compareTo(update2.getName());
550            }
551        }
