008    package org.dellroad.stuff.schema;
009    
010    import java.util.List;
011    
012    /**
013     * Spring-enabled {@link SchemaUpdate} supporting updates via SQL statements.
014     *
015     * <p>
016     * Instances can be created succintly using the <code>&lt;dellroad-stuff:sql-update&gt;</code> custom XML element, which works
017     * just like <code>&lt;dellroad-stuff:sql&gt;</code> except that it wraps the resulting {@link SQLDatabaseAction}
018     * as a delegate inside an instance of this class.
019     *
020     * <p>
021     * For example:
022     * <blockquote><pre>
023     *  &lt;beans xmlns="http://www.springframework.org/schema/beans"
024     *    <b>xmlns:dellroad-stuff="http://dellroad-stuff.googlecode.com/schema/dellroad-stuff"</b>
025     *    xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
026     *    xsi:schemaLocation="
027     *      http://www.springframework.org/schema/beans
028     *        http://www.springframework.org/schema/beans/spring-beans-3.0.xsd
029     *      <b>http://dellroad-stuff.googlecode.com/schema/dellroad-stuff
030     *        http://dellroad-stuff.googlecode.com/svn/wiki/schemas/dellroad-stuff-1.0.xsd</b>"&gt;
031     *
032     *      &lt;!-- Schema update to add the 'phone' column to the 'User' table --&gt;
033     *      <b>&lt;dellroad-stuff:sql-update id="addPhone"&gt;ALTER TABLE User ADD phone VARCHAR(64)&lt;/dellroad-stuff:sql-update&gt;</b>
034     *
035     *      &lt;!-- Schema update to run some complicated external SQL script --&gt;
036     *      <b>&lt;dellroad-stuff:sql-update id="majorChanges" depends-on="addPhone" resource="classpath:majorChanges.sql"/&gt;</b>
037     *
038     *      &lt;!-- more beans... --&gt;
039     *
040     *  &lt;/beans&gt;
041     * </pre></blockquote>
042     * </p>
043     *
044     * <p>
045     * A multi-statement SQL script is normally treated as a set of individual updates. For example:
046     * <blockquote><pre>
047     *      <b>&lt;dellroad-stuff:sql-update id="renameColumn"&gt;
048     *          ALTER TABLE User ADD newName VARCHAR(64);
049     *          ALTER TABLE User SET newName = oldName;
050     *          ALTER TABLE User DROP oldName;
051     *      &lt;/dellroad-stuff:sql-update&gt;</b>
052     * </pre></blockquote>
053     * This will create three separate update beans named <code>renameColumn-1</code>, <code>renameColumn-2</code>, and
054     * <code>renameColumn-3</code>. You can disable this behavior by adding the attribute <code>single-action="true"</code>,
055     * in which case all three of the statements will be executed together in the same transaction and recorded under the name
056     * <code>renameColumn</code>; this means that they must all complete successfully or you could end up with a partially
057     * completed update.
058     * </p>
059     *
060     * <p>
061     * Note that if the nested SQL script only contains one SQL statement, any <code>single-action</code> attribute is
062     * ignored and the bean's given name (e.g., <code>renameColumn</code>) is always used as the name of the single update.
063     * </p>
064     */
065    public class SpringSQLSchemaUpdate extends AbstractSpringSchemaUpdate {
066    
067        private SQLDatabaseAction action;
068    
069        @Override
070        public void afterPropertiesSet() throws Exception {
071            super.afterPropertiesSet();
072            if (this.action == null)
073                throw new Exception("no DatabaseAction configured");
074        }
075    
076        /**
077         * Configure the {@link SQLDatabaseAction}. This is a required property.
078         *
079         * @see DatabaseAction
080         */
081        public void setSQLDatabaseAction(SQLDatabaseAction action) {
082            this.action = action;
083        }
084    
085        @Override
086        public List<DatabaseAction> getDatabaseActions() {
087            return this.action.split();
088        }
089    }
