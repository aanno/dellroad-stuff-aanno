008    package org.dellroad.stuff.schema;
009    
010    import org.springframework.beans.factory.BeanFactory;
011    import org.springframework.beans.factory.BeanFactoryAware;
012    import org.springframework.beans.factory.BeanNameAware;
013    import org.springframework.beans.factory.InitializingBean;
014    
015    /**
016     * Provides the functionality of both {@link SQLUpdate} and {@link AbstractSpringSchemaUpdate}.
017     */
018    public class SpringSQLUpdate extends SQLUpdate implements BeanNameAware, BeanFactoryAware, InitializingBean {
019    
020        @Override
021        public void afterPropertiesSet() throws Exception {
022            if (this.sqlScript == null)
023                throw new Exception("no SQL script configured");
024        }
025    
026        @Override
027        public void setBeanName(String beanName) {
028            this.setName(beanName);
029        }
030    
031        @Override
032        public void setBeanFactory(BeanFactory beanFactory) {
033            AbstractSpringSchemaUpdate.setRequiredPredecessorsFromDependencies(this, beanFactory);
034        }
035    }
