008    package org.dellroad.stuff.schema;
009    
010    import java.util.HashSet;
011    import java.util.Set;
012    
013    import org.slf4j.Logger;
014    import org.slf4j.LoggerFactory;
015    
016    /**
017     * Support superclass for {@link SchemaUpdate}s implementations providing standard Java bean property
018     * implementations of {@link #getName} and {@link #getRequiredPredecessors}.
019     */
020    public abstract class AbstractSchemaUpdate implements ModifiableSchemaUpdate {
021    
022        protected final Logger log = LoggerFactory.getLogger(getClass());
023    
024        private String name;
025        private Set<SchemaUpdate> requiredPredecessors = new HashSet<SchemaUpdate>();
026    
027        @Override
028        public String getName() {
029            return this.name;
030        }
031        public void setName(String name) {
032            this.name = name;
033        }
034    
035        @Override
036        public Set<SchemaUpdate> getRequiredPredecessors() {
037            return this.requiredPredecessors;
038        }
039        public void setRequiredPredecessors(Set<SchemaUpdate> requiredPredecessors) {
040            this.requiredPredecessors = requiredPredecessors;
041        }
042    }
