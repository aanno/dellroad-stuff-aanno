008    package org.dellroad.stuff.schema;
009    
010    import java.sql.Connection;
011    import java.sql.SQLException;
012    import java.util.Comparator;
013    import java.util.HashMap;
014    import java.util.Map;
015    
016    import javax.sql.DataSource;
017    
018    import org.springframework.beans.factory.BeanFactory;
019    import org.springframework.beans.factory.BeanFactoryAware;
020    import org.springframework.beans.factory.InitializingBean;
021    import org.springframework.beans.factory.ListableBeanFactory;
022    import org.springframework.jdbc.BadSqlGrammarException;
023    import org.springframework.jdbc.core.ConnectionCallback;
024    import org.springframework.jdbc.core.JdbcTemplate;
025    
026    /**
027     * {@link SchemaUpdater} optimized for use with Spring. It is required that this updater and all of its
028     * schema updates are defined in the same {@link ListableBeanFactory}.
029     * <ul>
030     * <li>{@link #applyModification(DataSource, SchemaModification)} is overridden to use Spring's {@link JdbcTemplate}
031     *  so Spring {@link org.springframework.dao.DataAccessException}s are thrown.</li>
032     * <li>{@link #databaseNeedsInitialization} is overridden to catch exceptions more precisely to filter out
033     *  false positives.</li>
034     * <li>{@link #getOrderingTieBreaker} is overridden to break ties by ordering updates in the same order
035     *  as they are defined in the bean factory.</li>
036     * <li>This class implements {@link InitializingBean} and verifies all required properties are set.</li>
037     * </ul>
038     *
039     * <p>
040     * It is required that this updater and all of its schema updates are defined in the same {@link ListableBeanFactory}.
041     */
042    public class SpringSchemaUpdater extends SchemaUpdater implements BeanFactoryAware, InitializingBean {
043    
044        private ListableBeanFactory beanFactory;
045    
046        @Override
047        public void afterPropertiesSet() throws Exception {
048            if (this.getUpdates() == null)
049                throw new Exception("no updates configured");
050            if (this.getDatabaseInitialization() == null)
051                throw new Exception("no database initialization configured");
052            if (this.getUpdateTableInitialization() == null)
053                throw new Exception("no update table initialization configured");
054        }
055    
056        @Override
057        public void setBeanFactory(BeanFactory beanFactory) {
058            if (!(beanFactory instanceof ListableBeanFactory))
059                throw new IllegalArgumentException("containing BeanFactory is not a ListableBeanFactory: " + beanFactory);
060            this.beanFactory = (ListableBeanFactory)beanFactory;
061        }
062    
063        /**
064         * Determine if the database needs initialization.
065         *
066         * <p>
067         * The implementation in {@link SpringSchemaUpdater} invokes <code>SELECT COUNT(*) FROM <i>UPDATETABLE</i></code>
068         * and checks for a {@link BadSqlGrammarException}.
069         */
070        @Override
071        public boolean databaseNeedsInitialization(DataSource dataSource) throws SQLException {
072            try {
073                long numUpdates = new JdbcTemplate(dataSource).queryForLong("SELECT COUNT(*) FROM " + this.getUpdateTableName());
074                this.log.info("detected already initialized database, with " + numUpdates + " update(s) already applied");
075                return false;
076            } catch (BadSqlGrammarException e) {
077                log.warn("detected uninitialized database: update table `" + this.getUpdateTableName() + "' not found: " + e);
078                return true;
079            }
080        }
081    
082        /**
083         * Apply a schema modification to a {@link DataSource}.
084         *
085         * <p>
086         * The implementation in {@link SpringSchemaUpdater} uses {@link JdbcTemplate} to apply the modification so that
087         * Spring {@link org.springframework.dao.DataAccessException}s are thrown in case of errors.
088         */
089        @Override
090        protected void applyModification(DataSource dataSource, final SchemaModification schemaMod) throws SQLException {
091            new JdbcTemplate(dataSource).execute(new ConnectionCallback<Void>() {
092                @Override
093                public Void doInConnection(Connection c) throws SQLException {
094                    SpringSchemaUpdater.this.applyModification(c, schemaMod);
095                    return null;
096                }
097            });
098        }
099    
100        /**
101         * Get the preferred ordering of two updates that do not have any predecessor constraints
102         * (including implied indirect constraints) between them.
103         *
104         * <p>
105         * The {@link Comparator} returned by the implementation in {@link SchemaUpdater} simply sorts updates by name.
106         */
107        @Override
108        protected Comparator<SchemaUpdate> getOrderingTieBreaker() {
109            String[] beanNames = this.beanFactory.getBeanDefinitionNames();
110            final Map<String, Integer> sort = new HashMap<String, Integer>(beanNames.length);
111            for (int i = 0; i < beanNames.length; i++)
112                sort.put(beanNames[i], i);
113            return new Comparator<SchemaUpdate>() {
114                @Override
115                public int compare(SchemaUpdate update1, SchemaUpdate update2) {
116                    String[] names = new String[] { update1.getName(), update2.getName() };
117                    Integer[] indexes = new Integer[] { sort.get(update1.getName()), sort.get(update2.getName()) };
118                    for (int i = 0; i < 2; i++) {
119                        if (indexes[i] == null) {
120                            throw new IllegalArgumentException("failed to find update `" + names[i]
121                              + "' in bean factory " + SpringSchemaUpdater.this.beanFactory);
122                        }
123                    }
124                    return indexes[0] - indexes[1];
125                };
126            };
127        }
128    }
