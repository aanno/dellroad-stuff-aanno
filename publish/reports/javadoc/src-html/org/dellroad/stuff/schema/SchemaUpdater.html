033    /**
034     * Database schema updater that manages initializing and updates to a database schema.
035     *
036     * <p>
037     * The primary method is {@link #initializeAndUpdateDatabase initializeAndUpdateDatabase()}, which will:
038     * <ul>
039     * <li> Automatically initialize an empty database when necessary;</li>
040     * <li> Automatically apply configured {@link SchemaUpdate}s as needed, ordered properly according to
041     *      their predecessor constraints; and</li>
042     * <li> Automatically keep track of which {@link SchemaUpdate}s have already been applied across restarts.</li>
043     * </ul>
044     * </p>
045     *
046     * <p>
047     * Required properties are the {@linkplain #setDatabaseInitialization database initialization},
048     * {@linkplain #setUpdateTableInitialization update table initialization}, and the {@linkplain #setUpdates updates} themselves.
049     * </p>
050     *
051     * <p>
052     * Applied updates are recorded in a special <i>update table</i>, which contains two columns: one for the unique
053     * {@linkplain SchemaUpdate#getName update name} and one for a timestamp. The update table and column names
054     * are configurable via {@link #setUpdateTableName setUpdateTableName()},
055     * {@link #setUpdateTableNameColumn setUpdateTableNameColumn()}, and {@link #setUpdateTableTimeColumn setUpdateTableTimeColumn()}.
056     * </p>
057     *
058     * <p>
059     * By default, this class detects a completely uninitialized database by the absence of the update table itself
060     * in the schema (see {@link #databaseNeedsInitialization databaseNeedsInitialization()}).
061     * When an uninitialized database is encountered, the configured {@linkplain #setDatabaseInitialization database initialization}
062     * and {@linkplain #setUpdateTableInitialization update table initialization} actions are applied first to initialize
063     * the database schema.
064     * </p>
065     */
066    public class SchemaUpdater {

243         * Configure the schema updates.
244         * This should be the set of all updates that may need to be applied to the database.
245         *
246         * <p>
247         * For any given application, ideally this set should be "write only" in the sense that once an update is added to the set

249         * it would be possible for different databases to have inconsistent schemas even though the same updates were recorded.
250         *
251         * <p>
252         * Furthermore, if not configured to {@linkplain #setIgnoreUnrecognizedUpdates ignore unrecognized updates already applied}
253         * (the default behavior), then updates must never be removed from this set as the application evolves;
254         * see {@link #setIgnoreUnrecognizedUpdates} for more information on the rationale.
255         *
256         * @param updates all schema updates; each update must have a unique {@link SchemaUpdate#getName name}.
257         * @see #getUpdates
258         * @see #setIgnoreUnrecognizedUpdates
259         */

261            this.updates = updates;
262        }

274        /**
275         * Configure behavior when an unknown update is registered as having already been applied in the database.
276         *
277         * <p>
278         * The default behavior is <code>false</code>, which results in an exception being thrown. This protects against
279         * accidental downgrades (i.e., running older code against a database with a newer schema), which are not supported.
280         * However, this also requires that all updates that might ever possibly have been applied to the database be
281         * present in the set of configured updates.
282         *
283         * <p>
284         * Setting this to <code>true</code> will result in unrecognized updates simply being ignored.
285         * This setting loses the downgrade protection but allows obsolete schema updates to be dropped over time.
286         *
287         * @param ignoreUnrecognizedUpdates whether to ignore unrecognized updates
288         * @see #isIgnoreUnrecognizedUpdates
289         */
290        public void setIgnoreUnrecognizedUpdates(boolean ignoreUnrecognizedUpdates) {
291            this.ignoreUnrecognizedUpdates = ignoreUnrecognizedUpdates;
292        }

294        /**
295         * Perform database schema initialization and updates.
296         *
297         * <p>
298         * This method applies the following logic: if the {@linkplain #databaseNeedsInitialization database needs initialization},
299         * then {@linkplain #initializeDatabase initialize} the database; then, {@linkplain #applySchemaUpdates apply schema updates}
300         * as needed. The database initialization step and each schema update is applied within its own transaction.
301         *
302         * @throws SQLException if an update fails
303         * @throws IllegalStateException if the database needs initialization and either the
304         *  {@linkplain #setDatabaseInitialization database initialization} or
305         *  the {@linkplain #setUpdateTableInitialization update table initialization} has not been configured
306         * @throws IllegalStateException if this instance is not configured to {@linkplain #setIgnoreUnrecognizedUpdates ignore
307         * unrecognized updates} and an unrecognized update is registered in the update table as having already been applied
308         * @throws IllegalArgumentException if two configured updates have the same name
309         * @throws IllegalArgumentException if any configured update has a required predecessor which is not also a configured update
310         *  (i.e., if the updates are not transitively closed under predecessors)
311         */
312        public void initializeAndUpdateDatabase(DataSource dataSource) throws SQLException {
313            log.info("verifying database");
314            if (this.databaseNeedsInitialization(dataSource))
315                this.initializeDatabase(dataSource);
316            this.applySchemaUpdates(dataSource);
317            log.info("database verification complete");
318        }

320        /**
321         * Determine if the database needs initialization.
322         *
323         * <p>
324         * The implementation in {@link SchemaUpdater} simply invokes <code>SELECT COUNT(*) FROM <i>UPDATETABLE</i></code>
325         * and checks for success or failure. Subclasses may wish to override with a more discriminating implementation.
326         * This implementation does not distguish {@link SQLException}s caused by a missing update table from other causes.
327         */
328        public boolean databaseNeedsInitialization(DataSource dataSource) throws SQLException {
329            Connection c = dataSource.getConnection();
330            try {
331                Statement s = c.createStatement();
332                try {
333                    ResultSet resultSet = s.executeQuery("SELECT COUNT(*) FROM " + this.getUpdateTableName());
334                    long numUpdates = resultSet.getLong(1);
335                    log.info("detected already initialized database, with " + numUpdates + " update(s) already applied");
336                    return false;
337                } finally {
338                    s.close();
339                }
340            } catch (SQLException e) {
341                log.warn("detected uninitialized database: update table `" + this.getUpdateTableName() + "' not found: " + e);
342                return true;
343            } finally {
344                c.close();
345            }
346        }

369        /**
370         * Record an update as having been applied.
371         *
372         * <p>
373         * The implementation in {@link SchemaUpdater} does the standard JDBC thing using an INSERT statement
374         * into the update table.
375         * </p>
376         *
377         * @param c SQL connection
378         * @param name update name
379         */
380        protected void recordUpdateApplied(Connection c, String name) throws SQLException {
381            PreparedStatement s = c.prepareStatement("INSERT INTO " + this.getUpdateTableName()
382              + " (" + this.getUpdateTableNameColumn() + ", " + this.getUpdateTableTimeColumn() + ") VALUES (?, ?)");
383            try {
384                s.setString(1, name);
385                s.setDate(2, new java.sql.Date(new Date().getTime()));
386                int rows = s.executeUpdate();
387                if (rows != 1)
388                    throw new IllegalStateException("expected 1 row returned from INSERT but got zero for update `" + name + "'");
389            } finally {
390                s.close();
391            }
392        }

394        /**
395         * Determine which updates have already been applied.
396         *
397         * <p>
398         * The implementation in {@link SchemaUpdater} does the standard JDBC thing using a SELECT statement
399         * from the update table.
400         */
401        protected Set<String> getAppliedUpdateNames(Connection c) throws SQLException {
402            Statement s = c.createStatement();
403            try {
404                HashSet<String> updateNames = new HashSet<String>();
405                ResultSet resultSet = s.executeQuery("SELECT " + this.getUpdateTableNameColumn()
406                  + " FROM " + this.getUpdateTableName());
407                while (resultSet.next())
408                    updateNames.add(resultSet.getString(1));
409                return updateNames;
410            } finally {
411                s.close();
412            }
413        }

415        /**
416         * Get the preferred ordering of two updates that do not have any predecessor constraints
417         * (including implied indirect constraints) between them.
418         *
419         * <p>
420         * The {@link Comparator} returned by the implementation in {@link SchemaUpdater} simply sorts updates by name.
421         */
422        protected Comparator<SchemaUpdate> getOrderingTieBreaker() {
423            return new Comparator<SchemaUpdate>() {
424                @Override
425                public int compare(SchemaUpdate update1, SchemaUpdate update2) {
426                    return update1.getName().compareTo(update2.getName());
427                }
428            };
429        }

431        /**
432         * Generate the update name for one action within a multi-action update.
433         *
434         * <p>
435         * The implementation in {@link SchemaUpdater} just adds a suffix using {@code index + 1}, padded to
436         * 5 digits, producing names like {@code name-00001}, {@code name-00002}, etc.
437         *
438         * @param update the schema update
439         * @param index the index of the action (zero based)
440         */
441        protected String generateMultiUpdateName(SchemaUpdate update, int index) {
442            return String.format("%s-%05d", update.getName(), index + 1);
443        }

445        // Initialize the database
446        private void initializeDatabase(final DataSource dataSource) throws SQLException {
447    
448            // Sanity check
449            if (this.getDatabaseInitialization() == null)
450                throw new IllegalArgumentException("database needs initialization but no database initialization is configured");
451            if (this.getUpdateTableInitialization() == null)
452                throw new IllegalArgumentException("database needs initialization but no update table initialization is configured");
453    
454            // Initialize schema within a transaction
455            this.applyInTransaction(dataSource, new DatabaseAction() {
456    
457                @Override
458                public void apply(final Connection c) throws SQLException {
459    
460                    // Initialize application schema
461                    SchemaUpdater.this.log.info("intializing database schema");
462                    SchemaUpdater.this.applyAction(dataSource, SchemaUpdater.this.getDatabaseInitialization());
463    
464                    // Initialize update table
465                    SchemaUpdater.this.log.info("intializing update table");
466                    SchemaUpdater.this.applyAction(dataSource, SchemaUpdater.this.getUpdateTableInitialization());
467    
468                    // Record all schema updates as having already been applied
469                    for (SchemaUpdate update : SchemaUpdater.this.getUpdates()) {
470                        for (String name : SchemaUpdater.this.getUpdateNames(update))
471                            SchemaUpdater.this.recordUpdateApplied(c, name);
472                    }
473                }
474            });
475        }
476    
477        // Apply schema updates
478        private void applySchemaUpdates(final DataSource dataSource) throws SQLException {
479    
480            // Sanity check
481            final Collection<SchemaUpdate> allUpdates = this.getUpdates();
482            if (allUpdates == null)
483                throw new IllegalArgumentException("schema updates set is not configured");
484    
485            // Create mapping from update name to update; multiple updates will have multiple names
486            TreeMap<String, SchemaUpdate> updateMap = new TreeMap<String, SchemaUpdate>();
487            for (SchemaUpdate update : allUpdates) {
488                for (String updateName : this.getUpdateNames(update)) {
489                    if (updateMap.put(updateName, update) != null)
490                        throw new IllegalArgumentException("duplicate schema update name `" + updateName + "'");
491                }
492            }
493            this.log.debug("these are all known schema updates: " + updateMap.keySet());
494    
495            // Verify updates are transitively closed under predecessor constraints
496            for (SchemaUpdate update : allUpdates) {
497                for (SchemaUpdate predecessor : update.getRequiredPredecessors()) {
498                    if (!this.getUpdates().contains(predecessor)) {
499                        throw new IllegalArgumentException("schema update `" + update.getName()
500                          + "' has a required predecessor `" + predecessor.getName() + "' that is not a configured update");
501                    }
502                }
503            }
504    
505            // Sort updates in the order we should to apply them
506            List<SchemaUpdate> updateList = new TopologicalSorter<SchemaUpdate>(allUpdates,
507              new SchemaUpdateEdgeLister(), this.getOrderingTieBreaker()).sortEdgesReversed();
508    
509            // Determine which updates have already been applied
510            Set<String> appliedUpdates;
511            Connection c = dataSource.getConnection();
512            try {
513                appliedUpdates = this.getAppliedUpdateNames(c);
514            } finally {
515                c.close();
516            }
517            this.log.debug("these are the already-applied schema updates: " + appliedUpdates);
518    
519            // Check whether any unknown updates have been applied
520            TreeSet<String> unknown = new TreeSet<String>(appliedUpdates);
521            unknown.removeAll(updateMap.keySet());
522            if (!unknown.isEmpty()) {
523                if (!this.isIgnoreUnrecognizedUpdates())
524                    throw new IllegalStateException(unknown.size() + " unrecognized update(s) have already been applied: " + unknown);
525                this.log.info("ignoring " + unknown.size() + " unrecognized update(s) already applied: " + unknown);
526            }
527    
528            // Remove the already-applied updates
529            updateMap.keySet().removeAll(appliedUpdates);
530            HashSet<SchemaUpdate> remainingUpdates = new HashSet<SchemaUpdate>(updateMap.values());
531            for (Iterator<SchemaUpdate> i = updateList.iterator(); i.hasNext(); ) {
532                if (!remainingUpdates.contains(i.next()))
533                    i.remove();
534            }
535    
536            // Now are any updates needed?
537            if (updateList.isEmpty()) {
538                this.log.info("no schema updates are required");
539                return;
540            }
541    
542            // Log which updates we're going to apply
543            final LinkedHashSet<String> remainingUpdateNames = new LinkedHashSet<String>(updateMap.size());
544            for (SchemaUpdate update : updateList) {
545                ArrayList<String> updateNames = this.getUpdateNames(update);
546                updateNames.removeAll(appliedUpdates);
547                remainingUpdateNames.addAll(updateNames);
548            }
549            this.log.info("applying " + remainingUpdateNames.size() + " schema update(s): " + remainingUpdateNames);
550    
551            // Apply updates
552            for (SchemaUpdate nextUpdate : updateList) {
553                UpdateHandler updateHandler = new UpdateHandler(nextUpdate) {
554                    @Override
555                    protected void handleEmptyUpdate(final SchemaUpdate update) throws SQLException {
556                        assert remainingUpdateNames.contains(update.getName());
557                        SchemaUpdater.this.applyInTransaction(dataSource, new DatabaseAction() {
558                            @Override
559                            public void apply(Connection c) throws SQLException {
560                                SchemaUpdater.this.log.info("recording empty schema update `" + update.getName() + "'");
561                                SchemaUpdater.this.recordUpdateApplied(c, update.getName());
562                            }
563                        });
564                    }
565                    @Override
566                    protected void handleSingleUpdate(final SchemaUpdate update, final DatabaseAction action) throws SQLException {
567                        assert remainingUpdateNames.contains(update.getName());
568                        SchemaUpdater.this.applyInTransaction(dataSource, new DatabaseAction() {
569                            @Override
570                            public void apply(Connection c) throws SQLException {
571                                SchemaUpdater.this.applyAndRecordUpdate(dataSource, update.getName(), action);
572                            }
573                        });
574                    }
575                    @Override
576                    protected void handleSingleMultiUpdate(final SchemaUpdate update, final List<DatabaseAction> actions)
577                      throws SQLException {
578                        assert remainingUpdateNames.contains(update.getName());
579                        SchemaUpdater.this.applyInTransaction(dataSource, new DatabaseAction() {
580                            @Override
581                            public void apply(Connection c) throws SQLException {
582                                SchemaUpdater.this.applyAndRecordUpdate(dataSource, update.getName(), new MultiAction(actions));
583                            }
584                        });
585                    }
586                    @Override
587                    protected void handleMultiUpdate(SchemaUpdate update, final DatabaseAction action, int index) throws SQLException {
588                        final String updateName = SchemaUpdater.this.generateMultiUpdateName(update, index);
589                        if (!remainingUpdateNames.contains(updateName))                 // a partially completed multi-update
590                            return;
591                        SchemaUpdater.this.applyInTransaction(dataSource, new DatabaseAction() {
592                            @Override
593                            public void apply(Connection c) throws SQLException {
594                                SchemaUpdater.this.applyAndRecordUpdate(dataSource, updateName, action);
595                            }
596                        });
597                    }
598                };
599                updateHandler.process();
600            }
601        }
602    
603        private ArrayList<String> getUpdateNames(SchemaUpdate update) throws SQLException {
604            final ArrayList<String> names = new ArrayList<String>();
605            UpdateHandler updateHandler = new UpdateHandler(update) {
606                @Override
607                protected void handleSingleUpdate(SchemaUpdate update, DatabaseAction action) {
608                    names.add(update.getName());
609                }
610                @Override
611                protected void handleMultiUpdate(SchemaUpdate update, DatabaseAction action, int index) {
612                    names.add(SchemaUpdater.this.generateMultiUpdateName(update, index));
613                }
614            };
615            updateHandler.process();
616            return names;
617        }
618    
619        private void applyAndRecordUpdate(DataSource dataSource, final String name, final DatabaseAction action) throws SQLException {
620            this.applyInTransaction(dataSource, new DatabaseAction() {
621                @Override
622                public void apply(Connection c) throws SQLException {
623                    SchemaUpdater.this.log.info("applying schema update `" + name + "'");
624                    action.apply(c);
625                    SchemaUpdater.this.recordUpdateApplied(c, name);
626                }
627            });
628        }

630        private static class UpdateHandler {
631    
632            private final SchemaUpdate update;
633            private final List<DatabaseAction> actions;
634    
635            public UpdateHandler(SchemaUpdate update) {
636                this.update = update;
637                this.actions = update.getDatabaseActions();
638            }
639    
640            public final void process() throws SQLException {
641                switch (this.actions.size()) {
642                case 0:
643                    this.handleEmptyUpdate(this.update);
644                    break;
645                case 1:
646                    this.handleSingleUpdate(this.update, this.actions.get(0));
647                    break;
648                default:
649                    if (update.isSingleAction()) {
650                        this.handleSingleMultiUpdate(this.update, actions);
651                        break;
652                    } else {
653                        int index = 0;
654                        for (DatabaseAction action : this.actions)
655                            this.handleMultiUpdate(this.update, action, index++);
656                    }
657                    break;
658                }
659            }
660    
661            protected void handleEmptyUpdate(SchemaUpdate update) throws SQLException {
662                this.handleSingleUpdate(update, null);
663            }
664    
665            protected void handleSingleUpdate(SchemaUpdate update, DatabaseAction action) throws SQLException {
666            }
667    
668            protected void handleSingleMultiUpdate(SchemaUpdate update, List<DatabaseAction> actions) throws SQLException {
669                this.handleSingleUpdate(update, null);
670            }
671    
672            protected void handleMultiUpdate(SchemaUpdate update, DatabaseAction action, int index) throws SQLException {
673            }
674        }

676        private static final class MultiAction implements DatabaseAction {
677    
678            private final List<DatabaseAction> actions;
679    
680            public MultiAction(List<DatabaseAction> actions) {
681                this.actions = actions;
682            }
683    
684            @Override
685            public void apply(Connection c) throws SQLException {
686                for (DatabaseAction action : this.actions)
687                    action.apply(c);
688            }
689        }
