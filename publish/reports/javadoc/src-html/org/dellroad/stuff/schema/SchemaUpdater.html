028    /**
029     * Database schema updater that manages initializing and updates to a database schema, automatically
030     * initialzing the database schema when empty and ordering and applying updates according to their
031     * predecessor constraints, keeping track of which updates have already been applied.
032     *
033     * <p>
034     * Updates that have been applied are recorded in a special <i>update table</i>, which contains
035     * two columns for the unique update name and application timestamp. The table and column names
036     * are configurable via {@link #setUpdateTableName setUpdateTableName()}, {@link #setUpdateNameColumn setUpdateNameColumn()},
037     * and {@link #setUpdateTimeColumn setUpdateTimeColumn()}.
038     *
039     * <p>
040     * This class detects a completely uninitialized database by the absence of the update table.
041     * With an uninitialized database, the supplied {@link #setDatabaseInitialization database initialization}
042     * and {@link #setUpdateTableInitialization update table initialization} updates are applied first.
043     */
044    public class SchemaUpdater {
045    
046        /**
047         * Default nefault name of the table that tracks schema updates, <code>{@value}</code>.
048         */
049        public static final String DEFAULT_UPDATE_TABLE_NAME = "SchemaUpdate";
050    
051        /**
052         * Default name of the column in the updates table holding the unique update name, <code>{@value}</code>.
053         */
054        public static final String DEFAULT_UPDATE_NAME_COLUMN = "updateName";
055    
056        /**
057         * Default name of the column in the updates table holding the update's time applied, <code>{@value}</code>.
058         */
059        public static final String DEFAULT_UPDATE_TIME_COLUMN = "updateTime";
060

065        private String updateTimeColumn = DEFAULT_UPDATE_TIME_COLUMN;

068        private SchemaModification databaseInitialization;

176         * <p>
177         * This script is expected to initialize the database (i.e., creating all the tables) and
178         * is assumed to be "up to date" with respect to the configured schema updates, i.e.,
179         * when the script completes we assum all updates have already been (implicitly) applied.
180         *
181         * <p>
182         * Note this script is <i>not</i> expected to create the update table that tracks schema updates;
183         * that function is handled by the {@link #setUpdateTableInitialization update table initialization}.
184         *

201        /**
202         * Configure the schema updates.
203         * This should be the set of all updates that may need to be applied to the database.
204         *
205         * <p>
206         * For any given application, ideally this set should be "write only" in the sense that once an update is added to the set,
207         * the update and its name should never change. Furthermore, if not configured to {@link #setIgnoreUnrecognizedUpdates ignore
208         * unrecognized updates already applied} (the default behavior), then updates must never be removed from this set
209         * as the application evolves.
210         *
211         * @param updates all schema updates
212         * @see #getUpdates
213         */
214        public void setUpdates(Set<SchemaUpdate> updates) {
215            this.updates = updates;
216        }

219         * Determine whether unrecognized updates are ignored or cause an exception.

221         * @return true if unrecognized updates should be ignored, false if they should cause an exception to be thrown
222         * @see #setIgnoreUnrecognizedUpdates setIgnoreUnrecognizedUpdates()
223         */
224        public boolean isIgnoreUnrecognizedUpdates() {
225            return this.ignoreUnrecognizedUpdates;
226        }

228        /**
229         * Configure behavior when an unknown update is found and registered as having already been applied in the database.
230         *
231         * <p>
232         * The default behavior is <code>false</code>, which results in an exception being thrown to protect against
233         * accidental downgrades (i.e., running older code against a database with a newer schema), which are not supported.
234         *
235         * <p>
236         * Setting this to <code>true</code> will result in unrecognized updates simply being ignored.
237         * This setting loses the downgrade protection but allows ancient (obsolete) schema updates to drop off the list.
238         *
239         * @param ignoreUnrecognizedUpdates whether to ignore unrecognized updates
240         * @see #isIgnoreUnrecognizedUpdates
241         */
242        public void setIgnoreUnrecognizedUpdates(boolean ignoreUnrecognizedUpdates) {
243            this.ignoreUnrecognizedUpdates = ignoreUnrecognizedUpdates;
244        }

246        /**
247         * Perform database schema initialization and updates.
248         *
249         * @throws SQLException if an update fails
250         * @throws IllegalStateException if the database needs initialization and either the
251         *  {@link #setDatabaseInitialization database initialization} or
252         *  the {@link #setUpdateTableInitialization update table initialization} has not been configured
253         * @throws IllegalStateException if this instance is not configured to {@link #setIgnoreUnrecognizedUpdates ignore
254         * unrecognized updates} and an unrecognized update is registered in the update table as having already been applied
255         * @throws IllegalArgumentException if two configured updates have the same name
256         * @throws IllegalArgumentException if any configured update has a required predecessor which is not also a configured update
257         *  (i.e., if the updates are not transitively closed under predecessors)
258         */
259        public void initializeAndUpdateDatabase(DataSource dataSource) throws SQLException {
260            log.info("verifying database");
261            if (this.databaseNeedsInitialization(dataSource))
262                this.initializeDatabase(dataSource);
263            this.applySchemaUpdates(dataSource);
264            log.info("database verification complete");
265        }
266    
267        /**
268         * Determine if the database needs initialization.
269         *
270         * <p>
271         * The implementation in {@link SchemaUpdater} simply invokes <code>SELECT COUNT(*) FROM <i>UPDATETABLE</i></code>
272         * and checks for success or failure. Subclasses may wish to override with a more discriminating implementation
273         * that distguishes {@link SQLException}s caused by a missing update table from other causes.
274         */
275        public boolean databaseNeedsInitialization(DataSource dataSource) throws SQLException {
276            Connection c = dataSource.getConnection();
277            try {
278                Statement s = c.createStatement();
279                try {
280                    ResultSet resultSet = s.executeQuery("SELECT COUNT(*) FROM " + this.getUpdateTableName());
281                    long numUpdates = resultSet.getLong(1);
282                    log.info("detected already initialized database, with " + numUpdates + " update(s) already applied");
283                    return false;
284                } finally {
285                    s.close();
286                }
287            } catch (SQLException e) {
288                log.warn("detected uninitialized database: update table `" + this.getUpdateTableName() + "' not found: " + e);
289                return true;
290            } finally {
291                c.close();
292            }
293        }

295        /**
296         * Apply a schema modification to a {@link DataSource}.
297         *
298         * <p>
299         * The implementation in {@link SchemaUpdater} simply invokes {@link #applyModification(Connection, SchemaModification)}
300         * after acquiring a {@link Connection} from the given {@link DataSource} (and closing it in any event when done).
301         */
302        protected void applyModification(DataSource dataSource, SchemaModification schemaMod) throws SQLException {
303            Connection c = dataSource.getConnection();
304            try {
305                this.applyModification(c, schemaMod);
306            } finally {
307                c.close();
308            }
309        }

312         * Apply a schema modification to a {@link Connection}.
313         *
314         * <p>
315         * The implementation in {@link SchemaUpdater} simply invokes {@link SchemaModification#apply}.
316         */
317        protected void applyModification(Connection c, SchemaModification schemaMod) throws SQLException {
318            schemaMod.apply(c);
319        }
320    
321        /**
322         * Apply a schema modification within a transaction.
323         *
324         * <p>
325         * The implementation in {@link SchemaUpdater} does the standard JDBC thing using serializable isolation.
326         */
327        protected void applyInTransaction(DataSource dataSource, SchemaModification schemaMod) throws SQLException {
328            Connection c = dataSource.getConnection();
329            boolean success = false;
330            try {
331                c.setTransactionIsolation(Connection.TRANSACTION_SERIALIZABLE);
332                c.setAutoCommit(false);
333                this.applyModification(c, schemaMod);
334                c.commit();
335                success = true;
336            } finally {
337                if (!success)
338                    c.rollback();
339                c.close();
340            }
341        }

343        /**
344         * Record an update as having been applied.
345         *
346         * <p>
347         * The implementation in {@link SchemaUpdater} does the standard JDBC thing using an INSERT statement
348         * into the update table.
349         */
350        protected void recordUpdateApplied(Connection c, SchemaUpdate update) throws SQLException {
351            PreparedStatement s = c.prepareStatement("INSERT INTO " + this.getUpdateTableName()
352              + " (" + this.getUpdateNameColumn() + ", " + this.getUpdateTimeColumn() + ") VALUES (?, ?)");
353            try {
354                s.setString(1, update.getName());
355                s.setDate(2, new java.sql.Date(new Date().getTime()));
356                int rows = s.executeUpdate();
357                if (rows != 1) {
358                    throw new IllegalStateException("expected 1 row returned from INSERT but got zero for update `"
359                      + update.getName() + "'");
360                }
361            } finally {
362                s.close();
363            }
364        }
365    
366        /**
367         * Determine which updates have already been applied.
368         *
369         * <p>
370         * The implementation in {@link SchemaUpdater} does the standard JDBC thing using a SELECT statement
371         * from the update table.
372         */
373        protected HashSet<String> getAppliedUpdateNames(Connection c) throws SQLException {
374            Statement s = c.createStatement();
375            try {
376                HashSet<String> updateNames = new HashSet<String>();
377                ResultSet resultSet = s.executeQuery("SELECT " + this.getUpdateNameColumn() + " FROM " + this.getUpdateTableName());
378                while (resultSet.next())
379                    updateNames.add(resultSet.getString(1));
380                return updateNames;
381            } finally {
382                s.close();
383            }
384        }
385    
386        /**
387         * Get the preferred ordering of two updates that do not have any predecessor constraints
388         * (including implied indirect constraints) between them.
389         *
390         * <p>
391         * The {@link Comparator} returned by the implementation in {@link SchemaUpdater} simply sorts updates by name.
392         */
393        protected Comparator<SchemaUpdate> getOrderingTieBreaker() {
394            return new Comparator<SchemaUpdate>() {
395                @Override
396                public int compare(SchemaUpdate update1, SchemaUpdate update2) {
397                    return update1.getName().compareTo(update2.getName());
398                }
399            };
400        }

402        // Initialize the database
403        private void initializeDatabase(final DataSource dataSource) throws SQLException {
404    
405            // Sanity check
406            if (this.getDatabaseInitialization() == null)
407                throw new IllegalArgumentException("database needs initialization but no database initialization is configured");
408            if (this.getUpdateTableInitialization() == null)
409                throw new IllegalArgumentException("database needs initialization but no update table initialization is configured");
410    
411            // Initialize schema within a transaction
412            this.applyInTransaction(dataSource, new SchemaModification() {
413    
414                @Override
415                public void apply(Connection c) throws SQLException {
416    
417                    // Initialize application schema
418                    SchemaUpdater.this.log.info("intializing database schema");
419                    SchemaUpdater.this.applyModification(dataSource, SchemaUpdater.this.getDatabaseInitialization());
420    
421                    // Initialize update table
422                    SchemaUpdater.this.log.info("intializing update table");
423                    SchemaUpdater.this.applyModification(dataSource, SchemaUpdater.this.getUpdateTableInitialization());
424    
425                    // Record all schema updates as having already been applied
426                    for (SchemaUpdate update : SchemaUpdater.this.getUpdates())
427                        SchemaUpdater.this.recordUpdateApplied(c, update);
428                }
429            });
430        }

432        // Apply schema updates
433        private void applySchemaUpdates(DataSource dataSource) throws SQLException {
434    
435            // Sanity check
436            if (this.getUpdates() == null)
437                throw new IllegalArgumentException("schema updates set is not configured");
438    
439            // Verify updates are transitively closed under predecessor constraints
440            for (SchemaUpdate update : this.getUpdates()) {
441                for (SchemaUpdate predecessor : update.getRequiredPredecessors()) {
442                    if (!this.getUpdates().contains(predecessor)) {
443                        throw new IllegalArgumentException("schema update `" + update.getName()
444                          + "' has a required predecessor `" + predecessor.getName() + "' that is not a configured update");
445                    }
446                }
447            }
448    
449            // Create map of updates by name
450            HashMap<String, SchemaUpdate> updateMap = new HashMap<String, SchemaUpdate>(this.getUpdates().size());
451            for (SchemaUpdate update : this.getUpdates()) {
452                if (updateMap.put(update.getName(), update) != null)
453                    throw new IllegalArgumentException("duplicate schema update name `" + update.getName() + "'");
454            }
455    
456            // Determine which updates have already been applied
457            Set<String> appliedUpdates;
458            Connection c = dataSource.getConnection();
459            try {
460                appliedUpdates = this.getAppliedUpdateNames(c);
461            } finally {
462                c.close();
463            }
464    
465            // Check whether any unknown updates have been applied; remove already-applied updates from our update map
466            HashSet<String> unknown = new HashSet<String>();
467            for (String updateName : appliedUpdates) {
468                if (updateMap.remove(updateName) == null)
469                    unknown.add(updateName);
470            }
471            if (!unknown.isEmpty()) {
472                if (!this.isIgnoreUnrecognizedUpdates())
473                    throw new IllegalStateException(unknown.size() + " unrecognized update(s) have already been applied: " + unknown);
474                this.log.info("ignoring " + unknown.size() + " unrecognized update(s) already applied");
475            }
476    
477            // Any updates needed?
478            if (updateMap.isEmpty()) {
479                this.log.info("no schema updates are required");
480                return;
481            }
482    
483            // Log result
484            if (updates.isEmpty())
485                this.log.info("no schema updates are required");
486            else
487                this.log.info("applying required schema updates: " + updateMap.keySet());
488    
489            // Sort updates in the order we want to apply them
490            List<SchemaUpdate> updateList = new TopologicalSorter<SchemaUpdate>(
491              updateMap.values(), new SchemaUpdateEdgeLister(), this.getOrderingTieBreaker()).sortEdgesReversed();
492    
493            // Apply updates
494            for (SchemaUpdate update0 : updateList) {
495                final SchemaUpdate update = update0;
496                this.applyInTransaction(dataSource, new SchemaModification() {
497    
498                    @Override
499                    public void apply(Connection c) throws SQLException {
500                        SchemaUpdater.this.log.info("applying schema update `" + update.getName() + "'");
501                        SchemaUpdater.this.applyModification(c, update);
502                        SchemaUpdater.this.recordUpdateApplied(c, update);
503                    }
504                });
505            }
506        }
