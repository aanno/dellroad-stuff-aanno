008    package org.dellroad.stuff.schema;
009    
010    import java.util.Set;
011    
012    
013    /**
014     * A database update. Instances typically perform changes to the database schema, though non-schema updates
015     * that just manipulate the data in the database are perfectly valid as well.
016     *
017     * <p>
018     * Each {@link SchemaUpdate} has a unique name among all updates ever applied to a single database, and zero
019     * or more required predecessors, which are other dependent updates that must be applied first.
020     *
021     * <p>
022     * Once an update has been applied to a database, it must not be changed; otherwise, inconsistencies
023     * can exist between databases that were updated using the old version vs. databases that were
024     * updated using the new version. If an update has been applied but had the wrong behavior, instead of
025     * changing the update, it's better to create a new update that depends on the first as a predecessor
026     * and corrects the mistake.
027     *
028     * <p>
029     * If you do have to change an update after it has been applied, then those databases that have
030     * already seen the previous version of the update must be manually corrected so they are in the
031     * same state that the new version of the update would have left them.
032     */
033    public interface SchemaUpdate extends SchemaModification {
034    
035        /**
036         * Get the unique name of this update. This name must be unique among all updates ever applied to the database
037         * and must never change once this update has been applied to any database.
038         *
039         * @return the name of this update; must not be the empty string
040         */
041        String getName();
042    
043        /**
044         * Get the the other updates (if any) that must be applied <b>before</b> this update may be applied.
045         *
046         * @return set of zero or more other updates
047         * @see #getName
048         */
049        Set<SchemaUpdate> getRequiredPredecessors();
050    }
