008    package org.dellroad.stuff.schema;
009    
010    import java.sql.Connection;
011    import java.sql.SQLException;
012    import java.sql.Statement;
013    import java.util.ArrayList;
014    import java.util.List;
015    import java.util.regex.Pattern;
016    
017    import org.slf4j.Logger;
018    import org.slf4j.LoggerFactory;
019    
020    /**
021     * A {@link DatabaseAction} that executes a configured SQL script.
022     *
023     * <p>
024     * The script may contain multiple SQL statements, in which case individual statements will be parsed out and
025     * executed individually in order. However, this requires proper configuration of the {@link #setSplitPattern split pattern}.
026     *
027     * <p>
028     * When using Spring, beans of this type can be created succintly using the <code>&lt;dellroad-stuff:sql&gt;</code> custom
029     * XML element. The split pattern may be configured via the {@code split-pattern} attribute, and the SQL script is specified
030     * either directly via inline text or using the {@code resource} attribute. In the latter case, the character encoding can
031     * specified via the {@code charset} attribute (default is {@code "UTF-8"}).
032     *
033     * <p>
034     * For example:
035     * <blockquote><pre>
036     *  &lt;beans xmlns="http://www.springframework.org/schema/beans"
037     *    <b>xmlns:dellroad-stuff="http://dellroad-stuff.googlecode.com/schema/dellroad-stuff"</b>
038     *    xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
039     *    xsi:schemaLocation="
040     *      http://www.springframework.org/schema/beans
041     *        http://www.springframework.org/schema/beans/spring-beans-3.0.xsd
042     *      <b>http://dellroad-stuff.googlecode.com/schema/dellroad-stuff
043     *        http://dellroad-stuff.googlecode.com/svn/wiki/schemas/dellroad-stuff-1.0.xsd</b>"&gt;
044     *
045     *      &lt;!-- SQL action that clears the audit log --&gt;
046     *      <b>&lt;dellroad-stuff:sql&gt;DELETE * FROM AUDIT_LOG&lt;/dellroad-stuff:sql&gt;</b>
047     *
048     *      &lt;!-- A more complicated external SQL script --&gt;
049     *      <b>&lt;dellroad-stuff:sql resource="classpath:reset.sql" split-pattern=";\n"/&gt;</b>
050     *
051     *      &lt;!-- other beans... --&gt;
052     *
053     *  &lt;/beans&gt;
054     * </pre></blockquote>
055     */
056    public class SQLDatabaseAction implements DatabaseAction {
057    
058        /**
059         * The default split pattern: <code>{@value}</code>.
060         */
061        public static final String DEFAULT_SPLIT_PATTERN = ";[ \\t\\r]*\\n\\s*";
062    
063        protected final Logger log = LoggerFactory.getLogger(getClass());
064    
065        private String sqlScript;
066        private String splitPattern = DEFAULT_SPLIT_PATTERN;
067    
068        /**
069         * Configure the SQL script. This is a required property.
070         *
071         * <p>
072         * For scripts in external resources, consider {@link org.dellroad.stuff.spring.ResourceReaderFactoryBean}
073         * or use the <code>&lt;dellroad-stuff:sql&gt;</code> element.
074         *
075         * @param sqlScript script containing one or more SQL statements; if more than one statement is present,
076         *  a {@link #setSplitPattern split pattern} must also be configured
077         * @see #setSplitPattern setSplitPattern()
078         */
079        public void setSQLScript(String sqlScript) {
080            this.sqlScript = sqlScript;
081        }
082    
083        /**
084         * Set the <i>split pattern</i> used to split apart a script containing multiple SQL statements into individual statements.
085         *
086         * <p>
087         * The default value for this property is <code>";[ \t\r]*\n\s*"</code>, which should handle cases where
088         * SQL statements are terminated by semi-colons and each SQL statement starts on a new line.
089         *
090         * <p>
091         * If this is set to {@code null}, or the script does not contain any instances of the regular expression,
092         * the script is assumed to contain a single SQL statement. SQL statements are whitespace-trimmed and any
093         * "statements" that consist entirely of whitespace are ignored.
094         *
095         * @throws java.util.regex.PatternSyntaxException if the pattern is not a valid Java regular expression
096         */
097        public void setSplitPattern(String splitPattern) {
098            if (splitPattern != null)
099                Pattern.compile(splitPattern);
100            this.splitPattern = splitPattern;
101        }
102    
103        @Override
104        public void apply(Connection c) throws SQLException {
105            for (String sql : this.splitSQL())
106                this.apply(c, sql);
107        }
108    
109        /**
110         * Apply a single SQL statement.
111         */
112        protected void apply(Connection c, String sql) throws SQLException {
113            String sep = sql.indexOf('\n') != -1 ? "\n" : " ";
114            this.log.info("executing SQL statement:" + sep + sql);
115            Statement statement = c.createStatement();
116            try {
117                statement.execute(sql);
118            } catch (SQLException e) {
119                this.log.error("SQL statement failed: " + sql, e);
120                throw e;
121            } finally {
122                statement.close();
123            }
124        }
125    
126        /**
127         * Split the SQL script into indivicual statements and return them as {@link DatabaseAction}s.
128         */
129        public List<DatabaseAction> split() {
130            ArrayList<DatabaseAction> list = new ArrayList<DatabaseAction>();
131            for (String sql0 : this.splitSQL()) {
132                final String sql = sql0;
133                list.add(new DatabaseAction() {
134                    @Override
135                    public void apply(Connection c) throws SQLException {
136                        SQLDatabaseAction.this.apply(c, sql);
137                    }
138                });
139            }
140            return list;
141        }
142    
143        /**
144         * Split the {@linkplain #setSQLScript configured SQL script} into individual SQL statements
145         * using the configured {@linkplain #setSplitPattern split pattern}.
146         *
147         * @return an array of individual SQL statements
148         * @throws IllegalArgumentException if no SQL script is configured
149         */
150        public String[] splitSQL() {
151            if (this.sqlScript == null)
152                throw new IllegalArgumentException("no SQL script configured");
153            String[] sqls = this.splitPattern != null ? this.sqlScript.split(this.splitPattern) : new String[] { this.sqlScript };
154            ArrayList<String> list = new ArrayList<String>(sqls.length);
155            for (String sql : sqls) {
156                sql = sql.trim();
157                if (sql.length() == 0)
158                    continue;
159                list.add(sql);
160            }
161            return list.toArray(new String[list.size()]);
162        }
163    }
