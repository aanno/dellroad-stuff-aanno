008    package org.dellroad.stuff.schema;
009    
010    import java.sql.Connection;
011    import java.sql.SQLException;
012    import java.sql.Statement;
013    import java.util.regex.Pattern;
014    
015    import org.slf4j.Logger;
016    import org.slf4j.LoggerFactory;
017    
018    /**
019     * A {@link DatabaseAction} that executes a configured SQL script.
020     *
021     * <p>
022     * The script may contain multiple SQL statements, in which case individual statements will be parsed out and
023     * executed individually in order. However, this requires proper configuration of the {@link #setSplitPattern split pattern}.
024     *
025     * <p>
026     * When using Spring, beans of this type can be created succintly using the <code>&lt;dellroad-stuff:sql&gt;</code> custom
027     * XML element. The split pattern may be configured via the {@code split-pattern} attribute, and the SQL script is specified
028     * either directly via inline text or using the {@code resource} attribute. In the latter case, the character encoding can
029     * specified via the {@code charset} attribute (default is {@code "UTF-8"}).
030     *
031     * <p>
032     * For example:
033     * <blockquote><pre>
034     *  &lt;beans xmlns="http://www.springframework.org/schema/beans"
035     *    <b>xmlns:dellroad-stuff="http://dellroad-stuff.googlecode.com/schema/dellroad-stuff"</b>
036     *    xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
037     *    xsi:schemaLocation="
038     *      http://www.springframework.org/schema/beans
039     *        http://www.springframework.org/schema/beans/spring-beans-3.0.xsd
040     *      <b>http://dellroad-stuff.googlecode.com/schema/dellroad-stuff
041     *        http://dellroad-stuff.googlecode.com/svn/wiki/schemas/dellroad-stuff-1.0.xsd</b>"&gt;
042     *
043     *      &lt;!-- Schema update to add the 'phone' column to the 'User' table --&gt;
044     *      &lt;bean id="addPhoneColumnUpdate" class="org.dellroad.stuff.schema.SpringDelegatingSchemaUpdate"&gt;
045     *          &lt;property name="databaseAction"&gt;
046     *              <b>&lt;dellroad-stuff:sql&gt;ALTER TABLE User ADD phone VARCHAR(64)&lt;/dellroad-stuff:sql&gt;</b>
047     *          &lt;/property&gt;
048     *      &lt;/bean&gt;
049     *
050     *      &lt;!-- Schema update to run some complicated external SQL script --&gt;
051     *      &lt;bean id="majorChanges" depends-on="addPhoneColumnUpdate"
052     *        class="org.dellroad.stuff.schema.SpringDelegatingSchemaUpdate"&gt;
053     *          &lt;property name="databaseAction"&gt;
054     *              <b>&lt;dellroad-stuff:sql resource="classpath:majorSurgery.sql" split-pattern=";\n"/&gt;</b>
055     *          &lt;/property&gt;
056     *      &lt;/bean&gt;
057     *
058     *      &lt;!-- more beans... --&gt;
059     *
060     *  &lt;/beans&gt;
061     * </pre></blockquote>
062     */
063    public class SQLDatabaseAction implements DatabaseAction {
064    
065        /**
066         * The default split pattern: <code>{@value}</code>.
067         */
068        public static final String DEFAULT_SPLIT_PATTERN = ";[ \\t\\r]*\\n\\s*";
069    
070        protected final Logger log = LoggerFactory.getLogger(getClass());
071    
072        private String sqlScript;
073        private String splitPattern = DEFAULT_SPLIT_PATTERN;
074    
075        /**
076         * Configure the SQL script. This is a required property.
077         *
078         * <p>
079         * For scripts in external resources, consider {@link org.dellroad.stuff.spring.ResourceReaderFactoryBean}
080         * or use the <code>&lt;dellroad-stuff:sql&gt;</code> element.
081         *
082         * @param sqlScript script containing one or more SQL statements; if more than one statement is present,
083         *  a {@link #setSplitPattern split pattern} must also be configured
084         * @see #setSplitPattern setSplitPattern()
085         */
086        public void setSQLScript(String sqlScript) {
087            this.sqlScript = sqlScript;
088        }
089    
090        /**
091         * Set the <i>split pattern</i> used to split apart a script containing multiple SQL statements into individual statements.
092         *
093         * <p>
094         * The default value for this property is <code>";[ \t\r]*\n\s*"</code>, which should handle cases where
095         * SQL statements are terminated by semi-colons and each SQL statement starts on a new line.
096         *
097         * <p>
098         * If this is set to {@code null}, or the script does not contain any instances of the regular expression,
099         * the script is assumed to contain a single SQL statement. SQL statements are whitespace-trimmed and any
100         * "statements" that consist entirely of whitespace are ignored.
101         *
102         * @throws java.util.regex.PatternSyntaxException if the pattern is not a valid Java regular expression
103         */
104        public void setSplitPattern(String splitPattern) {
105            if (splitPattern != null)
106                Pattern.compile(splitPattern);
107            this.splitPattern = splitPattern;
108        }
109    
110        @Override
111        public void apply(Connection c) throws SQLException {
112            if (this.sqlScript == null)
113                throw new IllegalArgumentException("no SQL script configured");
114            String[] statements = this.splitPattern != null ? this.sqlScript.split(this.splitPattern) : new String[] { this.sqlScript };
115            for (int i = 0; i < statements.length; i++) {
116                String sql = statements[i].trim();
117                if (sql.length() == 0)
118                    continue;
119                String sep = sql.indexOf('\n') != -1 ? "\n" : " ";
120                this.log.info("executing SQL statement:" + sep + sql);
121                Statement statement = c.createStatement();
122                try {
123                    statement.execute(sql);
124                } catch (SQLException e) {
125                    this.log.error("SQL statement failed: " + sql, e);
126                    throw e;
127                } finally {
128                    statement.close();
129                }
130            }
131        }
132    }
