008    package org.dellroad.stuff.schema;
009    
010    import java.sql.Connection;
011    import java.sql.SQLException;
012    
013    /**
014     * Spring-enabled version of {@link DelegatingSchemaUpdate}.
015     *
016     * <p>
017     * This class provides the combined functionality of {@link DelegatingSchemaUpdate} and {@link AbstractSpringSchemaUpdate}.
018     *
019     * <p>
020     * When using Spring and setting the {@link #setDatabaseAction database action} to a {@link SQLDatabaseAction}, beans
021     * can be created succintly using the <code>&lt;dellroad-stuff:sql-update&gt;</code> custom XML element, which works
022     * just like <code>&lt;dellroad-stuff:sql&gt;</code> except that it wraps the resulting {@link SQLDatabaseAction}
023     * inside an instance of this class as the delegate.
024     *
025     * <p>
026     * For example:
027     * <blockquote><pre>
028     *  &lt;beans xmlns="http://www.springframework.org/schema/beans"
029     *    <b>xmlns:dellroad-stuff="http://dellroad-stuff.googlecode.com/schema/dellroad-stuff"</b>
030     *    xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
031     *    xsi:schemaLocation="
032     *      http://www.springframework.org/schema/beans
033     *        http://www.springframework.org/schema/beans/spring-beans-3.0.xsd
034     *      <b>http://dellroad-stuff.googlecode.com/schema/dellroad-stuff
035     *        http://dellroad-stuff.googlecode.com/svn/wiki/schemas/dellroad-stuff-1.0.xsd</b>"&gt;
036     *
037     *      &lt;!-- Schema update to add the 'phone' column to the 'User' table --&gt;
038     *      <b>&lt;dellroad-stuff:sql-update id="addPhone"&gt;ALTER TABLE User ADD phone VARCHAR(64)&lt;/dellroad-stuff:sql-update&gt;</b>
039     *
040     *      &lt;!-- Schema update to run some complicated external SQL script --&gt;
041     *      <b>&lt;dellroad-stuff:sql-update id="majorChanges" depends-on="addPhone" resource="classpath:majorChanges.sql"/&gt;</b>
042     *
043     *      &lt;!-- more beans... --&gt;
044     *
045     *  &lt;/beans&gt;
046     * </pre></blockquote>
047     */
048    public class SpringDelegatingSchemaUpdate extends AbstractSpringSchemaUpdate {
049    
050        private DatabaseAction action;
051    
052        @Override
053        public void afterPropertiesSet() throws Exception {
054            super.afterPropertiesSet();
055            if (this.action == null)
056                throw new Exception("no DatabaseAction configured");
057        }
058    
059        /**
060         * Configure the {@link DatabaseAction}. This is a required property.
061         *
062         * @see DatabaseAction
063         */
064        public void setDatabaseAction(DatabaseAction action) {
065            this.action = action;
066        }
067    
068        @Override
069        public void apply(Connection c) throws SQLException {
070            this.action.apply(c);
071        }
072    }
