008    package org.dellroad.stuff.main;
009    
010    import java.io.EOFException;
011    import java.io.File;
012    import java.util.ArrayList;
013    
014    import org.gnu.readline.Readline;
015    import org.gnu.readline.ReadlineLibrary;
016    import org.slf4j.Logger;
017    import org.slf4j.LoggerFactory;
018    
019    /**
020     * Support superclass for command line classes.
021     */
022    public abstract class MainClass {
023    
024        protected final Logger log = LoggerFactory.getLogger(getClass());
025    
026        protected MainClass() {
027        }
028    
029        /**
030         * Subclass main implementation. This method is free to throw exceptions; these will
031         * be displayed on standard error and converted into non-zero exit values.
032         *
033         * @return exit value
034         */
035        public abstract int run(String[] args) throws Exception;
036    
037        /**
038         * Enter command loop. Commands are read using GNU libreadline and handed off
039         * to {@link #handleCommand} for processing.
040         */
041        protected void commandLoop(String appName, String prompt) throws Exception {
042    
043            // Setup readline
044            try {
045                Readline.load(ReadlineLibrary.GnuReadline);
046            } catch (UnsatisfiedLinkError e) {
047                // ignore
048            }
049            Readline.initReadline(appName);
050            Readline.setThrowExceptionOnUnsupportedMethod(false);
051    
052            // Read init file(s)
053            String[] files = new String[] { System.getenv("INPUTRC"),
054              new File(new File(System.getProperty("user.home")), ".inputrc").getAbsolutePath(), "/etc/.inputrc" };
055            for (String file : files) {
056                try {
057                    Readline.readInitFile(file);
058                } catch (Exception e) {
059                    // ignore
060                }
061            }
062    
063            // Read history file
064            String historyFile = new File(new File(System.getProperty("user.home")), "." + appName + "_history").getAbsolutePath();
065            try {
066                Readline.readHistoryFile(historyFile);
067            } catch (Exception e) {
068                // ignore
069            }
070    
071            // Main loop
072            try {
073                while (true) {
074    
075                    // Read next line
076                    String line;
077                    try {
078                        line = Readline.readline(prompt);
079                    } catch (EOFException e) {
080                        break;
081                    }
082    
083                    // No input?
084                    if (line == null)
085                        continue;
086    
087                    // Update history
088                    Readline.addToHistory(line);
089    
090                    // Execute line
091                    if (!this.handleCommand(line))
092                        break;
093                }
094            } finally {
095    
096                // Save history
097                try {
098                    Readline.writeHistoryFile(historyFile);
099                } catch (Exception e) {
100                    // ignore
101                }
102    
103                // Clean up
104                Readline.cleanup();
105            }
106        }
107    
108        /**
109         * Callback used by {@link #commandLoop}.
110         *
111         * <p>
112         * The implementation in {@link MainClass} just returns {@code false}.
113         *
114         * @return true to continue reading the next command, false to exit
115         */
116        protected boolean handleCommand(String line) throws Exception {
117            return false;
118        }
119    
120        /**
121         * Display the usage message to standard error.
122         */
123        protected abstract void usageMessage();
124    
125        /**
126         * Print the usage message and exit with exit value 1.
127         */
128        protected void usageError() {
129            usageMessage();
130            System.exit(1);
131        }
132    
133        /**
134         * Emit an error message an exit with exit value 1.
135         */
136        protected final void errout(String message) {
137            System.err.println(getClass().getSimpleName() + ": " + message);
138            System.exit(1);
139        }
140    
141        /**
142         * Parse command line flags of the form {@code -Dname=value} and set the corresponding system properties.
143         * Parsing stops at the first argument not starting with a dash (or {@code --}).
144         *
145         * @return command line with all the property-setting flags removed
146         */
147        protected String[] parsePropertyFlags(String[] args) {
148            ArrayList<String> list = new ArrayList<String>(args.length);
149            boolean done = false;
150            for (String arg : args) {
151                if (done) {
152                    list.add(arg);
153                    continue;
154                }
155                if (arg.equals("--") || arg.length() == 0 || arg.charAt(0) != '-') {
156                    list.add(arg);
157                    done = true;
158                    continue;
159                }
160                if (arg.startsWith("-D")) {
161                    int eq = arg.indexOf('=');
162                    if (eq < 3)
163                        usageError();
164                    System.setProperty(arg.substring(2, eq), arg.substring(eq + 1));
165                    continue;
166                }
167                list.add(arg);
168            }
169            return list.toArray(new String[list.size()]);
170        }
171    
172        /**
173         * Invokes {@link #run}, catching any exceptions thrown and exiting with a non-zero
174         * value if and only if an exception was caught.
175         * <p/>
176         * <p>
177         * The concrete class' {@code main()} method should invoke this method.
178         * </p>
179         */
180        protected void doMain(String[] args) {
181            int exitValue = 1;
182            try {
183                exitValue = run(args);
184            } catch (Throwable t) {
185                t.printStackTrace(System.err);
186            } finally {
187                System.exit(exitValue);
188            }
189        }
190    }
