008    package org.dellroad.stuff.main;
009    
010    import java.io.EOFException;
011    import java.io.File;
012    
013    import org.gnu.readline.Readline;
014    import org.gnu.readline.ReadlineLibrary;
015    import org.slf4j.Logger;
016    import org.slf4j.LoggerFactory;
017    
018    /**
019     * Support superclass for command line classes.
020     */
021    public abstract class MainClass {
022    
023        protected final Logger log = LoggerFactory.getLogger(getClass());
024    
025        protected MainClass() {
026        }
027    
028        /**
029         * Subclass main implementation. This method is free to throw exceptions; these will
030         * be displayed on standard error and converted into non-zero exit values.
031         *
032         * @return exit value
033         */
034        protected abstract int run(String[] args) throws Exception;
035    
036        /**
037         * Enter command loop. Commands are read using GNU libreadline and handed off
038         * to {@link #handleCommand} for processing.
039         */
040        protected void commandLoop(String appName, String prompt) throws Exception {
041    
042            // Setup readline
043            try {
044                Readline.load(ReadlineLibrary.GnuReadline);
045            } catch (UnsatisfiedLinkError e) {
046                // ignore
047            }
048            Readline.initReadline(appName);
049            Readline.setThrowExceptionOnUnsupportedMethod(false);
050    
051            // Read init file(s)
052            String[] files = new String[] { System.getenv("INPUTRC"),
053              new File(new File(System.getProperty("user.home")), ".inputrc").getAbsolutePath(), "/etc/.inputrc" };
054            for (String file : files) {
055                try {
056                    Readline.readInitFile(file);
057                } catch (Exception e) {
058                    // ignore
059                }
060            }
061    
062            // Read history file
063            String historyFile = new File(new File(System.getProperty("user.home")), "." + appName + "_history").getAbsolutePath();
064            try {
065                Readline.readHistoryFile(historyFile);
066            } catch (Exception e) {
067                // ignore
068            }
069    
070            // Main loop
071            try {
072                while (true) {
073    
074                    // Read next line
075                    String line;
076                    try {
077                        line = Readline.readline(prompt);
078                    } catch (EOFException e) {
079                        break;
080                    }
081    
082                    // No input?
083                    if (line == null)
084                        continue;
085    
086                    // Update history
087                    Readline.addToHistory(line);
088    
089                    // Execute line
090                    if (!this.handleCommand(line))
091                        break;
092                }
093            } finally {
094    
095                // Save history
096                try {
097                    Readline.writeHistoryFile(historyFile);
098                } catch (Exception e) {
099                    // ignore
100                }
101    
102                // Clean up
103                Readline.cleanup();
104            }
105        }
106    
107        /**
108         * Callback used by {@link #commandLoop}.
109         *
110         * <p>
111         * The implementation in {@link MainClass} just returns {@code false}.
112         *
113         * @return true to continue reading the next command, false to exit
114         */
115        protected boolean handleCommand(String line) throws Exception {
116            return false;
117        }
118    
119        /**
120         * Display the usage message to standard error.
121         */
122        protected abstract void usageMessage();
123    
124        /**
125         * Print the usage message and exit with exit value 1.
126         */
127        protected void usageError() {
128            usageMessage();
129            System.exit(1);
130        }
131    
132        /**
133         * Emit an error message an exit with exit value 1.
134         */
135        protected final void errout(String message) {
136            System.err.println(getClass().getSimpleName() + ": " + message);
137            System.exit(1);
138        }
139    
140        /**
141         * Invokes {@link #run}, catching any exceptions thrown and exiting with a non-zero
142         * value if and only if an exception was caught.
143         * <p/>
144         * <p>
145         * The concrete class' {@code main()} method should invoke this method.
146         * </p>
147         */
148        protected void doMain(String[] args) {
149            int exitValue = 1;
150            try {
151                exitValue = run(args);
152            } catch (Throwable t) {
153                t.printStackTrace(System.err);
154            } finally {
155                System.exit(exitValue);
156            }
157        }
158    }
